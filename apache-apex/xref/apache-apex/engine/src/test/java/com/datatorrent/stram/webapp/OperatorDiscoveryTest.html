398   public static String[] getClassFileInClasspath()
399   {
400     String classpath = System.getProperty("java.class.path");
401     String[] paths = classpath.split(":");
402     List<String> fnames = new LinkedList<String>();
403     for (String cp : paths) {
404       File f = new File(cp);
405       if(!f.isDirectory()){
406         continue;
407       }
408       DirectoryScanner ds = new DirectoryScanner();
409       ds.setBasedir(f);
410       ds.setIncludes(new String[] { "**\\*.class" });
411       ds.scan();
412       for (String name : ds.getIncludedFiles()) {
413         fnames.add(new File(f, name).getAbsolutePath());
414       }
415 
416     }
417     return fnames.toArray(new String[]{});
418   }

420   private JSONObject getJSONProperty(JSONArray props, String name) throws JSONException
421   {
422     for (int i = 0; i < props.length(); i++) {
423       if(props.getJSONObject(i).get("name").equals(name)){
424         return props.getJSONObject(i);
425       }
426     }
427     return null;
428   }

523     private int size;
524     private String name;
525     private ArrayList<String> list;
526 
527     public int getSize()
528     {
529       return size;
530     }
531

535     }
536 
537     public String getName()
538     {
539       return name;
540     }
541 
542     public void setName(String name)
543     {
544       this.name = name;
545     }
546 
547     public ArrayList<String> getList()
548     {
549       return list;
550     }
551 
552     public void setList(ArrayList<String> list)
553     {
554       this.list = list;
555     }
556 
557     @Override
558     public int hashCode()
559     {
560       final int prime = 31;
561       int result = 1;
562       result = prime * result + ((list == null) ? 0 : list.hashCode());
563       result = prime * result + ((name == null) ? 0 : name.hashCode());
564       result = prime * result + size;
565       return result;
566     }
567 
568     @Override
569     public boolean equals(Object obj)
570     {
571       if (this == obj)
572         return true;
573       if (obj == null)
574         return false;
575       if (getClass() != obj.getClass())
576         return false;
577       Structured other = (Structured) obj;
578       if (list == null) {
579         if (other.list != null)
580           return false;
581       } else if (!list.equals(other.list))
582         return false;
583       if (name == null) {
584         if (other.name != null)
585           return false;
586       } else if (!name.equals(other.name))
587         return false;
588       if (size != other.size)
589         return false;
590       return true;
591     }

1035     Assert.assertTrue("Should contain property 'alias' !", s.contains("alias"));
1036     Assert.assertTrue("Shouldn't contain property 'getterOnly' !", !s.contains("getterOnly"));
1037     JSONObject jsonObj = new JSONObject(s);
1038 
1039     // create the json dag representation

1041     jsonPlan.put("streams", new JSONArray());
1042     JSONObject jsonOper = new JSONObject();
1043     jsonOper.put("name", "Test Operator");
1044     jsonOper.put("class", TestOperator.class.getName());

1051     // create logical plan from the json

1053     OperatorMeta om = lp.getOperatorMeta("Test Operator");
