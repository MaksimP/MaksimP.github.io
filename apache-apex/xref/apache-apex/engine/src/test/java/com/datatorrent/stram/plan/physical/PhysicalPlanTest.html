61      /**
62     * Stats listener for throughput based partitioning.
63     * Used when thresholds are configured on operator through attributes.
64     */
65    public static class PartitionLoadWatch implements StatsListener, java.io.Serializable
66    {
67      private static final Logger logger = LoggerFactory.getLogger(PartitionLoadWatch.class);
68      private static final long serialVersionUID = 201312231633L;
69      public long evalIntervalMillis = 30*1000;
70      private final long tpsMin;
71      private final long tpsMax;
72      private long lastEvalMillis;
73      private long lastTps = 0;
74  
75      private PartitionLoadWatch(long min, long max) {
76        this.tpsMin = min;
77        this.tpsMax = max;
78      }
79  
80      protected LoadIndicator getLoadIndicator(int operatorId, long tps) {
81        if ((tps < tpsMin && lastTps != 0) || tps > tpsMax) {
82          lastTps = tps;
83          if (tps < tpsMin) {
84            return new LoadIndicator(-1, String.format("Tuples per second %d is less than the minimum %d", tps, tpsMin));
85          }
86          else {
87            return new LoadIndicator(1, String.format("Tuples per second %d is greater than the maximum %d", tps, tpsMax));
88          }
89        }
90        lastTps = tps;
91        return new LoadIndicator(0, null);
92      }
93  
94      @Override
95      public Response processStats(BatchedOperatorStats status)
96      {
97  
98        long tps = status.getTuplesProcessedPSMA();
99  
100       if(tps == 0L) {
101         tps = status.getTuplesEmittedPSMA();
102       }
103 
104       Response rsp = new Response();
105       LoadIndicator loadIndicator = getLoadIndicator(status.getOperatorId(), tps);
106       rsp.loadIndicator = loadIndicator.indicator;
107       if (rsp.loadIndicator != 0) {
108         if (lastEvalMillis < (System.currentTimeMillis() - evalIntervalMillis)) {
109           lastEvalMillis = System.currentTimeMillis();
110           logger.debug("Requesting repartitioning {} {}", rsp.loadIndicator, tps);
111           rsp.repartitionRequired = true;
112           rsp.repartitionNote = loadIndicator.note;
113         }
114       }
115       return rsp;
116     }
117 
118   }
