19  package com.datatorrent.stram.util;

21  import java.io.IOException;
22  import java.net.JarURLConnection;
23  import java.net.URL;
24  import java.net.URLConnection;
25  import java.util.Enumeration;
26  import java.util.Properties;
27  import java.util.jar.Attributes;
28  import java.util.jar.Manifest;
29  
30  import org.apache.hadoop.classification.InterfaceAudience;
31  import org.apache.hadoop.classification.InterfaceStability;
32  
33  /**
34   * This class finds the build version info from the jar file.
35   *
36   * @since 0.3.2
37   */
38  @InterfaceAudience.Private
39  @InterfaceStability.Unstable
40  public class VersionInfo {
41  
42    private static String version = "Unknown";
43    private static String user = "Unknown";
44    private static String date = "Unknown";
45    private static String revision = "Unknown";
46  
47    static {
48      try {
49        URL res = VersionInfo.class.getResource(VersionInfo.class.getSimpleName() + ".class");
50        URLConnection conn = res.openConnection();
51        if (conn instanceof JarURLConnection) {
52          Manifest mf = ((JarURLConnection) conn).getManifest();
53          Attributes mainAttribs = mf.getMainAttributes();
54          String builtBy = mainAttribs.getValue("Built-By");
55          if(builtBy != null) {
56            VersionInfo.user = builtBy;
57          }
58        }
59  
60        Enumeration<URL> resources = VersionInfo.class.getClassLoader().getResources("META-INF/maven/org.apache.apex/apex-engine/pom.properties");
61        while (resources.hasMoreElements()) {
62          Properties pomInfo = new Properties();
63          pomInfo.load(resources.nextElement().openStream());
64          String v = pomInfo.getProperty("version", "unknown");
65          VersionInfo.version = v;
66        }
67  
68        resources = VersionInfo.class.getClassLoader().getResources("dt-git.properties");
69        while (resources.hasMoreElements()) {
70          Properties gitInfo = new Properties();
71          gitInfo.load(resources.nextElement().openStream());
72          String commitAbbrev = gitInfo.getProperty("git.commit.id.abbrev", "unknown");
73          String branch = gitInfo.getProperty("git.branch", "unknown");
74          VersionInfo.revision = "rev: " + commitAbbrev + " branch: " + branch;
75          VersionInfo.date = gitInfo.getProperty("git.build.time", VersionInfo.date);
76          VersionInfo.user = gitInfo.getProperty("git.build.user.name", VersionInfo.user);
77          break;
78        }
79  
80      }
81      catch (IOException e) {
82        org.slf4j.LoggerFactory.getLogger(VersionInfo.class).error("Failed to read version info", e);
83      }
84    }
85  
86    /**
87     * Get the version.
88     *
89     * @return the version string, e.g. "0.1.1-SNAPSHOT"
90     */
91    public static String getVersion() {
92      return version;
93    }
94  
95    /**
96     * The date of the build.
97     *
98     * @return the compilation date
99     */
100   public static String getDate() {
101     return date;
102   }
103 
104   /**
105    * The user that made the build.
106    *
107    * @return the username of the user
108    */
109   public static String getUser() {
110     return user;
111   }
112 
113   /**
114    * Get the SCM revision number
115    * @return the revision number, eg. "451451"
116    */
117   public static String getRevision() {
118     return revision;
119   }
120 
121   /**
122    * Returns the buildVersion which includes version, revision, user and date.
123    */
124   public static String getBuildVersion() {
125     return VersionInfo.getVersion() + " from " + VersionInfo.getRevision() + " by " + VersionInfo.getUser() + " on " + VersionInfo.getDate();
126   }
127 
128   /**
129    * Compares two version strings.
130    *
131    * @param str1 a string of ordinal numbers separated by decimal points.
132    * @param str2 a string of ordinal numbers separated by decimal points.
133    * @return The result is a negative integer if str1 is _numerically_ less than str2. The result is a positive integer
134    * if str1 is _numerically_ greater than str2. The result is zero if the strings are _numerically_ equal.
135    */
136   public static int compare(String str1, String str2)
137   {
138     String[] vals1 = normalizeVersion(str1).split("\\.");
139     String[] vals2 = normalizeVersion(str2).split("\\.");
140     int i = 0;
141     while (i < vals1.length && i < vals2.length && vals1[i].equals(vals2[i])) {
142       i++;
143     }
144     if (i < vals1.length && i < vals2.length) {
145       if (vals1[i].isEmpty()) {
146         vals1[i] = "0";
147       }
148       if (vals2[i].isEmpty()) {
149         vals2[i] = "0";
150       }
151       int diff = Integer.valueOf(vals1[i]).compareTo(Integer.valueOf(vals2[i]));
152       return Integer.signum(diff);
153     } else {
154       return Integer.signum(vals1.length - vals2.length);
155     }
156   }
157 
158   private static String normalizeVersion(String ver)
159   {
160     for (int i = 0; i < ver.length(); i++) {
161       char c = ver.charAt(i);
162       if (!Character.isDigit(c) && c != '.') {
163         return ver.substring(0, i);
164       }
165     }
166     return ver;
167   }
168 
169   @SuppressWarnings("UseOfSystemOutOrSystemErr")
170   public static void main(String[] args) {
171     System.out.println("Malhar " + getVersion());
172     System.out.println("Revision " + getRevision());
173     System.out.println("Compiled by " + getUser() + " on " + getDate());
174   }
