184     protected static void initialize()
185     {
186       STRAM.setChildren(Sets.newHashSet(APPLICATION, TEMPLATE));
187       APPLICATION.setChildren(Sets.newHashSet(GATEWAY, OPERATOR, STREAM));
188       OPERATOR.setChildren(Sets.newHashSet(PORT));
189       PORT.setChildren(Sets.newHashSet(UNIFIER));
190 
191       STRAM_ELEMENT_TO_CONF_ELEMENT.clear();
192 
193       //Initialize StramElement to ConfElement
194       for (ConfElement confElement: ConfElement.values()) {
195         STRAM_ELEMENT_TO_CONF_ELEMENT.put(confElement.getStramElement(), confElement);
196 
197         for (StramElement sElement: confElement.getAllRelatedElements()) {
198           STRAM_ELEMENT_TO_CONF_ELEMENT.put(sElement, confElement);
199         }
200       }
201 
202       //Initialize attributes
203       for (ConfElement confElement: ConfElement.values()) {
204         if (confElement.getParent() == null) {
205           continue;
206         }
207 
208         setAmbiguousAttributes(confElement);
209       }
210 
211       // build context to conf element map
212       CONTEXT_TO_CONF_ELEMENT.clear();
213 
214       for (ConfElement confElement: ConfElement.values()) {
215         CONTEXT_TO_CONF_ELEMENT.put(confElement.getContextClass(), confElement);
216       }
217 
218       //Check if all the context classes are accounted for
219       Set<Class<? extends Context>> confElementContextClasses = Sets.newHashSet();
220 
221       for (ConfElement confElement: ConfElement.values()) {
222         if (confElement.getContextClass() == null) {
223           continue;
224         }
225 
226         confElementContextClasses.add(confElement.getContextClass());
227       }
228 
229       if (!ContextUtils.CONTEXT_CLASSES.equals(confElementContextClasses)) {
230         throw new IllegalStateException("All the context classes " + ContextUtils.CONTEXT_CLASSES + " found in "
231                                         + Context.class + " are not used by ConfElements " + confElementContextClasses);
232       }
233     }

235     /**
236      * This is a recursive method to initialize the ambiguous elements for each
237      * {@link ConfElement}.
238      *
239      * @param element The current {@link ConfElement} at which to start initializing
240      * the ambiguous elements.
241      * @return The set of all simple attribute names encountered up to this point.
242      */
243     public static Set<String> setAmbiguousAttributes(ConfElement element)
244     {
245       Set<String> ambiguousAttributes = Sets.newHashSet();
246       Set<String> allChildAttributes = Sets.newHashSet(element.getContextAttributes());
247 
248       for (ConfElement childElement: element.getChildren()) {
249         Set<String> allAttributes = setAmbiguousAttributes(childElement);
250         ambiguousAttributes.addAll(childElement.getAmbiguousAttributes());
251 
252         @SuppressWarnings("unchecked")
253         Set<String> intersection = (Set<String>)Sets.newHashSet(CollectionUtils.intersection(allChildAttributes,
254                                                                                              allAttributes));
255         ambiguousAttributes.addAll(intersection);
256         allChildAttributes.addAll(allAttributes);
257       }
258 
259       element.setAmbiguousAttributes(ambiguousAttributes);
260       element.setAllChildAttributes(allChildAttributes);
261 
262       return allChildAttributes;
263     }

457     /**
458      * Creates a list of {@link StramElement}s which represent the path from the current {@link Conf} type to
459      * a parent {@link Conf} type. This path includes the current {@link Conf} type as well as the parent.
460      *
461      * @param child The current {@link Conf} type.
462      * @param parent The parent {@link Conf} type.
463      * @return A path from the current {@link Conf} type to a parent {@link Conf} type, which includes the current and parent
464      * {@link Conf} types.
465      */
466     public static List<StramElement> getPathFromChildToParentInclusive(StramElement child, StramElement parent)
467     {
468       ConfElement confElement = STRAM_ELEMENT_TO_CONF_ELEMENT.get(child);
469 
470       if (confElement == null) {
471         throw new IllegalArgumentException(child + " does not represent a valid configuration type.");
472       }
473 
474       List<StramElement> path = Lists.newArrayList();
475 
476       if (child == parent) {
477         path.add(child);
478         return path;
479       }
480 
481       for (; confElement != null; confElement = confElement.getParent()) {
482         path.add(confElement.getStramElement());
483 
484         if (confElement.getStramElement() == parent) {
485           break;
486         }
487       }
488 
489       if (path.get(path.size() - 1) != parent) {
490         throw new IllegalArgumentException(parent + " is not a valid parent of " + child);
491       }
492 
493       return path;
494     }

512     /**
513      * This method searches the current {@link ConfElement} and its children to find a {@link ConfElement}
514      * that contains the given simple {@link Attribute} name.
515      *
516      * @param current The current {@link ConfElement}.
517      * @param simpleAttributeName The simple {@link Attribute} name to search for.
518      * @return The {@link ConfElement} that contains the given attribute, or null if no {@link ConfElement} contains
519      * the given attribute.
520      */
521     public static ConfElement findConfElementWithAttribute(ConfElement current, String simpleAttributeName)
522     {
523       if (current.getContextAttributes().contains(simpleAttributeName)) {
524         return current;
525       }
526 
527       for (ConfElement childConfElement: current.getChildren()) {
528         ConfElement result = findConfElementWithAttribute(childConfElement,
529                                                           simpleAttributeName);
530 
531         if (result != null) {
532           return result;
533         }
534       }
535 
536       return null;
537     }

540     {
541       //Figure out what configurations need to be added to hold this attribute
542       List<StramElement> path = ConfElement.getPathFromParentToChildInclusive(childConfElement.getStramElement(),
543                                                                               parentConf.getConfElement().getStramElement());

545       for (int pathIndex = 1; pathIndex < path.size(); pathIndex++) {
546         LOG.debug("Adding conf");
547         StramElement pathElement = path.get(pathIndex);
548         //Add the configurations we need to hold this attribute
549         parentConf = addConf(pathElement, WILDCARD, parentConf);
550       }
551 
552       return parentConf;
553     }

557   /**
558    * Utility class that holds methods for handling {@link Context} classes.
559    */
560   @SuppressWarnings("unchecked")
561   protected static class ContextUtils
562   {
563     private static final Map<String, Type> ATTRIBUTES_TO_TYPE = Maps.newHashMap();
564     public static final Map<Class<? extends Context>, Set<String>> CONTEXT_CLASS_TO_ATTRIBUTES = Maps.newHashMap();
565     public static final Set<Class<? extends Context>> CONTEXT_CLASSES = Sets.newHashSet();
566     public static final Map<Class<? extends Context>, Map<String, Attribute<?>>> CONTEXT_TO_ATTRIBUTE_NAME_TO_ATTRIBUTE = Maps.newHashMap();
567 
568     static {
569       initialize();
570     }
571 
572     @VisibleForTesting
573     protected static void initialize()
574     {
575       CONTEXT_CLASSES.clear();
576 
577       for (Class<?> clazz: Context.class.getDeclaredClasses()) {
578         if (!Context.class.isAssignableFrom(clazz)) {
579           continue;
580         }
581 
582         CONTEXT_CLASSES.add((Class<? extends Context>)clazz);
583       }
584 
585       buildAttributeMaps(CONTEXT_CLASSES);
586     }
587 
588     @VisibleForTesting
589     protected static void buildAttributeMaps(Set<Class<? extends Context>> contextClasses)
590     {
591       CONTEXT_CLASS_TO_ATTRIBUTES.clear();
592       CONTEXT_TO_ATTRIBUTE_NAME_TO_ATTRIBUTE.clear();
593       ATTRIBUTES_TO_TYPE.clear();
594 
595       for (Class<? extends Context> contextClass: contextClasses) {
596         Set<String> contextAttributes = Sets.newHashSet();
597 
598         Field[] fields = contextClass.getDeclaredFields();
599 
600         for (Field field: fields) {
601           if (!Attribute.class.isAssignableFrom(field.getType())) {
602             continue;
603           }
604 
605           Type fieldType = ((ParameterizedType) field.getGenericType()).getActualTypeArguments()[0];
606           contextAttributes.add(field.getName());
607 
608           Type existingType = ATTRIBUTES_TO_TYPE.get(field.getName());
609 
610           if (existingType != null && !existingType.equals(fieldType)) {
611             throw new ValidationException("The attribute " + field.getName() +
612                                           " is defined with two different types in two different context classes: " +
613                                           fieldType + " and " + existingType + "\n" +
614                                           "Attributes with the same name are required to have the same type accross all Context classes.");
615           }
616 
617           ATTRIBUTES_TO_TYPE.put(field.getName(), fieldType);
618         }
619 
620         CONTEXT_CLASS_TO_ATTRIBUTES.put(contextClass, contextAttributes);
621       }
622 
623       for (Class<? extends Context> contextClass: contextClasses) {
624         Map<String, Attribute<?>> simpleAttributeNameToAttribute = Maps.newHashMap();
625         CONTEXT_TO_ATTRIBUTE_NAME_TO_ATTRIBUTE.put(contextClass, simpleAttributeNameToAttribute);
626 
627         Set<Attribute<Object>> attributes = AttributeInitializer.getAttributes(contextClass);
628 
629         LOG.debug("context class {} and attributes {}", contextClass, attributes);
630 
631         for (Attribute<Object> attribute: attributes) {
632           simpleAttributeNameToAttribute.put(AttributeParseUtils.getSimpleName(attribute), attribute);
633         }
634       }
635     }

1013       for (Map.Entry<String, T> entry : elChildren.entrySet()) {
1014         String key = entry.getKey();

1029           }

1031             match = true;
1032           }
1033           if (name.equals(key)) {
1034             exact = true;
1035           }
1036         }
1037         // There will be a better match preference order
1038         if (match) {
1039           if (!exact) {
1040             childConfs.add(entry.getValue());
1041           } else {

1043           }

1734   /**
1735    * This method is used to parse an Apex property name.
1736    * @param keys The keys into which an Apex property is split into.
1737    * @param index The current index that the parser is on for processing the property name.
1738    * @param propertyName The original unsplit Apex property name.
1739    * @param propertyValue The value corresponding to the Apex property.
1740    * @param conf The current {@link Conf} to add properties to.
1741    */
1742   private void parseStramPropertyTokens(String[] keys, int index, String propertyName, String propertyValue, Conf conf) {
1743     if (index < keys.length) {
1744       String key = keys[index];
1745       StramElement element = getElement(key, conf);
1746       if ((element == null) && conf.ignoreUnknownChildren()) {
1747         return;
1748       }
1749       if ((element == StramElement.APPLICATION) || (element == StramElement.OPERATOR) || (element == StramElement.STREAM)
1750               || (element == StramElement.PORT) || (element == StramElement.INPUT_PORT) || (element == StramElement.OUTPUT_PORT)
1751               || (element == StramElement.TEMPLATE)) {
1752         parseAppElement(index, keys, element, conf, propertyName, propertyValue);
1753       } else if (element == StramElement.GATEWAY) {
1754         parseGatewayElement(element, conf, keys, index, propertyName, propertyValue);
1755       } else if ((element == StramElement.UNIFIER)) {
1756         parseUnifierElement(element, conf, keys, index, propertyName, propertyValue);
1757       } else if ((element == StramElement.ATTR) || ((element == null) && (conf.getDefaultChildElement() == StramElement.ATTR))) {
1758         parseAttributeElement(element, keys, index, conf, propertyValue, propertyName);
1759       } else if ((element == StramElement.PROP) || ((element == null) && (conf.getDefaultChildElement() == StramElement.PROP))) {
1760         parsePropertyElement(element, keys, index, conf, propertyValue, propertyName);
1761       } else if (element != null) {
1762         conf.parseElement(element, keys, index, propertyValue);
1763       }
1764     }

2000    * Populate the logical plan structure from properties.
2001    * @param dag
2002    */
2003   public void populateDAG(LogicalPlan dag)
2004   {
2005     Configuration pconf = new Configuration(conf);
2006     for (final String propertyName : this.properties.stringPropertyNames()) {
2007       String propertyValue = this.properties.getProperty(propertyName);
2008       pconf.setIfUnset(propertyName, propertyValue);
2009     }
2010 
2011     AppConf appConf = this.stramConf.getChild(WILDCARD, StramElement.APPLICATION);
2012     if (appConf == null) {
2013       LOG.warn("Application configuration not found. Probably an empty app.");
2014       return;
2015     }
2016 
2017     Map<String, OperatorConf> operators = appConf.getChildren(StramElement.OPERATOR);
2018 
2019     Map<OperatorConf, Operator> nodeMap = Maps.newHashMapWithExpectedSize(operators.size());
2020     // add all operators first
2021     for (Map.Entry<String, OperatorConf> nodeConfEntry : operators.entrySet()) {
2022       OperatorConf nodeConf = nodeConfEntry.getValue();
2023       if (!WILDCARD.equals(nodeConf.id)) {
2024         Class<? extends Operator> nodeClass = StramUtils.classForName(nodeConf.getClassNameReqd(), Operator.class);
2025         String optJson = nodeConf.getProperties().get(nodeClass.getName());
2026         Operator nd = null;
2027         try {
2028           if (optJson != null) {
2029             // if there is a special key which is the class name, it means the operator is serialized in json format
2030             ObjectMapper mapper = ObjectMapperFactory.getOperatorValueDeserializer();
2031             nd = mapper.readValue("{\"" + nodeClass.getName() + "\":" + optJson + "}", nodeClass);
2032             dag.addOperator(nodeConfEntry.getKey(), nd);
2033           } else {
2034             nd = dag.addOperator(nodeConfEntry.getKey(), nodeClass);
2035           }
2036           setOperatorProperties(nd, nodeConf.getProperties());
2037         } catch (IOException e) {
2038           throw new IllegalArgumentException("Error setting operator properties " + e.getMessage(), e);
2039         }
2040         nodeMap.put(nodeConf, nd);
2041       }
2042     }
2043 
2044     Map<String, StreamConf> streams = appConf.getChildren(StramElement.STREAM);
2045 
2046     // wire operators
2047     for (Map.Entry<String, StreamConf> streamConfEntry : streams.entrySet()) {
2048       StreamConf streamConf = streamConfEntry.getValue();
2049       DAG.StreamMeta sd = dag.addStream(streamConfEntry.getKey());
2050       sd.setLocality(streamConf.getLocality());
2051 
2052       String schemaClassName = streamConf.properties.getProperty(STREAM_SCHEMA);
2053       Class<?> schemaClass = null;
2054       if (schemaClassName != null) {
2055         schemaClass = StramUtils.classForName(schemaClassName, Object.class);
2056       }
2057 
2058       if (streamConf.sourceNode != null) {
2059         String portName = null;
2060         for (Map.Entry<String, StreamConf> e : streamConf.sourceNode.outputs.entrySet()) {
2061           if (e.getValue() == streamConf) {
2062             portName = e.getKey();
2063           }
2064         }
2065         Operator sourceDecl = nodeMap.get(streamConf.sourceNode);
2066         Operators.PortMappingDescriptor sourcePortMap = new Operators.PortMappingDescriptor();
2067         Operators.describe(sourceDecl, sourcePortMap);
2068         sd.setSource(sourcePortMap.outputPorts.get(portName).component);
2069 
2070         if (schemaClass != null) {
2071           dag.setOutputPortAttribute(sourcePortMap.outputPorts.get(portName).component, PortContext.TUPLE_CLASS, schemaClass);
2072         }
2073       }
2074 
2075       for (OperatorConf targetNode : streamConf.targetNodes) {
2076         String portName = null;
2077         for (Map.Entry<String, StreamConf> e : targetNode.inputs.entrySet()) {
2078           if (e.getValue() == streamConf) {
2079             portName = e.getKey();
2080           }
2081         }
2082         Operator targetDecl = nodeMap.get(targetNode);
2083         Operators.PortMappingDescriptor targetPortMap = new Operators.PortMappingDescriptor();
2084         Operators.describe(targetDecl, targetPortMap);
2085         sd.addSink(targetPortMap.inputPorts.get(portName).component);
2086 
2087         if (schemaClass != null) {
2088           dag.setInputPortAttribute(targetPortMap.inputPorts.get(portName).component, PortContext.TUPLE_CLASS, schemaClass);
2089         }
2090       }
2091     }

2179     // direct settings
2180     // Apply the configurations in reverse order since the higher priority ones are at the beginning
