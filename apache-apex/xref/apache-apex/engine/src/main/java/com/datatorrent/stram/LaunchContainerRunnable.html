90      this.container = lcontainer;

97    private void setClasspath(Map<String, String> env)
98    {
99      // add localized application jar files to classpath
100     // At some point we should not be required to add
101     // the hadoop specific classpaths to the env.
102     // It should be provided out of the box.
103     // For now setting all required classpaths including
104     // the classpath to "." for the application jar
105     StringBuilder classPathEnv = new StringBuilder("./*");
106     String classpath = nmClient.getConfig().get(YarnConfiguration.YARN_APPLICATION_CLASSPATH);
107     for (String c : StringUtils.isBlank(classpath) ? YarnConfiguration.DEFAULT_YARN_APPLICATION_CLASSPATH : classpath.split(",")) {
108       if (c.equals("$HADOOP_CLIENT_CONF_DIR")) {
109         // SPOI-2501
110         continue;
111       }
112       classPathEnv.append(':');
113       classPathEnv.append(c.trim());
114     }
115     classPathEnv.append(":."); // include log4j.properties, if any
116 
117     env.put("CLASSPATH", classPathEnv.toString());
118     LOG.info("CLASSPATH: {}", classPathEnv);
119   }

121   public static void addFileToLocalResources(final String name, final FileStatus fileStatus, final LocalResourceType type, final Map<String, LocalResource> localResources)
122   {
123     final LocalResource localResource = LocalResource.newInstance(ConverterUtils.getYarnUrlFromPath(fileStatus.getPath()),
124             type, LocalResourceVisibility.APPLICATION, fileStatus.getLen(), fileStatus.getModificationTime());
125     localResources.put(name, localResource);
126   }

128   public static void addFilesToLocalResources(LocalResourceType type, String commaSeparatedFileNames, Map<String, LocalResource> localResources, FileSystem fs) throws IOException
129   {
130     String[] files = StringUtils.splitByWholeSeparator(commaSeparatedFileNames, StramClient.LIB_JARS_SEP);
131     for (String file : files) {
132       final Path dst = new Path(file);
133       addFileToLocalResources(dst.getName(), fs.getFileStatus(dst), type, localResources);
134     }
135   }

137   /**
138    * Connects to CM, sets up container launch context and eventually dispatches the container start request to the CM.
139    */
140   @Override
141   public void run()
142   {
143     LOG.info("Setting up container launch context for containerid={}", container.getId());
144     ContainerLaunchContext ctx = Records.newRecord(ContainerLaunchContext.class);
145 
146     setClasspath(containerEnv);
147     try {
148       // propagate to replace node managers user name (effective in non-secure mode)
149       containerEnv.put("HADOOP_USER_NAME", UserGroupInformation.getLoginUser().getUserName());
150     } catch (Exception e) {
151       LOG.error("Failed to retrieve principal name", e);
152     }
153     // Set the environment
154     ctx.setEnvironment(containerEnv);
155     ctx.setTokens(tokens);
156 
157     // Set the local resources
158     Map<String, LocalResource> localResources = new HashMap<String, LocalResource>();
159 
160     // add resources for child VM
161     try {
162       // child VM dependencies
163       FileSystem fs = StramClientUtils.newFileSystemInstance(nmClient.getConfig());
164       try {
165         addFilesToLocalResources(LocalResourceType.FILE, dag.getAttributes().get(LogicalPlan.LIBRARY_JARS), localResources, fs);
166         String archives = dag.getAttributes().get(LogicalPlan.ARCHIVES);
167         if (archives != null) {
168           addFilesToLocalResources(LocalResourceType.ARCHIVE, archives, localResources, fs);
169         }
170         ctx.setLocalResources(localResources);
171       }
172       finally {
173         fs.close();
174       }
175     }
176     catch (IOException e) {
177       LOG.error("Failed to prepare local resources.", e);
178       return;
179     }
180 
181     // Set the necessary command to execute on the allocated container
182     List<CharSequence> vargs = getChildVMCommand(container.getId().toString());
183 
184     // Get final command
185     StringBuilder command = new StringBuilder(1024);
186     for (CharSequence str : vargs) {
187       command.append(str).append(" ");
188     }
189     LOG.info("Launching on node: {} command: {}", container.getNodeId(), command);
190 
191     List<String> commands = new ArrayList<String>();
192     commands.add(command.toString());
193     ctx.setCommands(commands);
194 
195     nmClient.startContainerAsync(container, ctx);
196   }

198   /**
199    * Build the command to launch the child VM in the container
200    *
201    * @param jvmID
202    * @return
203    */
204   public List<CharSequence> getChildVMCommand(String jvmID)
205   {
206 
207     List<CharSequence> vargs = new ArrayList<CharSequence>(8);
208 
209     if (!StringUtils.isBlank(System.getenv(Environment.JAVA_HOME.key()))) {
210       // node manager provides JAVA_HOME
211       vargs.add(Environment.JAVA_HOME.$() + "/bin/java");
212     }
213     else {
214       vargs.add("java");
215     }
216 
217     String jvmOpts = dag.getAttributes().get(LogicalPlan.CONTAINER_JVM_OPTIONS);
218     if (jvmOpts == null) {
219       if (dag.isDebug()) {
220         vargs.add(JAVA_REMOTE_DEBUG_OPTS);
221       }
222     }
223     else {
224       Map<String, String> params = new HashMap<String, String>();
225       params.put("applicationId", Integer.toString(container.getId().getApplicationAttemptId().getApplicationId().getId()));
226       params.put("containerId", Integer.toString(container.getId().getId()));
227       StrSubstitutor sub = new StrSubstitutor(params, "%(", ")");
228       vargs.add(sub.replace(jvmOpts));
229       if (dag.isDebug() && !jvmOpts.contains("-agentlib:jdwp=")) {
230         vargs.add(JAVA_REMOTE_DEBUG_OPTS);
231       }
232     }
233 
234     List<DAG.OperatorMeta> operatorMetaList = Lists.newArrayList();
235     int bufferServerMemory = 0;
236     for(PTOperator operator: sca.getContainer().getOperators()){
237       bufferServerMemory += operator.getBufferServerMemory();
238       operatorMetaList.add(operator.getOperatorMeta());
239     }
240     Context.ContainerOptConfigurator containerOptConfigurator = dag.getAttributes().get(LogicalPlan.CONTAINER_OPTS_CONFIGURATOR);
241     jvmOpts = containerOptConfigurator.getJVMOptions(operatorMetaList);
242     jvmOpts = parseJvmOpts(jvmOpts, ((long) bufferServerMemory) * MB_TO_B);
243     LOG.info("Jvm opts {} for container {}",jvmOpts,container.getId());
244     vargs.add(jvmOpts);
245 
246     Path childTmpDir = new Path(Environment.PWD.$(),
247                                 YarnConfiguration.DEFAULT_CONTAINER_TEMP_DIR);
248     vargs.add(String.format("-D%s=%s", StreamingContainer.PROP_APP_PATH, dag.assertAppPath()));
249     vargs.add("-Djava.io.tmpdir=" + childTmpDir);
250     vargs.add(String.format("-D%scid=%s", StreamingApplication.DT_PREFIX, jvmID));
251     vargs.add("-Dhadoop.root.logger=" + (dag.isDebug() ? "DEBUG" : "INFO") + ",RFA");
252     vargs.add("-Dhadoop.log.dir=" + ApplicationConstants.LOG_DIR_EXPANSION_VAR);
253 
254     String loggersLevel = System.getProperty(DTLoggerFactory.DT_LOGGERS_LEVEL);
255     if (loggersLevel != null) {
256       vargs.add(String.format("-D%s=%s", DTLoggerFactory.DT_LOGGERS_LEVEL, loggersLevel));
257     }
258     // Add main class and its arguments
259     vargs.add(StreamingContainer.class.getName());  // main of Child
260 
261     vargs.add("1>" + ApplicationConstants.LOG_DIR_EXPANSION_VAR + "/stdout");
262     vargs.add("2>" + ApplicationConstants.LOG_DIR_EXPANSION_VAR + "/stderr");
263 
264     // Final commmand
265     StringBuilder mergedCommand = new StringBuilder(256);
266     for (CharSequence str : vargs) {
267       mergedCommand.append(str).append(" ");
268     }
269     List<CharSequence> vargsFinal = new ArrayList<CharSequence>(1);
270     vargsFinal.add(mergedCommand.toString());
271     return vargsFinal;
272 
273   }

275   private String parseJvmOpts(String jvmOpts, long memory)
276   {
277     String xmx = "-Xmx";
278     StringBuilder builder = new StringBuilder();
279     if (jvmOpts != null && jvmOpts.length() > 1) {
280       String[] splits = jvmOpts.split("(\\s+)");
281       boolean foundProperty = false;
282       for (String split : splits) {
283         if (split.startsWith(xmx)) {
284           foundProperty = true;
285           long heapSize = Long.valueOf(split.substring(xmx.length()));
286           heapSize += memory;
287           builder.append(xmx).append(heapSize).append(" ");
288         }
289         else {
290           builder.append(split).append(" ");
291         }
292       }
293       if (!foundProperty) {
294         builder.append(xmx).append(memory);
295       }
296     }
297     return builder.toString();
298   }

300   public static ByteBuffer getTokens(StramDelegationTokenManager delegationTokenManager, InetSocketAddress heartbeatAddress) throws IOException
301   {
302     if (UserGroupInformation.isSecurityEnabled()) {
303       UserGroupInformation ugi = UserGroupInformation.getLoginUser();
304       StramDelegationTokenIdentifier identifier = new StramDelegationTokenIdentifier(new Text(ugi.getUserName()), new Text(""), new Text(""));
305       String service = heartbeatAddress.getAddress().getHostAddress() + ":" + heartbeatAddress.getPort();
306       Token<StramDelegationTokenIdentifier> stramToken = new Token<StramDelegationTokenIdentifier>(identifier, delegationTokenManager);
307       stramToken.setService(new Text(service));
308       return getTokens(ugi, stramToken);
309     }
310     return null;
311   }

313   public static ByteBuffer getTokens(UserGroupInformation ugi, Token<StramDelegationTokenIdentifier> delegationToken)
314   {
315     try {
316       Collection<Token<? extends TokenIdentifier>> tokens = ugi.getCredentials().getAllTokens();
317       Credentials credentials = new Credentials();
318       for (Token<? extends TokenIdentifier> token : tokens) {
319         if (!token.getKind().equals(AMRMTokenIdentifier.KIND_NAME)) {
320           credentials.addToken(token.getService(), token);
321           LOG.debug("Passing container token {}", token);
322         }
323       }
324       credentials.addToken(delegationToken.getService(), delegationToken);
325       DataOutputBuffer dataOutput = new DataOutputBuffer();
326       credentials.writeTokenStorageToStream(dataOutput);
327       byte[] tokenBytes = dataOutput.getData();
328       ByteBuffer cTokenBuf = ByteBuffer.wrap(tokenBytes);
329       return cTokenBuf.duplicate();
330     }
331     catch (IOException e) {
332       throw new RuntimeException("Error generating delegation token", e);
333     }
334   }
