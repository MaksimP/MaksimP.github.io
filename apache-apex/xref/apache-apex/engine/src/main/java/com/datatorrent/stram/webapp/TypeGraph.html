117   private static boolean isAncestor(TypeGraphVertex typeTgv, TypeGraphVertex tgv)
118   {
119     if (tgv == typeTgv) {
120       return true;
121     }
122     if ((tgv.ancestors == null || tgv.ancestors.size() == 0)) {
123       return false;
124     }
125     for (TypeGraphVertex vertex : tgv.ancestors) {
126       if (isAncestor(typeTgv, vertex))
127         return true;
128     }
129     return false;
130   }

161     {
162 
163       ArrayList<String> l = new ArrayList<String>();
164       l.add(Class.class.getName());
165       Iterator<FromStringDeserializer<?>> iter = FromStringDeserializer.all().iterator();

171       return l.toArray(a);

183     public static UI_TYPE getEnumFor(String clazzName, Map<String, TypeGraphVertex> typeGraph)
184     {
185       TypeGraphVertex tgv = typeGraph.get(clazzName);
186       if (tgv == null) {
187         return null;
188       }
189       for (UI_TYPE type : UI_TYPE.values()) {
190         for (String assignable : type.allAssignableTypes) {
191           TypeGraphVertex typeTgv = typeGraph.get(assignable);
192           if (typeTgv == null) {
193             continue;
194           }
195           if (isAncestor(typeTgv, tgv)) {
196             return type;
197           }
198         }
199       }
200       return null;
201     }

203     public static UI_TYPE getEnumFor(TypeGraphVertex tgv)
204     {
205       List<String> allTypes = TypeGraph.getAllAncestors(tgv, true);
206       for (UI_TYPE type : UI_TYPE.values()) {
207         for (String assignable : type.allAssignableTypes){
208           if(allTypes.contains(assignable)){
209             return type;
210           }
211         }
212       }
213       return null;
214     }

245   private TypeGraphVertex addNode(InputStream input, String resName) throws IOException
246   {
247     try {
248       
249       ClassReader reader = new ClassReader(input);
250       ClassNode classN = new ClassNodeType();
251       reader.accept(classN, ClassReader.SKIP_CODE);
252       CompactClassNode ccn = CompactUtil.compactClassNode(classN);
253       String typeName = classN.name.replace('/', '.');
254       TypeGraphVertex tgv = null;
255       TypeGraphVertex ptgv = null;
256       if (typeGraph.containsKey(typeName)) {
257         tgv = typeGraph.get(typeName);
258         tgv.setClassNode(ccn);
259         tgv.setJarName(resName); // If tgv was already populated for superclass/interface, jar name needs to be updated 
260       } else {
261         tgv = new TypeGraphVertex(typeName, resName, ccn);
262         typeGraph.put(typeName, tgv);
263       }
264       String immediateP = reader.getSuperName();
265       if (immediateP != null) {
266         immediateP = immediateP.replace('/', '.');
267         ptgv = typeGraph.get(immediateP);
268         if (ptgv == null) {
269           ptgv = new TypeGraphVertex(immediateP, resName);
270           typeGraph.put(immediateP, ptgv);
271         }
272         tgv.ancestors.add(ptgv);
273         ptgv.descendants.add(tgv);
274       }
275       if (reader.getInterfaces() != null) {
276         for (String iface : reader.getInterfaces()) {
277           iface = iface.replace('/', '.');
278           ptgv = typeGraph.get(iface);
279           if (ptgv == null) {
280             ptgv = new TypeGraphVertex(iface, resName);
281             typeGraph.put(iface, ptgv);
282           }
283           tgv.ancestors.add(ptgv);
284           ptgv.descendants.add(tgv);
285         }
286       }
287 
288       updateInstantiableDescendants(tgv);
289       return tgv;
290     } finally {
291       if (input != null) {
292         input.close();
293       }
294     }
295   }

313   public void updatePortTypeInfoInTypeGraph(Map<String, JarFile> openJarFiles,
314       Map<String, File> openClassFiles, TypeGraphVertex tgv) {
315     if (tgv == null)
316       return;
317 
318     for (TypeGraphVertex operator : tgv.descendants) {
319       try {
320         String path = operator.getJarName();
321         JarFile jar = openJarFiles.get(path);
322         if (jar != null) {
323           String jarEntryName = operator.getClassNode().getName()
324               .replace('.', '/')
325               + ".class";
326           JarEntry jarEntry = jar.getJarEntry(jarEntryName);
327           if (jarEntry != null) {
328             updatePortInfo(operator, jar.getInputStream(jarEntry));
329           }
330         } else {
331           File f = openClassFiles.get(path);
332           if (f != null && f.exists() && f.getName().endsWith("class")) {
333             updatePortInfo(operator, new FileInputStream(f));
334           }
335         }
336         updatePortTypeInfoInTypeGraph(openJarFiles, openClassFiles, operator);
337       } catch (Exception e) {
338         DTThrowable.wrapIfChecked(e);
339       }
340     }
341   }

343   private void updatePortInfo(TypeGraphVertex tgv, InputStream input)
344       throws IOException {
345     try {
346       ClassReader reader;
347       reader = new ClassReader(input);
348       ClassNodeType classN = new ClassNodeType();
349       classN.setClassSignatureVisitor(tgv.getClassNode().getCsv());
350       classN.setVisitFields(true);
351       reader.accept(classN, ClassReader.SKIP_CODE);
352       CompactClassNode ccn = tgv.getClassNode();
353       CompactUtil.updateCompactClassPortInfo(classN, ccn);
354       List<CompactFieldNode> prunedFields = new LinkedList<CompactFieldNode>();
355       TypeGraphVertex portVertex = typeGraph.get(Operator.Port.class.getName());
356       for (CompactFieldNode field : ccn.getPorts()) {
357         TypeGraphVertex fieldVertex = typeGraph.get(field.getDescription());
358         if(fieldVertex != null) {
359           if (isAncestor(portVertex, fieldVertex)) {
360             prunedFields.add(field);
361           }
362         }
363       }
364       ccn.setPorts(prunedFields);
365 
366     } finally {
367       if (input != null) {
368         input.close();
369       }
370     }
371   }

373   private void updateInstantiableDescendants(TypeGraphVertex tgv)
374   {
375     if(tgv.isInstantiable()){
376       tgv.allInstantiableDescendants.add(tgv);
377     }
378     for (TypeGraphVertex parent : tgv.ancestors) {
379       updateInstantiableDescendants(parent, tgv.allInstantiableDescendants);
380     }
381   }

390       updateInstantiableDescendants(parent, allChildren);
391     }

430   private void tranverse(TypeGraphVertex tgv, boolean onlyInstantiable, Set<String> result, int limit)
431   {
432     if (!onlyInstantiable) {
433       result.add(tgv.typeName);
434     }
435 
436     if (onlyInstantiable && tgv.numberOfInstantiableDescendants() > limit) {
437       throw new RuntimeException("Too many public concrete sub types!");
438     }
439 
440     if (onlyInstantiable && tgv.isInstantiable()) {
441       result.add(tgv.typeName);
442     }
443 
444     if (tgv.descendants.size() > 0) {
445       for (TypeGraphVertex child : tgv.descendants) {
446         tranverse(child, onlyInstantiable, result, limit);
447       }
448     }

535     private boolean isPublicConcrete()
536     {
537       if (classNode == null) {
538         // If the class is not in the classpath
539         return false;
540       }
541       int opCode = getOpCode();
542 
543       // if the class is neither abstract nor interface
544       // and the class is public
545       return ((opCode & (Opcodes.ACC_ABSTRACT | Opcodes.ACC_INTERFACE)) == 0) && ((opCode & Opcodes.ACC_PUBLIC) == Opcodes.ACC_PUBLIC);
546     }

548     private int getOpCode()
549     {
550       List<CompactClassNode> icl = classNode.getInnerClasses();
551       if (typeName.contains("$")) {
552         for (CompactClassNode innerClassNode : icl) {
553           if (innerClassNode.getName().replace('/', '.').equals(typeName)) {
554             return innerClassNode.getAccess();
555           }
556         }
557       }
558       return classNode.getAccess();
559     }

614   /**
615    * @param clazz parent class
616    * @param filter
617    * @param packagePrefix 
618    * @param startsWith  case insensitive
619    * @return all instantiable descendants of class clazz which comfort to filter expression, packagePrefix and start with $startsWith
620    */
621   public List<String> getInstantiableDescendants(String clazz, String filter, String packagePrefix, String startsWith)
622   {
623     TypeGraphVertex tgv = typeGraph.get(clazz);
624     if(tgv == null) {
625       return null;
626     }
627     
628     List<String> result = new LinkedList<String>();
629     if (tgv != null) {
630       
631       for (TypeGraphVertex node : tgv.allInstantiableDescendants) {
632         String typeName = node.typeName;
633         if (filter != null && !Pattern.matches(filter, node.typeName)) {
634           continue;
635         }
636         if (packagePrefix != null && !node.typeName.startsWith(packagePrefix)) {
637           continue;
638         }
639         if (startsWith != null && !typeName.substring(typeName.lastIndexOf('.') + 1).toLowerCase().startsWith(startsWith.toLowerCase())){
640           continue;
641         }
642         result.add(node.typeName);
643       }
644     }
645     return result;
646   }

648   public JSONObject describeClass(String clazzName) throws JSONException
649   {
650     JSONObject desc = new JSONObject();
651     desc.put("name", clazzName);
652     TypeGraphVertex tgv = typeGraph.get(clazzName);
653     if (tgv == null) {
654       return desc;
655     }
656     CompactClassNode cn = tgv.classNode;
657     if (cn.isEnum()) {
658 
659       List<String> enumNames = cn.getEnumValues();
660       desc.put("enum", enumNames);
661       desc.put("uiType", UI_TYPE.ENUM.getName());
662     }
663     UI_TYPE uType = UI_TYPE.getEnumFor(tgv.typeName, typeGraph);
664     if (uType != null) {
665       desc.put("uiType", uType.getName());
666     }
667     
668     addClassPropertiesAndPorts(clazzName,  desc);
669 
670     if(tgv.hasResource()){
671       desc.put("hasResource", "true");
672     } else {
673       desc.put("hasResource", "false");
674     }
675 
676     return desc;
677   }

679   private Collection<JSONObject> getPortTypeInfo(String clazzName,
680       Map<Type, Type> typeReplacement, List<CompactFieldNode> ports) throws JSONException {
681     TypeGraphVertex tgv = typeGraph.get(clazzName);
682     if (tgv == null) {
683       return null;
684     }
685     
686     Collection<JSONObject> portInfo = new ArrayList<JSONObject>();
687     
688     for (CompactFieldNode port : ports) {
689         Type fieldType = port.getFieldSignatureNode().getFieldType();
690         Type t = fieldType;
691         if (fieldType instanceof ParameterizedTypeNode) {
692           // TODO: Right now getPortInfo assumes a single parameterized type
693           t = ((ParameterizedTypeNode) fieldType).getActualTypeArguments()[0];
694         } else {
695           // TODO: Check behavior for Ports not using Default Input/output ports
696           TypeGraphVertex portVertex = typeGraph.get(port.getDescription());
697           t = findTypeArgument(portVertex, typeReplacement);
698           LOG.debug("Field is of type {}", fieldType.getClass());
699         }
700 
701         JSONObject meta = new JSONObject();
702         try {
703           meta.put("name", port.getName()); 
704           setTypes(meta, t, typeReplacement);
705           portInfo.add(meta);
706         } catch (Exception e) {
707           DTThrowable.wrapIfChecked(e);
708         }

809   private void addClassPropertiesAndPorts(String clazzName, JSONObject desc) throws JSONException
810   {
811     TypeGraphVertex tgv = typeGraph.get(clazzName);
812     if (tgv == null) {
813       return;
814     }
815 
816     Map<String, JSONObject> results = new TreeMap<String, JSONObject>();
817     List<CompactMethodNode> getters = new LinkedList<CompactMethodNode>();
818     List<CompactMethodNode> setters = new LinkedList<CompactMethodNode>();
819     Map<Type, Type> typeReplacement = new HashMap<Type, Type>();
820     List<CompactFieldNode> ports = new LinkedList<CompactFieldNode>();
821 
822     getPublicSetterGetterAndPorts(tgv, setters, getters, typeReplacement, ports);
823     desc.put("portTypeInfo", getPortTypeInfo(clazzName, typeReplacement, ports));
824 
825     for (CompactMethodNode setter : setters) {
826       String prop = WordUtils.uncapitalize(setter.getName().substring(3));
827       JSONObject propJ = results.get(prop);
828       if (propJ == null) {
829         propJ = new JSONObject();
830         propJ.put("name", prop);
831         results.put(prop, propJ);
832       }
833       propJ.put("canSet", true);
834       propJ.put("canGet", false);
835 
836       MethodSignatureVisitor msv = null;
837       msv = setter.getMethodSignatureNode();
838       if (msv == null) {
839         continue;
840       }
841 
842       List<Type> param = msv.getParameters();
843       if (CollectionUtils.isEmpty(param)) {
844         propJ.put("type", "UNKNOWN");
845       } else {
846         // only one param in setter method
847         setTypes(propJ, param.get(0), typeReplacement);
848         // propJ.put("type", param.getTypeObj().getClassName());
849 
850       }
851       // propJ.put("type", typeString);
852     }
853 
854     for (CompactMethodNode getter : getters) {
855       int si = getter.getName().startsWith("is") ? 2 : 3;
856       String prop = WordUtils.uncapitalize(getter.getName().substring(si));
857       JSONObject propJ = results.get(prop);
858       if (propJ == null) {
859         propJ = new JSONObject();
860         propJ.put("name", prop);
861         results.put(prop, propJ);
862         propJ.put("canSet", false);
863         // propJ.put("type", Type.getReturnType(getter.desc).getClassName());
864 
865         MethodSignatureVisitor msv = null;
866         msv = getter.getMethodSignatureNode();
867         if (msv == null) {
868           continue;
869         }
870 
871         Type rt = msv.getReturnType();
872         if (rt == null) {
873           propJ.put("type", "UNKNOWN");
874         } else {
875           setTypes(propJ, rt, typeReplacement);
876           // propJ.put("type", param.getTypeObj().getClassName());
877         }
878 
879       }
880 
881       propJ.put("canGet", true);
882     }
883 
884     desc.put("properties", results.values());
885   }

887   private void getPublicSetterGetterAndPorts(TypeGraphVertex tgv, List<CompactMethodNode> setters, List<CompactMethodNode> getters, Map<Type, Type> typeReplacement, List<CompactFieldNode> ports)
888   {
889     CompactClassNode exClass = null;
890     // check if the class needs to be excluded
891     for (String e : EXCLUDE_CLASSES) {
892       if(e.equals(tgv.getClassNode().getName())) {
893         exClass = tgv.getClassNode();
894         break;
895       }
896     }
897     if (exClass != null) {
898       // if it's visiting classes need to be exclude from parsing, remove methods that override in sub class
899       // So the setter/getter methods in Operater, Object, Class won't be counted
900       for (CompactMethodNode compactMethodNode : exClass.getGetterMethods()) {
901         for (Iterator<CompactMethodNode> iterator = getters.iterator(); iterator.hasNext();) {
902           CompactMethodNode cmn = iterator.next();
903           if (cmn.getName().equals(compactMethodNode.getName())) {
904             iterator.remove();
905           }
906         }
907       }
908       for (CompactMethodNode compactMethodNode : exClass.getSetterMethods()) {
909         for (Iterator<CompactMethodNode> iterator = setters.iterator(); iterator.hasNext();) {
910           CompactMethodNode cmn = iterator.next();
911           if (cmn.getName().equals(compactMethodNode.getName())) {
912             iterator.remove();
913           }
914         }
915       }
916     } else {
917       if (tgv.getClassNode().getSetterMethods() != null) {
918         setters.addAll(tgv.getClassNode().getSetterMethods());
919       }
920       if (tgv.getClassNode().getGetterMethods() != null) {
921         getters.addAll(tgv.getClassNode().getGetterMethods());
922       }
923     }
924     
925     TypeGraphVertex portVertex = typeGraph.get(Operator.Port.class.getName());
926     List<CompactFieldNode> fields = tgv.getClassNode().getPorts();
927     if(fields != null) {
928       for (CompactFieldNode field : fields) {
929         TypeGraphVertex fieldVertex = typeGraph.get(field.getDescription());
930         if (isAncestor(portVertex, fieldVertex)) {
931           ports.add(field);
932         }
933       }
934     }
935     
936     ClassSignatureVisitor csv = tgv.getClassNode().getCsv();
937     Type superC = csv.getSuperClass();
938     
939     addReplacement(superC, typeReplacement);
940 
941     if(csv.getInterfaces()!=null){
942       for(Type it : csv.getInterfaces()){
943         addReplacement(it, typeReplacement);
944       };
945     }
946     for (TypeGraphVertex ancestor : tgv.ancestors) {
947       getPublicSetterGetterAndPorts(ancestor, setters, getters, typeReplacement, ports);
948     }
949   }

952   {

968   private void setTypes(JSONObject propJ, Type rawType, Map<Type, Type> typeReplacement, Set<Type> visitedType) throws JSONException
969   {
970     boolean stopRecursive = visitedType.contains(rawType);
971     visitedType.add(rawType);
972     // type could be replaced
973     Type t = resolveType(rawType, typeReplacement);
974     
975     if (propJ == null) {
976       return;
977     } else {
978       if (t instanceof WildcardTypeNode) {
979         propJ.put("type", "?");
980       } else if (t instanceof TypeNode) {
981         TypeNode tn = (TypeNode) t;
982         String typeS = tn.getTypeObj().getClassName();
983         propJ.put("type", typeS);
984         UI_TYPE uiType = UI_TYPE.getEnumFor(typeS, typeGraph);
985         if (uiType != null) {
986           switch (uiType) {
987             case FLOAT:
988             case LONG:
989             case INT:
990             case DOUBLE:
991             case BYTE:
992             case SHORT:
993             case STRING:
994               propJ.put("type",  uiType.getName());
995               break;
996             default:
997               propJ.put("uiType", uiType.getName());
998 
999           }
1000 
1001         }
1002         if (t instanceof ParameterizedTypeNode) {
1003           JSONArray jArray = new JSONArray();
1004           for (Type ttn : ((ParameterizedTypeNode) t).getActualTypeArguments()) {
1005             JSONObject objJ = new JSONObject();
1006             if (!stopRecursive) {
1007               setTypes(objJ, ttn, typeReplacement, visitedType);
1008             }
1009             jArray.put(objJ);
1010           }
1011           propJ.put("typeArgs", jArray);
1012         }
1013       }
1014       if (t instanceof WildcardTypeNode) {
1015         JSONObject typeBounds = new JSONObject();
1016         
1017         
1018         JSONArray jArray = new JSONArray();
1019         Type[] bounds = ((WildcardTypeNode) t).getUpperBounds();
1020         if(bounds!=null){
1021           for (Type type : bounds) {
1022             jArray.put(type.toString());
1023           }
1024         }
1025         typeBounds.put("upper", jArray);
1026         
1027         bounds = ((WildcardTypeNode) t).getLowerBounds();
1028 
1029         jArray = new JSONArray();
1030         if(bounds!=null){
1031           for (Type type : bounds) {
1032             jArray.put(type.toString());
1033           }
1034         }
1035         typeBounds.put("lower", jArray);
1036 
1037         propJ.put("typeBounds", typeBounds);
1038 
1039       }
1040       if(t instanceof ArrayTypeNode){
1041         propJ.put("type", t.getByteString());
1042         propJ.put("uiType", UI_TYPE.LIST.getName());
1043         
1044         JSONObject jObj = new JSONObject();
1045         if (!stopRecursive) {
1046           setTypes(jObj, ((ArrayTypeNode) t).getActualArrayType(), typeReplacement, visitedType);
1047         }
1048         propJ.put("itemType", jObj);
1049       }
1050       
1051       if(t instanceof TypeVariableNode){
1052         propJ.put("typeLiteral", ((TypeVariableNode)t).getTypeLiteral());
1053         if (!stopRecursive) {
1054           setTypes(propJ, ((TypeVariableNode) t).getRawTypeBound(), typeReplacement, visitedType);
1055         }
1056       }
1057 
1058 
1059     }
1060   }

1062   /*
1063    * Trim the node(s) that depends on external dependencies.
1064    * The classes that (in)directly implements/extends external classes can't be initialized anyways
1065    */
1066   public void trim()
1067   {
1068     List<TypeGraphVertex> invalidVertexes = new LinkedList<>();
1069     for (TypeGraphVertex tgv : typeGraph.values()) {
1070       if (tgv.getClassNode() == null) {
1071         invalidVertexes.add(tgv);
1072       }
1073     }
1074 
1075     for (TypeGraphVertex removeV : invalidVertexes) {
1076       // Removing node and all it's (in)direct descendants
1077       removeSubGraph(removeV);
1078     }
1079   }

1081   private void removeSubGraph(TypeGraphVertex v)
1082   {
1083 
1084     // Can't recursively remove because it will get into concurrent modification
1085     // Use queue to delete all nodes
1086     Queue<TypeGraphVertex> removingQueue = new LinkedList<>();
1087     removingQueue.add(v);
1088     while (!removingQueue.isEmpty()) {
1089       TypeGraphVertex tgv = removingQueue.poll();
1090       if (typeGraph.get(tgv.typeName) == null) {
1091         // skip node that's been removed already.
1092         // It comes from common descendants
1093         continue;
1094       }
1095       // put all the descendants to waiting queue
1096       for (TypeGraphVertex child : tgv.descendants) {
1097         removingQueue.offer(child);
1098       }
1099       // remove from global hashmap
1100       typeGraph.remove(tgv.typeName);
1101       // remove from instantiable descendants list of all the (in)direct ancestors
1102       if (!tgv.allInstantiableDescendants.isEmpty() && !tgv.ancestors.isEmpty()) {
1103         for (TypeGraphVertex p : tgv.ancestors) {
1104           removeFromInstantiableDescendants(p, tgv.allInstantiableDescendants);
1105         }
1106       }
1107       // cut links from parent to child
1108       for (TypeGraphVertex parent : tgv.ancestors) {
1109         parent.descendants.remove(tgv);
1110       }
1111       // cut links form child to parent
1112       tgv.ancestors.clear();
1113     }
1114   }
1115 
1116   private void removeFromInstantiableDescendants(TypeGraphVertex parentT, Set<TypeGraphVertex> childT)
1117   {
1118     for (TypeGraphVertex vertex : childT) {
1119       parentT.allInstantiableDescendants.remove(vertex);
1120     }
1121     for (TypeGraphVertex pt : parentT.ancestors) {
1122       removeFromInstantiableDescendants(pt, childT);
1123     }
1124   }

1136   /**
1137    * Type graph is big bidirectional object graph which can not be serialized by kryo.
1138    * This class is alternative {@link TypeGraph} kryo serializer
1139    * The serializer rule is
1140    * #ofNodes + node array + relationship array(int array which the value is index of the node array)
1141    */
1142   public static class TypeGraphSerializer extends Serializer<TypeGraph>
1143   {
1144 
1145     @Override
1146     public void write(Kryo kryo, Output output, TypeGraph tg)
1147     {
1148       Map<String, Integer> indexes = new HashMap<String, Integer>();
1149       // write the size first
1150       kryo.writeObject(output, tg.typeGraph.size());
1151       int i = 0;
1152       // Sequentially write the vertexes
1153       for (Entry<String, TypeGraphVertex> e : tg.typeGraph.entrySet()) {
1154         indexes.put(e.getKey(), i++);
1155         kryo.writeObject(output, e.getValue());
1156       }
1157       
1158       // Sequentially store the descendants and instantiable descendants relationships in index in vertex array
1159       for (Entry<String, TypeGraphVertex> e : tg.typeGraph.entrySet()) {
1160         int[] refs = fromSet(e.getValue().descendants, indexes);
1161         kryo.writeObject(output, refs);
1162         refs = fromSet(e.getValue().allInstantiableDescendants, indexes);
1163         kryo.writeObject(output, refs);
1164       }
1165       
1166     }
1167 
1168     private int[] fromSet(Set<TypeGraphVertex> tgvSet, Map<String, Integer> indexes)
1169     {
1170       int[] result = new int[tgvSet.size()];
1171       int j = 0;
1172       for (TypeGraphVertex t : tgvSet) {
1173         result[j++] = indexes.get(t.typeName);
1174       }
1175       return result;
1176     }
1177 
1178     @Override
1179     public TypeGraph read(Kryo kryo, Input input, Class<TypeGraph> type)
1180     {
1181       // read the #vertex
1182       int vertexNo = kryo.readObject(input, Integer.class);
1183       // read the vertexes into array
1184       TypeGraphVertex[] tgv = new TypeGraphVertex[vertexNo];
1185       for (int i = 0; i < vertexNo; i++) {
1186         tgv[i] = kryo.readObject(input, TypeGraphVertex.class);
1187       }
1188       
1189       // build relations between vertexes
1190       for (int i = 0; i < tgv.length; i++) {
1191         int[] ref = kryo.readObject(input, int[].class);
1192         for (int j = 0; j < ref.length; j++) {
1193           tgv[i].descendants.add(tgv[ref[j]]);
1194           tgv[ref[j]].ancestors.add(tgv[i]);
1195         }
1196         
1197         ref = kryo.readObject(input, int[].class);
1198         for (int j = 0; j < ref.length; j++) {
1199           tgv[i].allInstantiableDescendants.add(tgv[ref[j]]);
1200         }
1201         
1202       }
1203       TypeGraph result = new TypeGraph();
1204       for (TypeGraphVertex typeGraphVertex : tgv) {
1205         result.typeGraph.put(typeGraphVertex.typeName, typeGraphVertex);
1206       }
1207       return result;
1208     }
1209 
1210   }

1230   /**
1231    * A utility method that tells whether a class is considered a bean.<br/>
1232    * For simplicity we exclude classes that have any type-args.
1233    *
1234    * @param className name of the class
1235    * @return true if it is a bean false otherwise.
1236    */
1237   public boolean isInstantiableBean(String className) throws JSONException
1238   {
1239     JSONObject classDesc = describeClass(className);
1240     if (classDesc.has("typeArgs")) {
1241       //any type with generics is not considered a bean
1242       return false;
1243     }
1244     JSONArray classProps = classDesc.optJSONArray("properties");
1245     if (classProps == null || classProps.length() == 0) {
1246       //no properties then cannot be a bean
1247       return false;
1248     }
1249     for (int p = 0; p < classProps.length(); p++) {
1250       JSONObject propDesc = classProps.getJSONObject(p);
1251       if (propDesc.optBoolean("canGet", false)) {
1252         return true;
1253       }
1254     }
1255     return false;
1256   }
