270     {
271       out.writeString(operatorName);

383   public final ContainerInfo getAppMasterContainerInfo()
384   {
385     ContainerInfo ci = new ContainerInfo();
386     ci.id = System.getenv(ApplicationConstants.Environment.CONTAINER_ID.toString());
387     String nmHost = System.getenv(ApplicationConstants.Environment.NM_HOST.toString());
388     String nmPort = System.getenv(ApplicationConstants.Environment.NM_PORT.toString());
389     String nmHttpPort = System.getenv(ApplicationConstants.Environment.NM_HTTP_PORT.toString());
390     ci.state = "ACTIVE";
391     ci.jvmName = ManagementFactory.getRuntimeMXBean().getName();
392     ci.numOperators = 0;
393     YarnConfiguration conf = new YarnConfiguration();
394 
395     if (nmHost != null) {
396       if (nmPort != null) {
397         ci.host = nmHost + ":" + nmPort;
398       }
399       if (nmHttpPort != null) {
400         String nodeHttpAddress = nmHost + ":" + nmHttpPort;
401         if (allocatedMemoryMB == 0) {
402           String url = ConfigUtils.getSchemePrefix(conf) + nodeHttpAddress + "/ws/v1/node/containers/" + ci.id;
403           WebServicesClient webServicesClient = new WebServicesClient();
404           try {
405             String content = webServicesClient.process(url, String.class, new WebServicesClient.GetWebServicesHandler<String>());
406             JSONObject json = new JSONObject(content);
407             int totalMemoryNeededMB = json.getJSONObject("container").getInt("totalMemoryNeededMB");
408             if (totalMemoryNeededMB > 0) {
409               allocatedMemoryMB = totalMemoryNeededMB;
410             } else {
411               LOG.warn("Could not determine the memory allocated for the streaming application master.  Node manager is reporting {} MB from {}", totalMemoryNeededMB, url);
412             }
413           }
414           catch (Exception ex) {
415             LOG.warn("Could not determine the memory allocated for the streaming application master", ex);
416           }
417         }
418         ci.containerLogsUrl = ConfigUtils.getSchemePrefix(conf) + nodeHttpAddress + "/node/containerlogs/" + ci.id + "/" + System.getenv(ApplicationConstants.Environment.USER.toString());
419         ci.rawContainerLogsUrl = ConfigUtils.getRawContainerLogsUrl(conf, nodeHttpAddress, plan.getLogicalPlan().getAttributes().get(LogicalPlan.APPLICATION_ID), ci.id);
420       }
421     }
422     ci.memoryMBAllocated = allocatedMemoryMB;
423     ci.memoryMBFree = ((int)(Runtime.getRuntime().freeMemory() / (1024 * 1024)));
424     ci.lastHeartbeat = -1;
425     ci.startedTime = startTime;
426     ci.finishedTime = -1;
427     return ci;
428   }

560   public List<AppDataSource> getAppDataSources()
561   {
562     synchronized (appDataSourcesLock) {
563       if (appDataSources == null) {
564         appDataSources = new ArrayList<AppDataSource>();
565         operators:
566         for (LogicalPlan.OperatorMeta operatorMeta : plan.getLogicalPlan().getAllOperators()) {
567           Map<LogicalPlan.InputPortMeta, LogicalPlan.StreamMeta> inputStreams = operatorMeta.getInputStreams();
568           Map<LogicalPlan.OutputPortMeta, LogicalPlan.StreamMeta> outputStreams = operatorMeta.getOutputStreams();
569 
570           String queryOperatorName = null;
571           String queryUrl = null;
572           String queryTopic = null;
573 
574           boolean hasEmbeddedQuery = false;
575 
576           //Discover embeddable query connectors
577           if (operatorMeta.getOperator() instanceof AppData.Store<?>) {
578             AppData.Store<?> store = (AppData.Store<?>)operatorMeta.getOperator();
579             AppData.EmbeddableQueryInfoProvider<?> embeddableQuery = store.getEmbeddableQueryInfoProvider();
580 
581             if (embeddableQuery != null) {
582               hasEmbeddedQuery = true;
583               queryOperatorName = operatorMeta.getName() + EMBEDDABLE_QUERY_NAME_SUFFIX;
584               queryUrl = embeddableQuery.getAppDataURL();
585               queryTopic = embeddableQuery.getTopic();
586             }
587           }
588 
589           //Discover separate query operators
590           LOG.warn("DEBUG: looking at operator {} {}", operatorMeta.getName(), Thread.currentThread().getId());
591           for (Map.Entry<LogicalPlan.InputPortMeta, LogicalPlan.StreamMeta> entry : inputStreams.entrySet()) {
592             LogicalPlan.InputPortMeta portMeta = entry.getKey();
593             if (portMeta.isAppDataQueryPort()) {
594               if (queryUrl == null) {
595                 OperatorMeta queryOperatorMeta = entry.getValue().getSource().getOperatorMeta();
596                 if (queryOperatorMeta.getOperator() instanceof AppData.ConnectionInfoProvider) {
597                   if (!hasEmbeddedQuery) {
598                     AppData.ConnectionInfoProvider queryOperator = (AppData.ConnectionInfoProvider)queryOperatorMeta.getOperator();
599                     queryOperatorName = queryOperatorMeta.getName();
600                     queryUrl = queryOperator.getAppDataURL();
601                     queryTopic = queryOperator.getTopic();
602                   } else {
603                     LOG.warn("An embeddable query connector and the {} query operator were discovered. " +
604                              "The query operator will be ignored and the embeddable query connector will be used instead.",
605                              operatorMeta.getName());
606                   }
607                 }
608               } else {
609                 LOG.warn("Multiple query ports found in operator {}. Ignoring the App Data Source.", operatorMeta.getName());
610                 continue operators;
611               }
612             }
613           }
614 
615           for (Map.Entry<LogicalPlan.OutputPortMeta, LogicalPlan.StreamMeta> entry : outputStreams.entrySet()) {
616             LogicalPlan.OutputPortMeta portMeta = entry.getKey();
617             LOG.warn("DEBUG: looking at port {} {}", portMeta.getPortName(), Thread.currentThread().getId());
618 
619             if (portMeta.isAppDataResultPort()) {
620               AppDataSource appDataSource = new AppDataSource();
621               appDataSource.setType(AppDataSource.Type.DAG);
622               appDataSource.setOperatorName(operatorMeta.getName());
623               appDataSource.setPortName(portMeta.getPortName());
624 
625               if (queryOperatorName == null) {
626                 LOG.warn("There is no query operator for the App Data Source {}.{}. Ignoring the App Data Source.", operatorMeta.getName(), portMeta.getPortName());
627                 continue;
628               }
629               appDataSource.setQueryOperatorName(queryOperatorName);
630               appDataSource.setQueryTopic(queryTopic);
631               appDataSource.setQueryUrl(convertAppDataUrl(queryUrl));
632               List<LogicalPlan.InputPortMeta> sinks = entry.getValue().getSinks();
633               if (sinks.isEmpty()) {
634                 LOG.warn("There is no result operator for the App Data Source {}.{}. Ignoring the App Data Source.", operatorMeta.getName(), portMeta.getPortName());
635                 continue;
636               }
637               if (sinks.size() > 1) {
638                 LOG.warn("There are multiple result operators for the App Data Source {}.{}. Ignoring the App Data Source.", operatorMeta.getName(), portMeta.getPortName());
639                 continue;
640               }
641               OperatorMeta resultOperatorMeta = sinks.get(0).getOperatorWrapper();
642               if (resultOperatorMeta.getOperator() instanceof AppData.ConnectionInfoProvider) {
643                 AppData.ConnectionInfoProvider resultOperator = (AppData.ConnectionInfoProvider) resultOperatorMeta.getOperator();
644                 appDataSource.setResultOperatorName(resultOperatorMeta.getName());
645                 appDataSource.setResultTopic(resultOperator.getTopic());
646                 appDataSource.setResultUrl(convertAppDataUrl(resultOperator.getAppDataURL()));
647                 AppData.AppendQueryIdToTopic queryIdAppended = resultOperator.getClass().getAnnotation(AppData.AppendQueryIdToTopic.class);
648                 if (queryIdAppended != null && queryIdAppended.value()) {
649                   appDataSource.setResultAppendQIDTopic(true);
650                 }
651               } else {
652                 LOG.warn("Result operator for the App Data Source {}.{} does not implement the right interface. Ignoring the App Data Source.", operatorMeta.getName(), portMeta.getPortName());
653                 continue;
654               }
655               LOG.warn("DEBUG: Adding appDataSource {} {}", appDataSource.getName(), Thread.currentThread().getId());
656               appDataSources.add(appDataSource);
657             }
658           }
659         }
660       }
661     }
662     return appDataSources;
663   }

671   /**
672    * Check periodically that deployed containers phone home.
673    * Run from the master main loop (single threaded access).
674    */
675   public void monitorHeartbeat()
676   {
677     long currentTms = clock.getTime();
678 
679     // look for resource allocation timeout
680     if (!pendingAllocation.isEmpty()) {
681       // look for resource allocation timeout
682       if (lastResourceRequest + plan.getLogicalPlan().getValue(LogicalPlan.RESOURCE_ALLOCATION_TIMEOUT_MILLIS) < currentTms) {
683         String msg = String.format("Shutdown due to resource allocation timeout (%s ms) waiting for %s containers", currentTms - lastResourceRequest, pendingAllocation.size());
684         LOG.warn(msg);
685         for (PTContainer c : pendingAllocation) {
686           LOG.warn("Waiting for resource: {}m priority: {} {}", c.getRequiredMemoryMB(), c.getResourceRequestPriority(), c);
687         }
688         shutdownAllContainers(msg);
689         this.forcedShutdown = true;
690       }
691       else {
692         for (PTContainer c : pendingAllocation) {
693           LOG.debug("Waiting for resource: {}m {}", c.getRequiredMemoryMB(), c);
694         }
695       }
696     }
697 
698     // monitor currently deployed containers
699     for (StreamingContainerAgent sca : containers.values()) {
700       PTContainer c = sca.container;
701 
702       if (!pendingAllocation.contains(c) && c.getExternalId() != null) {
703         if (sca.lastHeartbeatMillis == 0) {
704           //LOG.debug("{} {} {}", c.getExternalId(), currentTms - sca.createdMillis, this.vars.heartbeatTimeoutMillis);
705           // container allocated but process was either not launched or is not able to phone home
706           if (currentTms - sca.createdMillis > 2 * this.vars.heartbeatTimeoutMillis) {
707             LOG.info("Container {}@{} startup timeout ({} ms).", c.getExternalId(), c.host, currentTms - sca.createdMillis);
708             containerStopRequests.put(c.getExternalId(), c.getExternalId());
709           }
710         }
711         else {
712           if (currentTms - sca.lastHeartbeatMillis > this.vars.heartbeatTimeoutMillis) {
713             if (!isApplicationIdle()) {
714               // request stop (kill) as process may still be hanging around (would have been detected by Yarn otherwise)
715               LOG.info("Container {}@{} heartbeat timeout ({} ms).", c.getExternalId(), c.host, currentTms - sca.lastHeartbeatMillis);
716               containerStopRequests.put(c.getExternalId(), c.getExternalId());
717             }
718           }
719         }
720       }
721     }
722 
723     // events that may modify the plan
724     processEvents();
725 
726     committedWindowId = updateCheckpoints(false);
727     calculateEndWindowStats();
728     if (this.vars.enableStatsRecording) {
729       recordStats(currentTms);
730     }
731   }

744   private void calculateEndWindowStats()
745   {
746     if (!endWindowStatsOperatorMap.isEmpty()) {
747       Set<Integer> allCurrentOperators = plan.getAllOperators().keySet();
748 
749       if (endWindowStatsOperatorMap.size() > this.vars.maxWindowsBehindForStats) {
750         LOG.warn("Some operators are behind for more than {} windows! Trimming the end window stats map", this.vars.maxWindowsBehindForStats);
751         while (endWindowStatsOperatorMap.size() > this.vars.maxWindowsBehindForStats) {
752           LOG.debug("Removing incomplete end window stats for window id {}. Collected operator set: {}. Complete set: {}",
753             endWindowStatsOperatorMap.firstKey(),
754             endWindowStatsOperatorMap.get(endWindowStatsOperatorMap.firstKey()).keySet(), allCurrentOperators);
755           endWindowStatsOperatorMap.remove(endWindowStatsOperatorMap.firstKey());
756         }
757       }
758       //logicalMetrics.clear();
759       int numOperators = allCurrentOperators.size();
760       Long windowId = endWindowStatsOperatorMap.firstKey();
761       while (windowId != null) {
762         Map<Integer, EndWindowStats> endWindowStatsMap = endWindowStatsOperatorMap.get(windowId);
763         Set<Integer> endWindowStatsOperators = endWindowStatsMap.keySet();
764 
765         aggregateMetrics(windowId, endWindowStatsMap);
766 
767         if (allCurrentOperators.containsAll(endWindowStatsOperators)) {
768           if (endWindowStatsMap.size() < numOperators) {
769             if (windowId < completeEndWindowStatsWindowId) {
770               LOG.debug("Disregarding stale end window stats for window {}", windowId);
771               endWindowStatsOperatorMap.remove(windowId);
772             }
773             else {
774               break;
775             }
776           }
777           else {
778             // collected data from all operators for this window id.  start latency calculation
779             List<OperatorMeta> rootOperatorMetas = plan.getLogicalPlan().getRootOperators();
780             Set<PTOperator> endWindowStatsVisited = new HashSet<PTOperator>();
781             Set<PTOperator> leafOperators = new HashSet<PTOperator>();
782             for (OperatorMeta root : rootOperatorMetas) {
783               List<PTOperator> rootOperators = plan.getOperators(root);
784               for (PTOperator rootOperator : rootOperators) {
785                 // DFS for visiting the operators for latency calculation
786                 LOG.debug("Calculating latency starting from operator {}", rootOperator.getId());
787                 calculateLatency(rootOperator, endWindowStatsMap, endWindowStatsVisited, leafOperators);
788               }
789             }
790             CriticalPathInfo cpi = new CriticalPathInfo();
791             //LOG.debug("Finding critical path...");
792             cpi.latency = findCriticalPath(endWindowStatsMap, leafOperators, cpi.path);
793             criticalPathInfo = cpi;
794             endWindowStatsOperatorMap.remove(windowId);
795             currentEndWindowStatsWindowId = windowId;
796           }
797         }
798         else {
799           // the old stats contains operators that do not exist any more
800           // this is probably right after a partition happens.
801           LOG.debug("Stats for non-existent operators detected. Disregarding end window stats for window {}", windowId);
802           endWindowStatsOperatorMap.remove(windowId);
803         }
804         windowId = endWindowStatsOperatorMap.higherKey(windowId);
805       }
806     }

809   private void aggregateMetrics(long windowId, Map<Integer, EndWindowStats> endWindowStatsMap)
810   {
811     Collection<OperatorMeta> logicalOperators = getLogicalPlan().getAllOperators();
812     //for backward compatibility
813     for (OperatorMeta operatorMeta : logicalOperators) {
814       Context.CountersAggregator aggregator = operatorMeta.getValue(OperatorContext.COUNTERS_AGGREGATOR);
815       if (aggregator == null) {
816         continue;
817       }
818       Collection<PTOperator> physicalOperators = plan.getAllOperators(operatorMeta);
819       List<Object> counters = Lists.newArrayList();
820       for (PTOperator operator : physicalOperators) {
821         EndWindowStats stats = endWindowStatsMap.get(operator.getId());
822         if (stats != null && stats.counters != null) {
823           counters.add(stats.counters);
824         }
825       }
826       if (counters.size() > 0) {
827         Object aggregate = aggregator.aggregate(counters);
828         latestLogicalCounters.put(operatorMeta.getName(), aggregate);
829       }
830     }
831 
832     for (OperatorMeta operatorMeta : logicalOperators) {
833       AutoMetric.Aggregator aggregator = operatorMeta.getMetricAggregatorMeta() != null ?
834         operatorMeta.getMetricAggregatorMeta().getAggregator() : null;
835       if (aggregator == null) {
836         continue;
837       }
838       Collection<PTOperator> physicalOperators = plan.getAllOperators(operatorMeta);
839       List<AutoMetric.PhysicalMetricsContext> metricPool = Lists.newArrayList();
840 
841       for (PTOperator operator : physicalOperators) {
842         EndWindowStats stats = endWindowStatsMap.get(operator.getId());
843         if (stats != null && stats.metrics != null) {
844           PhysicalMetricsContextImpl physicalMetrics = new PhysicalMetricsContextImpl(operator.getId(), stats.metrics);
845           metricPool.add(physicalMetrics);
846         }
847       }
848       if (metricPool.isEmpty()) {
849         //nothing to aggregate
850         continue;
851       }
852       Map<String, Object> lm = aggregator.aggregate(windowId, metricPool);
853 
854       if (lm != null && lm.size() > 0) {
855         Queue<Pair<Long, Map<String, Object>>> windowMetrics = logicalMetrics.get(operatorMeta.getName());
856         if (windowMetrics == null) {
857           windowMetrics = new LinkedBlockingQueue<Pair<Long, Map<String, Object>>>(METRIC_QUEUE_SIZE)
858           {
859             @Override
860             public boolean add(Pair<Long, Map<String, Object>> longMapPair)
861             {
862               if (remainingCapacity() <= 1) {
863                 remove();
864               }
865               return super.add(longMapPair);
866             }
867           };
868           logicalMetrics.put(operatorMeta.getName(), windowMetrics);
869         }
870         LOG.debug("Adding to logical metrics for {}", operatorMeta.getName());
871         windowMetrics.add(new Pair<Long, Map<String, Object>>(windowId, lm));
872         Map<String, Object> oldValue = latestLogicalMetrics.put(operatorMeta.getName(), lm);
873         if (oldValue == null) {
874           try {
875             saveMetaInfo();
876           } catch (IOException ex) {
877             LOG.error("Cannot save application meta info to DFS. App data sources will not be available.", ex);
878           }
879         }
880       }
881     }

915   private void calculateLatency(PTOperator oper, Map<Integer, EndWindowStats> endWindowStatsMap, Set<PTOperator> endWindowStatsVisited, Set<PTOperator> leafOperators)
916   {
917     endWindowStatsVisited.add(oper);
918     OperatorStatus operatorStatus = oper.stats;
919 
920     EndWindowStats endWindowStats = endWindowStatsMap.get(oper.getId());
921     if (endWindowStats == null) {
922       LOG.info("End window stats is null for operator {}, probably a new operator after partitioning", oper);
923       return;
924     }
925 
926     // find the maximum end window emit time from all input ports
927     long upstreamMaxEmitTimestamp = -1;
928     PTOperator upstreamMaxEmitTimestampOperator = null;
929     for (PTOperator.PTInput input : oper.getInputs()) {
930       if (null != input.source.source) {
931         PTOperator upstreamOp = input.source.source;
932         EndWindowStats upstreamEndWindowStats = endWindowStatsMap.get(upstreamOp.getId());
933         if (upstreamEndWindowStats == null) {
934           LOG.info("End window stats is null for operator {}", oper);
935           return;
936         }
937         long adjustedEndWindowEmitTimestamp = upstreamEndWindowStats.emitTimestamp;
938         MovingAverageLong rpcLatency = rpcLatencies.get(upstreamOp.getContainer().getExternalId());
939         if (rpcLatency != null) {
940           adjustedEndWindowEmitTimestamp += rpcLatency.getAvg();
941         }
942         if (adjustedEndWindowEmitTimestamp > upstreamMaxEmitTimestamp) {
943           upstreamMaxEmitTimestamp = adjustedEndWindowEmitTimestamp;
944           upstreamMaxEmitTimestampOperator = upstreamOp;
945         }
946       }
947     }
948 
949     if (upstreamMaxEmitTimestamp > 0) {
950       long adjustedEndWindowEmitTimestamp = endWindowStats.emitTimestamp;
951       MovingAverageLong rpcLatency = rpcLatencies.get(oper.getContainer().getExternalId());
952       if (rpcLatency != null) {
953         adjustedEndWindowEmitTimestamp += rpcLatency.getAvg();
954       }
955       if (upstreamMaxEmitTimestamp <= adjustedEndWindowEmitTimestamp) {
956         LOG.debug("Adding {} to latency MA for {}", adjustedEndWindowEmitTimestamp - upstreamMaxEmitTimestamp, oper);
957         operatorStatus.latencyMA.add(adjustedEndWindowEmitTimestamp - upstreamMaxEmitTimestamp);
958       } else {
959         operatorStatus.latencyMA.add(0);
960         if (lastLatencyWarningTime < System.currentTimeMillis() - LATENCY_WARNING_THRESHOLD_MILLIS) {
961           LOG.warn("Latency calculation for this operator may not be correct because upstream end window timestamp is greater than this operator's end window timestamp: {} ({}) > {} ({}). Please verify that the system clocks are in sync in your cluster. You can also try tweaking the RPC_LATENCY_COMPENSATION_SAMPLES application attribute (currently set to {}).",
962                   upstreamMaxEmitTimestamp, upstreamMaxEmitTimestampOperator, adjustedEndWindowEmitTimestamp, oper, this.vars.rpcLatencyCompensationSamples);
963           lastLatencyWarningTime = System.currentTimeMillis();
964         }
965       }
966     }
967 
968     if (oper.getOutputs().isEmpty()) {
969       // it is a leaf operator
970       leafOperators.add(oper);
971     }
972     else {
973       for (PTOperator.PTOutput output : oper.getOutputs()) {
974         for (PTOperator.PTInput input : output.sinks) {
975           if (input.target != null) {
976             PTOperator downStreamOp = input.target;
977             if (!endWindowStatsVisited.contains(downStreamOp)) {
978               calculateLatency(downStreamOp, endWindowStatsMap, endWindowStatsVisited, leafOperators);
979             }
980           }
981         }
982       }
983     }
984   }

988   private long findCriticalPath(Map<Integer, EndWindowStats> endWindowStatsMap, Set<PTOperator> operators, LinkedList<Integer> criticalPath)
989   {
990     long maxEndWindowTimestamp = 0;
991     PTOperator maxOperator = null;
992     for (PTOperator operator : operators) {
993       EndWindowStats endWindowStats = endWindowStatsMap.get(operator.getId());
994       if (maxEndWindowTimestamp < endWindowStats.emitTimestamp) {
995         maxEndWindowTimestamp = endWindowStats.emitTimestamp;
996         maxOperator = operator;
997       }
998     }
999     if (maxOperator == null) {
1000       return 0;
1001     }
1002     criticalPath.addFirst(maxOperator.getId());
1003     OperatorStatus operatorStatus = maxOperator.stats;
1004 
1005     operators.clear();
1006     if (maxOperator.getInputs() == null || maxOperator.getInputs().isEmpty()) {
1007       return operatorStatus.latencyMA.getAvg();
1008     }
1009     for (PTOperator.PTInput input : maxOperator.getInputs()) {
1010       if (null != input.source.source) {
1011         operators.add(input.source.source);
1012       }
1013     }
1014     return operatorStatus.latencyMA.getAvg() + findCriticalPath(endWindowStatsMap, operators, criticalPath);
1015   }

1017   public int processEvents()
1018   {
1019     for (PTOperator o : reportStats.keySet()) {
1020       List<OperatorStats> stats = o.stats.listenerStats.poll();
1021       if (stats != null) {
1022         // append into single list
1023         List<OperatorStats> moreStats;
1024         while ((moreStats = o.stats.listenerStats.poll()) != null) {
1025           stats.addAll(moreStats);
1026         }
1027       }
1028       o.stats.lastWindowedStats = stats;
1029       if (o.stats.lastWindowedStats != null) {
1030         // call listeners only with non empty window list
1031         if (o.statsListeners != null) {
1032           plan.onStatusUpdate(o);
1033         }
1034       }
1035       reportStats.remove(o);
1036     }
1037 
1038     if (!this.shutdownOperators.isEmpty()) {
1039       synchronized (this.shutdownOperators) {
1040         Iterator<Map.Entry<Long, Set<PTOperator>>> it = shutdownOperators.entrySet().iterator();
1041         while (it.hasNext()) {
1042           Map.Entry<Long, Set<PTOperator>> windowAndOpers = it.next();
1043           if (windowAndOpers.getKey().longValue() <= this.committedWindowId || checkDownStreamOperators(windowAndOpers)) {
1044             LOG.info("Removing inactive operators at window {} {}", Codec.getStringWindowId(windowAndOpers.getKey()), windowAndOpers.getValue());
1045             for (PTOperator oper : windowAndOpers.getValue()) {
1046               plan.removeTerminatedPartition(oper);
1047             }
1048             it.remove();
1049           }
1050         }
1051       }
1052     }
1053 
1054     if (!eventQueue.isEmpty()) {
1055       for (PTOperator oper : plan.getAllOperators().values()) {
1056         if (oper.getState() != PTOperator.State.ACTIVE) {
1057           LOG.debug("Skipping plan updates due to inactive operator {} {}", oper, oper.getState());
1058           return 0;
1059         }
1060       }
1061     }
1062 
1063     int count = 0;
1064     Runnable command;
1065     while ((command = this.eventQueue.poll()) != null) {
1066       eventQueueProcessing.set(true);
1067       try {
1068         command.run();
1069         count++;
1070       } catch (Exception e) {
1071         // TODO: handle error
1072         LOG.error("Failed to execute {}", command, e);
1073       }
1074       eventQueueProcessing.set(false);
1075     }
1076 
1077     if (count > 0) {
1078       try {
1079         checkpoint();
1080       } catch (Exception e) {
1081         throw new RuntimeException("Failed to checkpoint state.", e);
1082       }
1083     }
1084 
1085     return count;
1086   }

1101   /**
1102    * Schedule container restart. Called by Stram after a container was terminated
1103    * and requires recovery (killed externally, or after heartbeat timeout). <br>
1104    * Recovery will resolve affected operators (within the container and
1105    * everything downstream with respective recovery checkpoint states).
1106    * Dependent operators will be undeployed and buffer server connections reset prior to
1107    * redeploy to recovery checkpoint.
1108    *
1109    * @param containerId
1110    */
1111   public void scheduleContainerRestart(String containerId)
1112   {
1113     StreamingContainerAgent cs = this.getContainerAgent(containerId);
1114     if (cs == null || cs.shutdownRequested) {
1115       // the container is no longer used / was released by us
1116       return;
1117     }
1118 
1119     LOG.info("Initiating recovery for {}@{}", containerId, cs.container.host);
1120 
1121     cs.container.setState(PTContainer.State.KILLED);
1122     cs.container.bufferServerAddress = null;
1123     cs.container.setResourceRequestPriority(-1);
1124     cs.container.setAllocatedMemoryMB(0);
1125     cs.container.setAllocatedVCores(0);
1126 
1127     // resolve dependencies
1128     UpdateCheckpointsContext ctx = new UpdateCheckpointsContext(clock);
1129     for (PTOperator oper : cs.container.getOperators()) {
1130       updateRecoveryCheckpoints(oper, ctx);
1131     }
1132     includeLocalUpstreamOperators(ctx);
1133 
1134     // redeploy cycle for all affected operators
1135     LOG.info("Affected operators {}", ctx.visited);
1136     deploy(Collections.<PTContainer>emptySet(), ctx.visited, Sets.newHashSet(cs.container), ctx.visited);
1137   }

1139   /**
1140    * Transitively add operators that are container local to the dependency set.
1141    * (All downstream operators were traversed during checkpoint update.)
1142    *
1143    * @param ctx
1144    */
1145   private void includeLocalUpstreamOperators(UpdateCheckpointsContext ctx)
1146   {
1147     Set<PTOperator> newOperators = Sets.newHashSet();
1148     // repeat until no more local upstream operators are found
1149     do {
1150       newOperators.clear();
1151       for (PTOperator oper : ctx.visited) {
1152         for (PTInput input : oper.getInputs()) {
1153           if (input.source.source.getContainer() == oper.getContainer()) {
1154             if (!ctx.visited.contains(input.source.source)) {
1155               newOperators.add(input.source.source);
1156             }
1157           }
1158         }
1159       }
1160       if (!newOperators.isEmpty()) {
1161         for (PTOperator oper : newOperators) {
1162           updateRecoveryCheckpoints(oper, ctx);
1163         }
1164       }
1165     } while (!newOperators.isEmpty());
1166   }

1314   private void processOperatorDeployStatus(final PTOperator oper, OperatorHeartbeat ohb, StreamingContainerAgent sca)
1315   {
1316     OperatorHeartbeat.DeployState ds = null;
1317     if (ohb != null) {
1318       ds = ohb.getState();
1319     }
1320 
1321     LOG.debug("heartbeat {} {}/{} {}", oper, oper.getState(), ds, oper.getContainer().getExternalId());
1322 
1323     switch (oper.getState()) {
1324       case ACTIVE:
1325         // Commented out the warning below because it's expected when the operator does something
1326         // quickly and goes out of commission, it will report SHUTDOWN correcly whereas this code
1327         // is incorrectly expecting ACTIVE to be reported.
1328         //LOG.warn("status out of sync {} expected {} remote {}", oper, oper.getState(), ds);
1329         // operator expected active, check remote status
1330         if (ds == null) {
1331           sca.deployOpers.add(oper);
1332         }
1333         else {
1334           switch (ds) {
1335             case SHUTDOWN:
1336               // schedule operator deactivation against the windowId
1337               // will be processed once window is committed and all dependent operators completed processing
1338               long windowId = oper.stats.currentWindowId.get();
1339               if (ohb.windowStats != null && !ohb.windowStats.isEmpty()) {
1340                 windowId = ohb.windowStats.get(ohb.windowStats.size()-1).windowId;
1341               }
1342               LOG.debug("Operator {} deactivated at window {}", oper, windowId);
1343               synchronized (this.shutdownOperators) {
1344                 Set<PTOperator> deactivatedOpers = this.shutdownOperators.get(windowId);
1345                 if (deactivatedOpers == null) {
1346                   this.shutdownOperators.put(windowId, deactivatedOpers = new HashSet<>());
1347                 }
1348                 deactivatedOpers.add(oper);
1349               }
1350               sca.undeployOpers.add(oper.getId());
1351               // record operator stop event
1352               recordEventAsync(new StramEvent.StopOperatorEvent(oper.getName(), oper.getId(), oper.getContainer().getExternalId()));
1353               break;
1354             case FAILED:
1355               processOperatorFailure(oper);
1356               sca.undeployOpers.add(oper.getId());
1357               recordEventAsync(new StramEvent.StopOperatorEvent(oper.getName(), oper.getId(), oper.getContainer().getExternalId()));
1358               break;
1359             case ACTIVE:
1360               break;
1361           }
1362         }
1363         break;
1364       case PENDING_UNDEPLOY:
1365         if (ds == null) {
1366           // operator no longer deployed in container
1367           recordEventAsync(new StramEvent.StopOperatorEvent(oper.getName(), oper.getId(), oper.getContainer().getExternalId()));
1368           oper.setState(State.PENDING_DEPLOY);
1369           sca.deployOpers.add(oper);
1370         }
1371         else {
1372           // operator is currently deployed, request undeploy
1373           sca.undeployOpers.add(oper.getId());
1374         }
1375         break;
1376       case PENDING_DEPLOY:
1377         if (ds == null) {
1378           // operator to be deployed
1379           sca.deployOpers.add(oper);
1380         }
1381         else {
1382           // operator was deployed in container
1383           PTContainer container = oper.getContainer();
1384           LOG.debug("{} marking deployed: {} remote status {}", container.getExternalId(), oper, ds);
1385           oper.setState(PTOperator.State.ACTIVE);
1386           oper.stats.lastHeartbeat = null; // reset on redeploy
1387           oper.stats.lastWindowIdChangeTms = clock.getTime();
1388           recordEventAsync(new StramEvent.StartOperatorEvent(oper.getName(), oper.getId(), container.getExternalId()));
1389         }
1390         break;
1391       default:
1392         //LOG.warn("Unhandled operator state {} {} remote {}", oper, oper.getState(), ds);
1393         if (ds != null) {
1394           // operator was removed and needs to be undeployed from container
1395           sca.undeployOpers.add(oper.getId());
1396           recordEventAsync(new StramEvent.StopOperatorEvent(oper.getName(), oper.getId(), oper.getContainer().getExternalId()));
1397         }
1398     }
1399   }

1401   private void processOperatorFailure(PTOperator oper)
1402   {
1403     // count failure transitions *->FAILED, applies to initialization as well as intermittent failures
1404     if (oper.getState() == PTOperator.State.ACTIVE) {
1405       oper.setState(PTOperator.State.INACTIVE);
1406       oper.failureCount++;
1407       oper.getOperatorMeta().getStatus().failureCount++;
1408       LOG.warn("Operator failure: {} count: {}", oper, oper.failureCount);
1409       Integer maxAttempts = oper.getOperatorMeta().getValue(OperatorContext.RECOVERY_ATTEMPTS);
1410       if (maxAttempts == null || oper.failureCount <= maxAttempts) {
1411         // restart entire container in attempt to recover operator
1412         // in the future a more sophisticated recovery strategy could
1413         // involve initial redeploy attempt(s) of affected operator in
1414         // existing container or sandbox container for just the operator
1415         LOG.error("Initiating container restart after operator failure {}", oper);
1416         containerStopRequests.put(oper.getContainer().getExternalId(), oper.getContainer().getExternalId());
1417       }
1418       else {
1419         String msg = String.format("Shutdown after reaching failure threshold for %s", oper);
1420         LOG.warn(msg);
1421         shutdownAllContainers(msg);
1422         forcedShutdown = true;
1423       }
1424     }
1425     else {
1426       // should not get here
1427       LOG.warn("Failed operator {} {} {} to be undeployed by container", oper, oper.getState());
1428     }

1901   /**
1902    * Compute checkpoints required for a given operator instance to be recovered.
1903    * This is done by looking at checkpoints available for downstream dependencies first,
1904    * and then selecting the most recent available checkpoint that is smaller than downstream.
1905    *
1906    * @param operator Operator instance for which to find recovery checkpoint
1907    * @param ctx      Context into which to collect traversal info
1908    */
1909   public void updateRecoveryCheckpoints(PTOperator operator, UpdateCheckpointsContext ctx)
1910   {
1911     if (operator.getRecoveryCheckpoint().windowId < ctx.committedWindowId.longValue()) {
1912       ctx.committedWindowId.setValue(operator.getRecoveryCheckpoint().windowId);
1913     }
1914 
1915     if (operator.getState() == PTOperator.State.ACTIVE && (ctx.currentTms - operator.stats.lastWindowIdChangeTms) > operator.stats.windowProcessingTimeoutMillis) {
1916       // if the checkpoint is ahead, then it is not blocked but waiting for activation (state-less recovery, at-most-once)
1917       if (ctx.committedWindowId.longValue() >= operator.getRecoveryCheckpoint().windowId) {
1918         ctx.blocked.add(operator);
1919       }
1920     }
1921 
1922     long maxCheckpoint = operator.getRecentCheckpoint().windowId;
1923     if (ctx.recovery && maxCheckpoint == Stateless.WINDOW_ID && operator.isOperatorStateLess()) {
1924       long currentWindowId = WindowGenerator.getWindowId(ctx.currentTms, this.vars.windowStartMillis, this.getLogicalPlan().getValue(LogicalPlan.STREAMING_WINDOW_SIZE_MILLIS));
1925       maxCheckpoint = currentWindowId;
1926     }
1927 
1928     // DFS downstream operators
1929     for (PTOperator.PTOutput out : operator.getOutputs()) {
1930       for (PTOperator.PTInput sink : out.sinks) {
1931         PTOperator sinkOperator = sink.target;
1932         if (!ctx.visited.contains(sinkOperator)) {
1933           // downstream traversal
1934           updateRecoveryCheckpoints(sinkOperator, ctx);
1935         }
1936         // recovery window id cannot move backwards
1937         // when dynamically adding new operators
1938         if (sinkOperator.getRecoveryCheckpoint().windowId >= operator.getRecoveryCheckpoint().windowId) {
1939           maxCheckpoint = Math.min(maxCheckpoint, sinkOperator.getRecoveryCheckpoint().windowId);
1940         }
1941 
1942         if (ctx.blocked.contains(sinkOperator)) {
1943           if (sinkOperator.stats.getCurrentWindowId() == operator.stats.getCurrentWindowId()) {
1944             // downstream operator is blocked by this operator
1945             ctx.blocked.remove(sinkOperator);
1946           }
1947         }
1948       }
1949     }
1950 
1951     // checkpoint frozen during deployment
1952     if (ctx.recovery || operator.getState() != PTOperator.State.PENDING_DEPLOY) {
1953       // remove previous checkpoints
1954       Checkpoint c1 = Checkpoint.INITIAL_CHECKPOINT;
1955       synchronized (operator.checkpoints) {
1956         if (!operator.checkpoints.isEmpty() && (operator.checkpoints.getFirst()).windowId <= maxCheckpoint) {
1957           c1 = operator.checkpoints.getFirst();
1958           Checkpoint c2;
1959           while (operator.checkpoints.size() > 1 && ((c2 = operator.checkpoints.get(1)).windowId) <= maxCheckpoint) {
1960             operator.checkpoints.removeFirst();
1961             //LOG.debug("Checkpoint to delete: operator={} windowId={}", operator.getName(), c1);
1962             this.purgeCheckpoints.add(new Pair<PTOperator, Long>(operator, c1.windowId));
1963             c1 = c2;
1964           }
1965         }
1966         else {
1967           if (ctx.recovery && operator.checkpoints.isEmpty() && operator.isOperatorStateLess()) {
1968             LOG.debug("Adding checkpoint for stateless operator {} {}", operator, Codec.getStringWindowId(maxCheckpoint));
1969             c1 = operator.addCheckpoint(maxCheckpoint, this.vars.windowStartMillis);
1970           }
1971         }
1972       }
1973       //LOG.debug("Operator {} checkpoints: commit {} recent {}", new Object[] {operator.getName(), c1, operator.checkpoints});
1974       operator.setRecoveryCheckpoint(c1);
1975     }
1976     else {
1977       LOG.debug("Skipping checkpoint update {} during {}", operator, operator.getState());
1978     }
1979 
1980     ctx.visited.add(operator);
1981   }

2884    */
2885   public static StreamingContainerManager getInstance(RecoveryHandler rh, LogicalPlan dag, boolean enableEventRecording) throws IOException
2886   {
2887     try {
2888       CheckpointState checkpointedState = (CheckpointState) rh.restore();
2889       StreamingContainerManager scm;
2890       if (checkpointedState == null) {
2891         scm = new StreamingContainerManager(dag, enableEventRecording, new SystemClock());
2892       }
2893       else {
2894         // find better way to support final transient members
2895         PhysicalPlan plan = checkpointedState.physicalPlan;
2896         plan.getLogicalPlan().setAttribute(LogicalPlan.APPLICATION_ATTEMPT_ID, dag.getAttributes().get(LogicalPlan.APPLICATION_ATTEMPT_ID));
2897         scm = new StreamingContainerManager(checkpointedState, enableEventRecording);
2898         for (Field f : plan.getClass().getDeclaredFields()) {
2899           if (f.getType() == PlanContext.class) {
2900             f.setAccessible(true);
2901             try {
2902               f.set(plan, scm);
2903             }
2904             catch (Exception e) {
2905               throw new RuntimeException("Failed to set " + f, e);
2906             }
2907             f.setAccessible(false);
2908           }
2909         }
2910         DataInputStream logStream = rh.getLog();
2911         scm.journal.replay(logStream);
2912         logStream.close();
2913 
2914         // restore checkpoint info
2915         plan.syncCheckpoints(scm.vars.windowStartMillis, scm.clock.getTime());
2916         scm.committedWindowId = scm.updateCheckpoints(true);
2917 
2918         // at this point the physical plan has been fully restored
2919         // populate container agents for existing containers
2920         for (PTContainer c : plan.getContainers()) {
2921           if (c.getExternalId() != null) {
2922             LOG.debug("Restore container agent {} for {}", c.getExternalId(), c);
2923             StreamingContainerAgent sca = new StreamingContainerAgent(c, scm.newStreamingContainerContext(c), scm);
2924             scm.containers.put(c.getExternalId(), sca);
