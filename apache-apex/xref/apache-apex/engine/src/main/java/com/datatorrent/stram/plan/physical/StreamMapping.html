51  /**
52   * Encapsulates the mapping of input to output operators, including unifiers. Depending on logical plan setting and
53   * number of partitions, unifiers are created as needed and potentially cascaded.
54   *
55   * @since 0.9.0
56   */
57  public class StreamMapping implements java.io.Serializable
58  {
59    private static final long serialVersionUID = 8572852828117485193L;
60  
61    private final static Logger LOG = LoggerFactory.getLogger(StreamMapping.class);
62  
63    private final StreamMeta streamMeta;
64    private final PhysicalPlan plan;
65    PTOperator finalUnifier;
66    final Set<PTOperator> cascadingUnifiers = Sets.newHashSet();
67    final Set<PTOperator> slidingUnifiers = Sets.newHashSet();
68    private final List<PTOutput> upstream = Lists.newArrayList();
69  
70  
71    public StreamMapping(StreamMeta streamMeta, PhysicalPlan plan) {
72      this.streamMeta = streamMeta;
73      this.plan = plan;
74    }
75  
76    void addTo(Collection<PTOperator> opers) {
77      if (finalUnifier != null) {
78        opers.add(finalUnifier);
79      }
80      opers.addAll(cascadingUnifiers);
81      opers.addAll(slidingUnifiers);
82    }
83  
84    public void setSources(Collection<PTOperator> partitions) {
85      upstream.clear();
86      // add existing inputs
87      for (PTOperator uoper : partitions) {
88        for (PTOutput source : uoper.outputs) {
89          if (source.logicalStream == streamMeta) {
90            upstream.add(source);
91          }
92        }
93      }
94      redoMapping();
95    }

97    public static PTOperator createSlidingUnifier(StreamMeta streamMeta, PhysicalPlan plan, int operatorApplicationWindowCount, int slidingWindowCount)
98    {
99      int gcd = IntMath.gcd(operatorApplicationWindowCount, slidingWindowCount);
100     OperatorMeta um = streamMeta.getSource().getSlidingUnifier(operatorApplicationWindowCount / gcd, gcd, slidingWindowCount / gcd);
101     PTOperator pu = plan.newOperator(um, um.getName());
102 
103     Operator unifier = um.getOperator();
104     PortMappingDescriptor mergeDesc = new PortMappingDescriptor();
105     Operators.describe(unifier, mergeDesc);
106     if (mergeDesc.outputPorts.size() != 1) {
107       throw new AssertionError("Unifier must have a single output port, instead found : " + mergeDesc.outputPorts);
108     }
109     pu.unifiedOperatorMeta = streamMeta.getSource().getOperatorMeta();
110     pu.outputs.add(new PTOutput(mergeDesc.outputPorts.keySet().iterator().next(), streamMeta, pu));
111     plan.newOpers.put(pu, unifier);
112     return pu;
113   }

115   public static PTOperator createUnifier(StreamMeta streamMeta, PhysicalPlan plan)
116   {
117     OperatorMeta um = streamMeta.getSource().getUnifierMeta();
118     PTOperator pu = plan.newOperator(um, um.getName());
119 
120     Operator unifier = um.getOperator();
121     PortMappingDescriptor mergeDesc = new PortMappingDescriptor();
122     Operators.describe(unifier, mergeDesc);
123     if (mergeDesc.outputPorts.size() != 1) {
124       throw new AssertionError("Unifier must have a single output port, instead found : " + mergeDesc.outputPorts);
125     }
126 
127     pu.unifiedOperatorMeta = streamMeta.getSource().getOperatorMeta();
128     pu.outputs.add(new PTOutput(mergeDesc.outputPorts.keySet().iterator().next(), streamMeta, pu));
129     plan.newOpers.put(pu, unifier);
130     return pu;
131   }

133   private void addSlidingUnifiers()
134   {
135     OperatorMeta sourceOM = streamMeta.getSource().getOperatorMeta();
136     if (sourceOM.getAttributes().contains(Context.OperatorContext.SLIDE_BY_WINDOW_COUNT)) {
137       if (sourceOM.getValue(Context.OperatorContext.SLIDE_BY_WINDOW_COUNT) <
138         sourceOM.getValue(Context.OperatorContext.APPLICATION_WINDOW_COUNT)) {
139         plan.undeployOpers.addAll(slidingUnifiers);
140         slidingUnifiers.clear();
141         List<PTOutput> newUpstream = Lists.newArrayList();
142         PTOperator slidingUnifier;
143         for (PTOutput source : upstream) {
144           slidingUnifier = StreamMapping.createSlidingUnifier(streamMeta, plan,
145             sourceOM.getValue(Context.OperatorContext.APPLICATION_WINDOW_COUNT),
146             sourceOM.getValue(Context.OperatorContext.SLIDE_BY_WINDOW_COUNT));
147           addInput(slidingUnifier, source, null);
148           this.slidingUnifiers.add(slidingUnifier);
149           newUpstream.add(slidingUnifier.outputs.get(0));
150         }
151         upstream.clear();
152         upstream.addAll(newUpstream);
153       }
154       else {
155         LOG.warn("Sliding Window Count {} should be less than APPLICATION WINDOW COUNT {}", sourceOM.getValue(Context.OperatorContext.SLIDE_BY_WINDOW_COUNT), sourceOM.getValue(Context.OperatorContext.APPLICATION_WINDOW_COUNT));
156       }
157     }
158   }

160   @SuppressWarnings("AssignmentToForLoopParameter")
161   private List<PTOutput> setupCascadingUnifiers(List<PTOutput> upstream, List<PTOperator> pooledUnifiers, int limit, int level) {
162     List<PTOutput> nextLevel = Lists.newArrayList();
163     PTOperator pu = null;
164     for (int i=0; i<upstream.size(); i++) {
165       if (i % limit == 0) {
166         if (upstream.size() - i < limit) {
167           while(i< upstream.size()) {
168             nextLevel.add(upstream.get(i));
169             i++;
170           }
171           continue;
172         }
173         if (!pooledUnifiers.isEmpty()) {
174           pu = pooledUnifiers.remove(0);
175         } else {
176           pu = createUnifier(streamMeta, plan);
177         }
178         assert (pu.outputs.size() == 1) : "unifier has single output";
179         nextLevel.addAll(pu.outputs);
180         this.cascadingUnifiers.add(pu);
181       }
182 
183       PTOutput source = upstream.get(i);
184       addInput(pu, source, null);
185     }
186 
187     if (nextLevel.size() > limit) {
188       return setupCascadingUnifiers(nextLevel, pooledUnifiers, limit, level);
189     } else {
190       return nextLevel;
191     }
192   }

194   /**
195    * rebuild the tree, which may cause more changes to execution layer than need be
196    * TODO: investigate incremental logic
197    */
198   private void redoMapping() {
199 
200     Set<Pair<PTOperator, InputPortMeta>> downstreamOpers = Sets.newHashSet();
201 
202     // figure out the downstream consumers
203     for (InputPortMeta ipm : streamMeta.getSinks()) {
204       // gets called prior to all logical operators mapped
205       // skipped for parallel partitions - those are handled elsewhere
206       if (!ipm.getValue(PortContext.PARTITION_PARALLEL) && plan.hasMapping(ipm.getOperatorWrapper())) {
207         List<PTOperator> partitions = plan.getOperators(ipm.getOperatorWrapper());
208         for (PTOperator doper : partitions) {
209           downstreamOpers.add(new Pair<PTOperator, InputPortMeta>(doper, ipm));
210         }
211       }
212     }
213 
214     if (!downstreamOpers.isEmpty()) {
215       // unifiers are required
216       for (PTOperator unifier : this.cascadingUnifiers) {
217         detachUnifier(unifier);
218       }
219       if (this.finalUnifier != null) {
220         detachUnifier(finalUnifier);
221       }
222 
223       List<PTOperator> currentUnifiers = Lists.newArrayList(this.cascadingUnifiers);
224       this.cascadingUnifiers.clear();
225       plan.undeployOpers.addAll(currentUnifiers);
226       addSlidingUnifiers();
227 
228       int limit = streamMeta.getSource().getValue(PortContext.UNIFIER_LIMIT);
229 
230       boolean separateUnifiers = false;
231       Integer lastId = null;
232       for (InputPortMeta ipm : streamMeta.getSinks()) {
233         StreamCodec<?> streamCodecInfo = StreamingContainerAgent.getStreamCodec(ipm);
234         Integer id = plan.getStreamCodecIdentifier(streamCodecInfo);
235         if (lastId == null) {
236           lastId = id;
237         } else if (!id.equals(lastId)) {
238           separateUnifiers = true;
239           break;
240         }
241       }
242 
243       List<PTOutput> unifierSources = this.upstream;
244       Map<StreamCodec<?>, List<PTOutput>> cascadeUnifierSourcesMap = Maps.newHashMap();
245 
246       if (limit > 1 && this.upstream.size() > limit) {
247         // cascading unifier
248         if (!separateUnifiers) {
249           unifierSources = setupCascadingUnifiers(this.upstream, currentUnifiers, limit, 0);
250         } else {
251           for (InputPortMeta ipm : streamMeta.getSinks()) {
252             StreamCodec<?> streamCodecInfo = StreamingContainerAgent.getStreamCodec(ipm);
253             if (!cascadeUnifierSourcesMap.containsKey(streamCodecInfo)) {
254               unifierSources = setupCascadingUnifiers(this.upstream, currentUnifiers, limit, 0);
255               cascadeUnifierSourcesMap.put(streamCodecInfo, unifierSources);
256             }
257           }
258         }
259       }
260 
261       // remove remaining unifiers
262       for (PTOperator oper : currentUnifiers) {
263         plan.removePTOperator(oper);
264       }
265 
266       // Directly getting attribute from map to know if it is set or not as it can be overriden by the input
267       Boolean sourceSingleFinal = streamMeta.getSource().getAttributes().get(PortContext.UNIFIER_SINGLE_FINAL);
268 
269       // link the downstream operators with the unifiers
270       for (Pair<PTOperator, InputPortMeta> doperEntry : downstreamOpers) {
271 
272         Map<LogicalPlan.InputPortMeta, PartitionKeys> partKeys = doperEntry.first.partitionKeys;
273         PartitionKeys pks = partKeys != null ? partKeys.get(doperEntry.second) : null;
274         Boolean sinkSingleFinal = doperEntry.second.getAttributes().get(PortContext.UNIFIER_SINGLE_FINAL);
275         boolean lastSingle = (sinkSingleFinal != null) ? sinkSingleFinal :
276                                 (sourceSingleFinal != null ? sourceSingleFinal.booleanValue() : PortContext.UNIFIER_SINGLE_FINAL.defaultValue);
277 
278         if (upstream.size() > 1) {
279           if (!separateUnifiers && ((pks == null || pks.mask == 0) || lastSingle)) {
280             if (finalUnifier == null) {
281               finalUnifier = createUnifier(streamMeta, plan);
282             }
283             setInput(doperEntry.first, doperEntry.second, finalUnifier, (pks == null) || (pks.mask == 0) ? null : pks);
284             if (finalUnifier.inputs.isEmpty()) {
285               // set unifier inputs once, regardless how many downstream operators there are
286               for (PTOutput out : unifierSources) {
287                 addInput(this.finalUnifier, out, null);
288               }
289             }
290           } else {
291             // MxN partitioning: unifier per downstream partition
292             LOG.debug("MxN unifier for {} {} {}", new Object[] {doperEntry.first, doperEntry.second.getPortName(), pks});
293             PTOperator unifier = doperEntry.first.upstreamMerge.get(doperEntry.second);
294             if (unifier == null) {
295               unifier = createUnifier(streamMeta, plan);
296               doperEntry.first.upstreamMerge.put(doperEntry.second, unifier);
297               setInput(doperEntry.first, doperEntry.second, unifier, null);
298             }
299             // sources may change dynamically, rebuild inputs (as for cascading unifiers)
300             for (PTInput in : unifier.inputs) {
301               in.source.sinks.remove(in);
302             }
303             unifier.inputs.clear();
304             List<PTOutput> doperUnifierSources = unifierSources;
305             if (separateUnifiers) {
306               StreamCodec<?> streamCodecInfo = StreamingContainerAgent.getStreamCodec(doperEntry.second);
307               List<PTOutput> cascadeSources = cascadeUnifierSourcesMap.get(streamCodecInfo);
308               if (cascadeSources != null) {
309                 doperUnifierSources = cascadeSources;
310               }
311             }
312             // add new inputs
313             for (PTOutput out : doperUnifierSources) {
314               addInput(unifier, out, pks);
315             }
316           }
317         } else {
318           // no partitioning
319           PTOperator unifier = doperEntry.first.upstreamMerge.remove(doperEntry.second);
320           if (unifier != null) {
321             plan.removePTOperator(unifier);
322           }
323           setInput(doperEntry.first, doperEntry.second, upstream.get(0).source, pks);
324         }
325       }
326 
327       // Remove the unattached final unifier
328       // Unattached final unifier is from
329       // 1) Upstream operator partitions are scaled down to one. (no unifier needed)
330       // 2) Downstream operators partitions are scaled up from one to multiple. (replaced by merged unifier)
331       if (finalUnifier != null && finalUnifier.inputs.isEmpty()) {
332           plan.removePTOperator(finalUnifier);
333           finalUnifier = null;
334       }
335 
336     }
337 
338   }

340   private void setInput(PTOperator oper, InputPortMeta ipm, PTOperator sourceOper, PartitionKeys pks) {
341     // TODO: see if this can be handled more efficiently
342     for (PTInput in : oper.inputs) {
343       if (in.source.source == sourceOper && in.logicalStream == streamMeta && ipm.getPortName().equals(in.portName)) {
344         return;
345       }
346     }
347     // link to upstream output(s) for this stream
348     for (PTOutput upstreamOut : sourceOper.outputs) {
349       if (upstreamOut.logicalStream == streamMeta) {
350         PTInput input = new PTInput(ipm.getPortName(), streamMeta, oper, pks, upstreamOut);
351         oper.inputs.add(input);
352       }
353     }
354   }

356   public static void addInput(PTOperator target, PTOutput upstreamOut, PartitionKeys pks)
357   {
358     StreamMeta lStreamMeta = upstreamOut.logicalStream;
359     PTInput input = new PTInput("<merge#" + lStreamMeta.getSource().getPortName() + ">", lStreamMeta, target, pks, upstreamOut);
360     target.inputs.add(input);
361   }
362 
363   private void detachUnifier(PTOperator unifier) {
364     // remove existing unifiers from downstream inputs
365     for (PTOutput out : unifier.outputs) {
366       for (PTInput input : out.sinks) {
367         input.target.inputs.remove(input);
368       }
369       out.sinks.clear();
370     }
371     // remove from upstream outputs
372     for (PTInput in : unifier.inputs) {
373       in.source.sinks.remove(in);
374     }
375     unifier.inputs.clear();
376   }
