127   Boolean hasAccess(HttpServletRequest request)
128   {
129     String remoteUser = request.getRemoteUser();
130     if (remoteUser != null) {
131       UserGroupInformation callerUGI = UserGroupInformation.createRemoteUser(remoteUser);
132       if (callerUGI != null) {
133         return false;
134       }
135     }
136     return true;
137   }

139   @SuppressWarnings({"rawtypes", "unchecked"})
140   private void init()
141   {
142     //clear content type
143     httpResponse.setContentType(null);
144     if (!initialized) {
145       Map<Class<?>, Class<? extends StringCodec<?>>> codecs = dagManager.getApplicationAttributes().get(DAGContext.STRING_CODECS);
146       StringCodecs.loadConverters(codecs);
147       if (codecs != null) {
148         SimpleModule sm = new SimpleModule("DTSerializationModule", new Version(1, 0, 0, null));
149         for (Map.Entry<Class<?>, Class<? extends StringCodec<?>>> entry : codecs.entrySet()) {
150           try {
151             final StringCodec<Object> codec = (StringCodec<Object>)entry.getValue().newInstance();
152             sm.addSerializer(new SerializerBase(entry.getKey())
153             {
154               @Override
155               public void serialize(Object value, JsonGenerator jgen, SerializerProvider provider) throws IOException, JsonProcessingException
156               {
157                 jgen.writeString(codec.toString(value));
158               }
159 
160             });
161           }
162           catch (Exception ex) {
163             LOG.error("Caught exception when instantiating codec for class {}", entry.getKey().getName(), ex);
164           }
165         }
166 
167         objectMapper.registerModule(sm);
168       }
169       initialized = true;
170     }
171   }

280   public JSONObject getOperatorClasses(@QueryParam("q") String searchTerm, @QueryParam("parent") String parent)
281   {
282     init();
283     JSONObject result = new JSONObject();
284     JSONArray classNames = new JSONArray();
285 
286     if (parent != null) {
287       if (parent.equals("chart")) {
288         parent = "com.datatorrent.lib.chart.ChartOperator";
289       }
290       else if (parent.equals("filter")) {
291         parent = "com.datatorrent.common.util.SimpleFilterOperator";
292       }
293     }
294 
295     try {
296       Set<String> operatorClasses = operatorDiscoverer.getOperatorClasses(parent, searchTerm);
297 
298       for (String clazz : operatorClasses) {
299         JSONObject j = new JSONObject();
300         j.put("name", clazz);
301         classNames.put(j);
302       }
303 
304       result.put("operatorClasses", classNames);
305     }
306     catch (ClassNotFoundException ex) {
307       throw new NotFoundException();
308     }
309     catch (JSONException ex) {
310     }
311     return result;
312   }

349   private static String getTupleRecordingId()
350   {
351     SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddHHmm");
352     String val = sdf.format(new Date());
353     val += "-";
354     byte[] r = new byte[4];
355     new Random().nextBytes(r);
356     val += Base64.encodeBase64URLSafeString(r);
357     return val;
358   }

422   @GET
423   @Path(PATH_PHYSICAL_PLAN_CONTAINERS)
424   @Produces(MediaType.APPLICATION_JSON)
425   public JSONObject listContainers(@QueryParam("states") String states) throws Exception
426   {
427     init();
428     Set<String> stateSet = null;
429     if (states != null) {
430       stateSet = new HashSet<String>();
431       stateSet.addAll(Arrays.asList(StringUtils.split(states, ',')));
432     }
433     ContainersInfo ci = new ContainersInfo();
434     for (ContainerInfo containerInfo : dagManager.getCompletedContainerInfo()) {
435       if (stateSet == null || stateSet.contains(containerInfo.state)) {
436         ci.add(containerInfo);
437       }
438     }
439 
440     Collection<StreamingContainerAgent> containerAgents = dagManager.getContainerAgents();
441     // add itself (app master container)
442     ContainerInfo appMasterContainerInfo = dagManager.getAppMasterContainerInfo();
443     if (stateSet == null || stateSet.contains(appMasterContainerInfo.state)) {
444       ci.add(appMasterContainerInfo);
445     }
446     for (StreamingContainerAgent sca : containerAgents) {
447       ContainerInfo containerInfo = sca.getContainerInfo();
448       if (stateSet == null || stateSet.contains(containerInfo.state)) {
449         ci.add(containerInfo);
450       }
451     }
452     // To get around the nasty JAXB problem for lists
453     return new JSONObject(objectMapper.writeValueAsString(ci));
454   }

456   @GET
457   @Path(PATH_PHYSICAL_PLAN_CONTAINERS + "/{containerId}")
458   @Produces(MediaType.APPLICATION_JSON)
459   public JSONObject getContainer(@PathParam("containerId") String containerId) throws Exception
460   {
461     init();
462     ContainerInfo ci = null;
463     if (containerId.equals(System.getenv(ApplicationConstants.Environment.CONTAINER_ID.toString()))) {
464       ci = dagManager.getAppMasterContainerInfo();
465     }
466     else {
467       for (ContainerInfo containerInfo : dagManager.getCompletedContainerInfo()) {
468         if (containerInfo.id.equals(containerId)) {
469           ci = containerInfo;
470         }
471       }
472       if (ci == null) {
473         StreamingContainerAgent sca = dagManager.getContainerAgent(containerId);
474         if (sca == null) {
475           throw new NotFoundException();
476         }
477         ci = sca.getContainerInfo();
478       }
479     }
480     return new JSONObject(objectMapper.writeValueAsString(ci));
481   }

579     }

793   @POST // not supported by WebAppProxyServlet, can only be called directly
794   @Path(PATH_LOGICAL_PLAN)
795   @Consumes(MediaType.APPLICATION_JSON)
796   @Produces(MediaType.APPLICATION_JSON)
797   public JSONObject logicalPlanModification(JSONObject request)
798   {
799     init();
800     JSONObject response = new JSONObject();
801     try {
802       JSONArray jsonArray = request.getJSONArray("requests");
803       List<LogicalPlanRequest> requests = new ArrayList<LogicalPlanRequest>();
804       for (int i = 0; i < jsonArray.length(); i++) {
805         JSONObject jsonObj = (JSONObject)jsonArray.get(i);
806         LogicalPlanRequest requestObj = (LogicalPlanRequest)Class.forName(LogicalPlanRequest.class.getPackage().getName() + "." + jsonObj.getString("requestType")).newInstance();
807         @SuppressWarnings("unchecked")
808         Map<String, String> properties = BeanUtils.describe(requestObj);
809         @SuppressWarnings("unchecked")
810         Iterator<String> keys = jsonObj.keys();
811 
812         while (keys.hasNext()) {
813           String key = keys.next();
814           if (!key.equals("requestType")) {
815             properties.put(key, jsonObj.get(key).toString());
816           }
817         }
818         BeanUtils.populate(requestObj, properties);
819         requests.add(requestObj);
820       }
821       Future<?> fr = dagManager.logicalPlanModification(requests);
822       fr.get(3000, TimeUnit.MILLISECONDS);
823     }
824     catch (Exception ex) {
825       LOG.error("Error processing plan change", ex);
826       try {
827         if (ex instanceof ExecutionException) {
828           response.put("error", ex.getCause().toString());
829         }
830         else {
831           response.put("error", ex.toString());
832         }
833       }
834       catch (Exception e) {
835         // ignore
836       }
837     }
838 
839     return response;
840   }

843   @Path(PATH_LOGGERS)
844   @Consumes(MediaType.APPLICATION_JSON)
845   @Produces(MediaType.APPLICATION_JSON)
846   public JSONObject setLoggersLevel(JSONObject request)
847   {
848     init();
849     JSONObject response = new JSONObject();
850     Map<String, String> targetChanges = Maps.newHashMap();
851     try {
852       @SuppressWarnings("unchecked")
853       JSONArray loggerArray = request.getJSONArray("loggers");
854       for (int i = 0; i < loggerArray.length(); i++) {
855         JSONObject loggerNode = loggerArray.getJSONObject(i);
856         String target = loggerNode.getString("target");
857         String level = loggerNode.getString("logLevel");
858         if (ConfigValidator.validateLoggersLevel(target, level)) {
859           LOG.info("changing logger level for {} to {}", target, level);
860           targetChanges.put(target, level);
861         }
862         else {
863           LOG.warn("incorrect logger settings {}:{}", target, level);
864         }
865       }
866 
867       if (!targetChanges.isEmpty()) {
868         dagManager.setLoggersLevel(Collections.unmodifiableMap(targetChanges));
869         //Changing the levels on Stram after sending the message to all containers.
870         DTLoggerFactory.getInstance().changeLoggersLevel(targetChanges);
871       }
872     }
873     catch (JSONException ex) {
874       LOG.warn("Got JSON Exception: ", ex);
875     }
876     return response;
877   }
