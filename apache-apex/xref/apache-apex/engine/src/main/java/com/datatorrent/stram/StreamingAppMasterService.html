452   /**
453    * Dump out contents of $CWD and the environment to stdout for debugging
454    */
455   @SuppressWarnings("UseOfSystemOutOrSystemErr")
456   public void dumpOutDebugInfo()
457   {
458     LOG.info("Dump debug output");
459     Map<String, String> envs = System.getenv();
460     LOG.info("\nDumping System Env: begin");
461     for (Map.Entry<String, String> env : envs.entrySet()) {
462       LOG.info("System env: key=" + env.getKey() + ", val=" + env.getValue());
463     }
464     LOG.info("Dumping System Env: end");
465 
466     String cmd = "ls -al";
467     Runtime run = Runtime.getRuntime();
468     Process pr;
469     try {
470       pr = run.exec(cmd);
471       pr.waitFor();
472 
473       BufferedReader buf = new BufferedReader(new InputStreamReader(pr.getInputStream()));
474       String line;
475       LOG.info("\nDumping files in local dir: begin");
476       try {
477         while ((line = buf.readLine()) != null) {
478           LOG.info("System CWD content: " + line);
479         }
480         LOG.info("Dumping files in local dir: end");
481       } finally {
482         buf.close();
483       }
484     } catch (IOException e) {
485       LOG.debug("Exception", e);
486     } catch (InterruptedException e) {
487       LOG.info("Interrupted", e);
488     }
489 
490     LOG.info("Classpath: {}", System.getProperty("java.class.path"));
491     LOG.info("Config resources: {}", getConfig().toString());
492     try {
493       // find a better way of logging this using the logger.
494       Configuration.dumpConfiguration(getConfig(), new PrintWriter(System.out));
495     } catch (Exception e) {
496       LOG.error("Error dumping configuration.", e);
497     }
498   }

500   @Override
501   protected void serviceInit(Configuration conf) throws Exception
502   {
503     LOG.info("Application master" + ", appId=" + appAttemptID.getApplicationId().getId() + ", clustertimestamp=" + appAttemptID.getApplicationId().getClusterTimestamp() + ", attemptId=" + appAttemptID.getAttemptId());
504 
505     FileInputStream fis = new FileInputStream("./" + LogicalPlan.SER_FILE_NAME);
506     try {
507       this.dag = LogicalPlan.read(fis);
508     } finally {
509       fis.close();
510     }
511     // "debug" simply dumps all data using LOG.info
512     if (dag.isDebug()) {
513       dumpOutDebugInfo();
514     }
515     dag.setAttribute(LogicalPlan.APPLICATION_ATTEMPT_ID, appAttemptID.getAttemptId());
516     FSRecoveryHandler recoveryHandler = new FSRecoveryHandler(dag.assertAppPath(), conf);
517     this.dnmgr = StreamingContainerManager.getInstance(recoveryHandler, dag, true);
518     dag = this.dnmgr.getLogicalPlan();
519     this.appContext = new ClusterAppContextImpl(dag.getAttributes());
520 
521     Map<Class<?>, Class<? extends StringCodec<?>>> codecs = dag.getAttributes().get(DAG.STRING_CODECS);
522     StringCodecs.loadConverters(codecs);
523 
524     LOG.info("Starting application with {} operators in {} containers", dnmgr.getPhysicalPlan().getAllOperators().size(), dnmgr.getPhysicalPlan().getContainers().size());
525 
526     if (UserGroupInformation.isSecurityEnabled()) {
527       // TODO :- Need to perform token renewal
528       delegationTokenManager = new StramDelegationTokenManager(DELEGATION_KEY_UPDATE_INTERVAL, DELEGATION_TOKEN_MAX_LIFETIME, DELEGATION_TOKEN_RENEW_INTERVAL, DELEGATION_TOKEN_REMOVER_SCAN_INTERVAL);
529     }
530     this.nmClient = new NMClientAsyncImpl(new NMCallbackHandler());
531     addService(nmClient);
532     this.amRmClient = AMRMClient.createAMRMClient();
533     addService(amRmClient);
534 
535     // start RPC server
536     int rpcListenerCount = dag.getValue(DAGContext.HEARTBEAT_LISTENER_THREAD_COUNT);
537     this.heartbeatListener = new StreamingContainerParent(this.getClass().getName(), dnmgr, delegationTokenManager, rpcListenerCount);
538     addService(heartbeatListener);
539 
540     String appDataPushTransport = dag.getValue(LogicalPlan.METRICS_TRANSPORT);
541     if (appDataPushTransport != null) {
542       this.appDataPushAgent = new AppDataPushAgent(dnmgr, appContext);
543       addService(this.appDataPushAgent);
544     }
545     // initialize all services added above
546     super.serviceInit(conf);
547   }

549   @Override
550   protected void serviceStart() throws Exception
551   {
552     super.serviceStart();
553     if (UserGroupInformation.isSecurityEnabled()) {
554       delegationTokenManager.startThreads();
555     }
556 
557     // write the connect address for containers to DFS
558     InetSocketAddress connectAddress = NetUtils.getConnectAddress(this.heartbeatListener.getAddress());
559     URI connectUri = RecoverableRpcProxy.toConnectURI(connectAddress);
560     FSRecoveryHandler recoveryHandler = new FSRecoveryHandler(dag.assertAppPath(), getConfig());
561     recoveryHandler.writeConnectUri(connectUri.toString());
562 
563     // start web service
564     try {
565       org.mortbay.log.Log.setLog(null);
566     }
567     catch (Throwable throwable) {
568       // SPOI-2687. As part of Pivotal Certification, we need to catch ClassNotFoundException as Pivotal was using Jetty 7 where as other distros are using Jetty 6.
569       // LOG.error("can't set the log to null: ", throwable);
570     }
571 
572     try {
573       Configuration config = getConfig();
574       if (UserGroupInformation.isSecurityEnabled()) {
575         config = new Configuration(config);
576         config.set("hadoop.http.filter.initializers", StramWSFilterInitializer.class.getCanonicalName());
577       }
578       WebApp webApp = WebApps.$for("stram", StramAppContext.class, appContext, "ws").with(config).start(new StramWebApp(this.dnmgr));
579       LOG.info("Started web service at port: " + webApp.port());
580       this.appMasterTrackingUrl = NetUtils.getConnectAddress(webApp.getListenerAddress()).getHostName() + ":" + webApp.port();
581       LOG.info("Setting tracking URL to: " + appMasterTrackingUrl);
582     }
583     catch (Exception e) {
584       LOG.error("Webapps failed to start. Ignoring for now:", e);
585     }
586   }

606   public boolean run() throws Exception
607   {
608     boolean status = true;
609     try {
610       StreamingContainer.eventloop.start();
611       execute();
612     }
613     finally {
614       StreamingContainer.eventloop.stop();
615     }
616     return status;
617   }

619   /**
620    * Main run function for the application master
621    *
622    * @throws YarnException
623    */
624   @SuppressWarnings("SleepWhileInLoop")
625   private void execute() throws YarnException, IOException
626   {
627     LOG.info("Starting ApplicationMaster");
628     final Credentials credentials = UserGroupInformation.getCurrentUser().getCredentials();
629     LOG.info("number of tokens: {}", credentials.getAllTokens().size());
630     Iterator<Token<?>> iter = credentials.getAllTokens().iterator();
631     while (iter.hasNext()) {
632       Token<?> token = iter.next();
633       LOG.debug("token: {}", token);
634     }
635     final Configuration conf = getConfig();
636     long tokenLifeTime = (long)(dag.getValue(LogicalPlan.TOKEN_REFRESH_ANTICIPATORY_FACTOR) * Math.min(dag.getValue(LogicalPlan.HDFS_TOKEN_LIFE_TIME), dag.getValue(LogicalPlan.RM_TOKEN_LIFE_TIME)));
637     long expiryTime = System.currentTimeMillis() + tokenLifeTime;
638     LOG.debug(" expiry token time {}", tokenLifeTime);
639     String hdfsKeyTabFile = dag.getValue(LogicalPlan.KEY_TAB_FILE);
640 
641     // Register self with ResourceManager
642     RegisterApplicationMasterResponse response = amRmClient.registerApplicationMaster(appMasterHostname, 0, appMasterTrackingUrl);
643 
644     // Dump out information about cluster capability as seen by the resource manager
645     int maxMem = response.getMaximumResourceCapability().getMemory();
646     int maxVcores = response.getMaximumResourceCapability().getVirtualCores();
647     int minMem = conf.getInt("yarn.scheduler.minimum-allocation-mb", 0);
648     int minVcores = conf.getInt("yarn.scheduler.minimum-allocation-vcores", 0);
649     LOG.info("Max mem {}m, Min mem {}m, Max vcores {} and Min vcores {} capabililty of resources in this cluster ", maxMem, minMem, maxVcores, minVcores);
650 
651     int maxConsecutiveContainerFailures = conf.getInt("MAX_CONSECUTIVE_CONTAINER_FAILURES", MAX_CONTAINER_FAILURES_PER_NODE);
652     long blacklistRemovalTime = conf.getLong("BLACKLIST_REMOVAL_TIME", BLACKLIST_REMOVAL_TIME);
653 
654     // for locality relaxation fall back
655     Map<StreamingContainerAgent.ContainerStartRequest, MutablePair<Integer, ContainerRequest>> requestedResources = Maps.newHashMap();
656 
657     // Setup heartbeat emitter
658     // TODO poll RM every now and then with an empty request to let RM know that we are alive
659     // The heartbeat interval after which an AM is timed out by the RM is defined by a config setting:
660     // RM_AM_EXPIRY_INTERVAL_MS with default defined by DEFAULT_RM_AM_EXPIRY_INTERVAL_MS
661     // The allocate calls to the RM count as heartbeat so, for now, this additional heartbeat emitter
662     // is not required.
663 
664     int loopCounter = -1;
665     List<ContainerId> releasedContainers = new ArrayList<ContainerId>();
666     int numTotalContainers = 0;
667     // keep track of already requested containers to not request them again while waiting for allocation
668     int numRequestedContainers = 0;
669     int numReleasedContainers = 0;
670     int nextRequestPriority = 0;
671     ResourceRequestHandler resourceRequestor = new ResourceRequestHandler();
672 
673     YarnClient clientRMService = YarnClient.createYarnClient();
674 
675     try {
676       // YARN-435
677       // we need getClusterNodes to populate the initial node list,
678       // subsequent updates come through the heartbeat response
679       clientRMService.init(conf);
680       clientRMService.start();
681 
682       ApplicationReport ar = StramClientUtils.getStartedAppInstanceByName(clientRMService,
683         dag.getAttributes().get(DAG.APPLICATION_NAME),
684         UserGroupInformation.getLoginUser().getUserName(),
685         dag.getAttributes().get(DAG.APPLICATION_ID));
686       if (ar != null) {
687         appDone = true;
688         dnmgr.shutdownDiagnosticsMessage = String.format("Application master failed due to application %s with duplicate application name \"%s\" by the same user \"%s\" is already started.",
689           ar.getApplicationId().toString(), ar.getName(), ar.getUser());
690         LOG.info("Forced shutdown due to {}", dnmgr.shutdownDiagnosticsMessage);
691         finishApplication(FinalApplicationStatus.FAILED, numTotalContainers);
692         return;
693       }
694       resourceRequestor.updateNodeReports(clientRMService.getNodeReports());
695     } catch (Exception e) {
696       throw new RuntimeException("Failed to retrieve cluster nodes report.", e);
697     } finally {
698       clientRMService.stop();
699     }
700 
701     // check for previously allocated containers
702     // as of 2.2, containers won't survive AM restart, but this will change in the future - YARN-1490
703     checkContainerStatus();
704     FinalApplicationStatus finalStatus = FinalApplicationStatus.SUCCEEDED;
705     final InetSocketAddress rmAddress = conf.getSocketAddr(YarnConfiguration.RM_ADDRESS,
706       YarnConfiguration.DEFAULT_RM_ADDRESS,
707       YarnConfiguration.DEFAULT_RM_PORT);
708 
709     while (!appDone) {
710       loopCounter++;
711 
712       if (UserGroupInformation.isSecurityEnabled() && System.currentTimeMillis() >= expiryTime && hdfsKeyTabFile != null) {
713         String applicationId = appAttemptID.getApplicationId().toString();
714         expiryTime = StramUserLogin.refreshTokens(tokenLifeTime, FileUtils.getTempDirectoryPath(), applicationId, conf, hdfsKeyTabFile, credentials, rmAddress, true);
715       }
716 
717       Runnable r;
718       while ((r = this.pendingTasks.poll()) != null) {
719         r.run();
720       }
721 
722       // log current state
723       /*
724        * LOG.info("Current application state: loop=" + loopCounter + ", appDone=" + appDone + ", total=" +
725        * numTotalContainers + ", requested=" + numRequestedContainers + ", completed=" + numCompletedContainers +
726        * ", failed=" + numFailedContainers + ", currentAllocated=" + this.allAllocatedContainers.size());
727        */
728       // Sleep before each loop when asking RM for containers
729       // to avoid flooding RM with spurious requests when it
730       // need not have any available containers
731       try {
732         sleep(1000);
733       }
734       catch (InterruptedException e) {
735         LOG.info("Sleep interrupted " + e.getMessage());
736       }
737 
738       // Setup request to be sent to RM to allocate containers
739       List<ContainerRequest> containerRequests = new ArrayList<ContainerRequest>();
740       List<ContainerRequest> removedContainerRequests = new ArrayList<ContainerRequest>();
741 
742       // request containers for pending deploy requests
743       if (!dnmgr.containerStartRequests.isEmpty()) {
744         StreamingContainerAgent.ContainerStartRequest csr;
745         while ((csr = dnmgr.containerStartRequests.poll()) != null) {
746           if (csr.container.getRequiredMemoryMB() > maxMem) {
747             LOG.warn("Container memory {}m above max threshold of cluster. Using max value {}m.", csr.container.getRequiredMemoryMB(), maxMem);
748             csr.container.setRequiredMemoryMB(maxMem);
749           }
750           if(csr.container.getRequiredMemoryMB() < minMem){
751             csr.container.setRequiredMemoryMB(minMem);
752           }
753           if (csr.container.getRequiredVCores() > maxVcores) {
754             LOG.warn("Container vcores {} above max threshold of cluster. Using max value {}.", csr.container.getRequiredVCores(), maxVcores);
755             csr.container.setRequiredVCores(maxVcores);
756           }
757           if(csr.container.getRequiredVCores() < minVcores){
758             csr.container.setRequiredVCores(minVcores);
759           }
760           csr.container.setResourceRequestPriority(nextRequestPriority++);
761           ContainerRequest cr = resourceRequestor.createContainerRequest(csr, true);
762           MutablePair<Integer, ContainerRequest> pair = new MutablePair<Integer, ContainerRequest>(loopCounter, cr);
763           requestedResources.put(csr, pair);
764           containerRequests.add(cr);
765         }
766       }
767 
768       if (!requestedResources.isEmpty()) {
769         //resourceRequestor.clearNodeMapping();
770         for (Map.Entry<StreamingContainerAgent.ContainerStartRequest, MutablePair<Integer, ContainerRequest>> entry : requestedResources.entrySet()) {
771           if ((loopCounter - entry.getValue().getKey()) > NUMBER_MISSED_HEARTBEATS) {
772             StreamingContainerAgent.ContainerStartRequest csr = entry.getKey();
773             removedContainerRequests.add(entry.getValue().getRight());
774             ContainerRequest cr = resourceRequestor.createContainerRequest(csr, false);
775             entry.getValue().setLeft(loopCounter);
776             entry.getValue().setRight(cr);
777             containerRequests.add(cr);
778           }
779         }
780       }
781 
782      /* Remove nodes from blacklist after timeout */
783       long currentTime = System.currentTimeMillis();
784       List<String> blacklistRemovals = new ArrayList<String>();
785       for (Iterator<Pair<Long, List<String>>> it = blacklistedNodesQueueWithTimeStamp.iterator(); it.hasNext();) {
786         Pair<Long, List<String>> entry = it.next();
787         Long timeDiff = currentTime - entry.getFirst();
788         if (timeDiff > blacklistRemovalTime) {
789           blacklistRemovals.addAll(entry.getSecond());
790           it.remove();
791         } else {
792           break;
793         }
794       }
795       if (!blacklistRemovals.isEmpty()) {
796         amRmClient.updateBlacklist(null, blacklistRemovals);
797       }
798 
799       numTotalContainers += containerRequests.size();
800       numRequestedContainers += containerRequests.size();
801       AllocateResponse amResp = sendContainerAskToRM(containerRequests, removedContainerRequests, releasedContainers);
802       if (amResp.getAMCommand() != null) {
803         LOG.info(" statement executed:{}", amResp.getAMCommand());
804         switch (amResp.getAMCommand()) {
805           case AM_RESYNC:
806           case AM_SHUTDOWN:
807             throw new YarnRuntimeException("Received the " + amResp.getAMCommand() + " command from RM");
808           default:
809             throw new YarnRuntimeException("Received the " + amResp.getAMCommand() + " command from RM");
810 
811         }
812       }
813       releasedContainers.clear();
814 
815       // Retrieve list of allocated containers from the response
816       List<Container> newAllocatedContainers = amResp.getAllocatedContainers();
817       // LOG.info("Got response from RM for container ask, allocatedCnt=" + newAllocatedContainers.size());
818       numRequestedContainers -= newAllocatedContainers.size();
819       long timestamp = System.currentTimeMillis();
820       for (Container allocatedContainer : newAllocatedContainers) {
821 
822         LOG.info("Got new container." + ", containerId=" + allocatedContainer.getId() + ", containerNode=" + allocatedContainer.getNodeId() + ", containerNodeURI=" + allocatedContainer.getNodeHttpAddress() + ", containerResourceMemory" + allocatedContainer.getResource().getMemory() + ", priority" + allocatedContainer.getPriority());
823         // + ", containerToken" + allocatedContainer.getContainerToken().getIdentifier().toString());
824 
825         boolean alreadyAllocated = true;
826         StreamingContainerAgent.ContainerStartRequest csr = null;
827         for (Map.Entry<StreamingContainerAgent.ContainerStartRequest, MutablePair<Integer, ContainerRequest>> entry : requestedResources.entrySet()) {
828           if (entry.getKey().container.getResourceRequestPriority() == allocatedContainer.getPriority().getPriority()) {
829             alreadyAllocated = false;
830             csr = entry.getKey();
831             break;
832           }
833         }
834 
835         if (alreadyAllocated) {
836           LOG.info("Releasing {} as resource with priority {} was already assigned", allocatedContainer.getId(), allocatedContainer.getPriority());
837           releasedContainers.add(allocatedContainer.getId());
838           numReleasedContainers++;
839           numRequestedContainers++;
840           continue;
841         }
842         if (csr != null) {
843           requestedResources.remove(csr);
844         }
845 
846         // allocate resource to container
847         ContainerResource resource = new ContainerResource(allocatedContainer.getPriority().getPriority(), allocatedContainer.getId().toString(), allocatedContainer.getNodeId().toString(), allocatedContainer.getResource().getMemory(), allocatedContainer.getResource().getVirtualCores(), allocatedContainer.getNodeHttpAddress());
848         StreamingContainerAgent sca = dnmgr.assignContainer(resource, null);
849 
850         if (sca == null) {
851           // allocated container no longer needed, add release request
852           LOG.warn("Container {} allocated but nothing to deploy, going to release this container.", allocatedContainer.getId());
853           releasedContainers.add(allocatedContainer.getId());
854         }
855         else {
856           AllocatedContainer allocatedContainerHolder = new AllocatedContainer(allocatedContainer);
857           this.allocatedContainers.put(allocatedContainer.getId().toString(), allocatedContainerHolder);
858           ByteBuffer tokens = null;
859           if (UserGroupInformation.isSecurityEnabled()) {
860             UserGroupInformation ugi = UserGroupInformation.getLoginUser();
861             Token<StramDelegationTokenIdentifier> delegationToken = allocateDelegationToken(ugi.getUserName(), heartbeatListener.getAddress());
862             allocatedContainerHolder.delegationToken = delegationToken;
863             //ByteBuffer tokens = LaunchContainerRunnable.getTokens(delegationTokenManager, heartbeatListener.getAddress());
864             tokens = LaunchContainerRunnable.getTokens(ugi, delegationToken);
865           }
866           LaunchContainerRunnable launchContainer = new LaunchContainerRunnable(allocatedContainer, nmClient, sca, tokens);
867           // Thread launchThread = new Thread(runnableLaunchContainer);
868           // launchThreads.add(launchThread);
869           // launchThread.start();
870           launchContainer.run(); // communication with NMs is now async
871 
872           // record container start event
873           StramEvent ev = new StramEvent.StartContainerEvent(allocatedContainer.getId().toString(), allocatedContainer.getNodeId().toString());
874           ev.setTimestamp(timestamp);
875           dnmgr.recordEventAsync(ev);
876         }
877       }
878 
879       // track node updates for future locality constraint allocations
880       // TODO: it seems 2.0.4-alpha doesn't give us any updates
881       resourceRequestor.updateNodeReports(amResp.getUpdatedNodes());
882 
883       // Check the completed containers
884       List<ContainerStatus> completedContainers = amResp.getCompletedContainersStatuses();
885       // LOG.debug("Got response from RM for container ask, completedCnt=" + completedContainers.size());
886       List<String> blacklistAdditions = new ArrayList<String>();
887       for (ContainerStatus containerStatus : completedContainers) {
888         LOG.info("Completed containerId=" + containerStatus.getContainerId() + ", state=" + containerStatus.getState() + ", exitStatus=" + containerStatus.getExitStatus() + ", diagnostics=" + containerStatus.getDiagnostics());
889 
890         // non complete containers should not be here
891         assert (containerStatus.getState() == ContainerState.COMPLETE);
892 
893         AllocatedContainer allocatedContainer = allocatedContainers.remove(containerStatus.getContainerId().toString());
894         if (allocatedContainer != null && allocatedContainer.delegationToken != null) {
895           UserGroupInformation ugi = UserGroupInformation.getLoginUser();
896           delegationTokenManager.cancelToken(allocatedContainer.delegationToken, ugi.getUserName());
897         }
898         int exitStatus = containerStatus.getExitStatus();
899         if (0 != exitStatus) {
900           if (allocatedContainer != null) {
901             numFailedContainers.incrementAndGet();
902             if (exitStatus != 1 && enableFailedNodeBlacklisting) {
903               // If container failure due to framework
904               String hostname = allocatedContainer.container.getNodeId().getHost();
905               int failedTimes = 1;
906               AtomicInteger failed = failedContainersMap.putIfAbsent(hostname, new AtomicInteger(1));
907               if (failed != null) {
908                 failedTimes = failed.incrementAndGet();
909               }
910               if (failedTimes >= maxConsecutiveContainerFailures) {
911                 // Blacklist the node
912                 LOG.info("Node {} failed {} times consecutively, marking the node blacklisted", hostname, failedTimes);
913                 blacklistAdditions.add(hostname);
914               }
915             }
916           }
917 //          if (exitStatus == 1) {
918 //            // non-recoverable StreamingContainer failure
919 //            appDone = true;
920 //            finalStatus = FinalApplicationStatus.FAILED;
921 //            dnmgr.shutdownDiagnosticsMessage = "Unrecoverable failure " + containerStatus.getContainerId();
922 //            LOG.info("Exiting due to: {}", dnmgr.shutdownDiagnosticsMessage);
923 //          }
924 //          else {
925           // Recoverable failure or process killed (externally or via stop request by AM)
926           // also occurs when a container was released by the application but never assigned/launched
927           LOG.debug("Container {} failed or killed.", containerStatus.getContainerId());
928           dnmgr.scheduleContainerRestart(containerStatus.getContainerId().toString());
929 //          }
930         }
931         else {
932           // container completed successfully
933           numCompletedContainers.incrementAndGet();
934           LOG.info("Container completed successfully." + ", containerId=" + containerStatus.getContainerId());
935           // Reset counter for node failure, if exists
936           String hostname = allocatedContainer.container.getNodeId().getHost();
937           AtomicInteger failedTimes = failedContainersMap.get(hostname);
938           if(failedTimes != null) {
939             failedTimes.set(0);
940           }
941         }
942 
943         String containerIdStr = containerStatus.getContainerId().toString();
944         dnmgr.removeContainerAgent(containerIdStr);
945 
946         // record container stop event
947         StramEvent ev = new StramEvent.StopContainerEvent(containerIdStr, containerStatus.getExitStatus());
948         ev.setReason(containerStatus.getDiagnostics());
949         dnmgr.recordEventAsync(ev);
950       }
951 
952       if (!blacklistAdditions.isEmpty()) {
953         amRmClient.updateBlacklist(blacklistAdditions, null);
954         blacklistedNodesQueueWithTimeStamp.add(new Pair<Long, List<String>>(System.currentTimeMillis(), blacklistAdditions));
955       }
956       if (dnmgr.forcedShutdown) {
957         LOG.info("Forced shutdown due to {}", dnmgr.shutdownDiagnosticsMessage);
958         finalStatus = FinalApplicationStatus.FAILED;
959         appDone = true;
960       }
961       else if (allocatedContainers.isEmpty() && numRequestedContainers == 0 && dnmgr.containerStartRequests.isEmpty()) {
962         LOG.debug("Exiting as no more containers are allocated or requested");
963         finalStatus = FinalApplicationStatus.SUCCEEDED;
964         appDone = true;
965       }
966 
967       LOG.debug("Current application state: loop=" + loopCounter + ", appDone=" + appDone + ", total=" + numTotalContainers + ", requested=" + numRequestedContainers + ", released=" + numReleasedContainers + ", completed=" + numCompletedContainers + ", failed=" + numFailedContainers + ", currentAllocated=" + allocatedContainers.size());
968 
969       // monitor child containers
970       dnmgr.monitorHeartbeat();
971     }
972 
973     finishApplication(finalStatus, numTotalContainers);
974   }

976   private void finishApplication(FinalApplicationStatus finalStatus, int numTotalContainers) throws YarnException, IOException
977   {
978     LOG.info("Application completed. Signalling finish to RM");
979     FinishApplicationMasterRequest finishReq = Records.newRecord(FinishApplicationMasterRequest.class);
980     finishReq.setFinalApplicationStatus(finalStatus);
981 
982     if (finalStatus != FinalApplicationStatus.SUCCEEDED) {
983       String diagnostics = "Diagnostics." + ", total=" + numTotalContainers + ", completed=" + numCompletedContainers.get() + ", allocated=" + allocatedContainers.size() + ", failed=" + numFailedContainers.get();
984       if (!StringUtils.isEmpty(dnmgr.shutdownDiagnosticsMessage)) {
985         diagnostics += "\n";
986         diagnostics += dnmgr.shutdownDiagnosticsMessage;
987       }
988       // YARN-208 - as of 2.0.1-alpha dropped by the RM
989       finishReq.setDiagnostics(diagnostics);
990       // expected termination of the master process
991       // application status and diagnostics message are set above
992     }
993     LOG.info("diagnostics: " + finishReq.getDiagnostics());
994     amRmClient.unregisterApplicationMaster(finishReq.getFinalApplicationStatus(), finishReq.getDiagnostics(), null);
995   }

1007    * Check for containers that were allocated in a previous attempt.
1008    * If the containers are still alive, wait for them to check in via heartbeat.
1009    */
1010   private void checkContainerStatus()
1011   {
1012     Collection<StreamingContainerAgent> containers = this.dnmgr.getContainerAgents();
1013     for (StreamingContainerAgent ca : containers) {
1014       ContainerId containerId = ConverterUtils.toContainerId(ca.container.getExternalId());
1015       NodeId nodeId = ConverterUtils.toNodeId(ca.container.host);
1016 
1017       // put container back into the allocated list
1018       org.apache.hadoop.yarn.api.records.Token containerToken = null;
1019       Resource resource = Resource.newInstance(ca.container.getAllocatedMemoryMB(), ca.container.getAllocatedVCores());
1020       Priority priority = Priority.newInstance(ca.container.getResourceRequestPriority());
1021       Container yarnContainer = Container.newInstance(containerId, nodeId, ca.container.nodeHttpAddress, resource, priority, containerToken);
1022       this.allocatedContainers.put(containerId.toString(), new AllocatedContainer(yarnContainer));
1023 
1024       // check the status
1025       nmClient.getContainerStatusAsync(containerId, nodeId);
1026     }
1027   }

1029   /**
1030    * Ask RM to allocate given no. of containers to this Application Master
1031    *
1032    * @param containerRequests        Containers to ask for from RM
1033    * @param removedContainerRequests Container requests to be removed
1034    * @param releasedContainers
1035    * @return Response from RM to AM with allocated containers
1036    * @throws YarnException
1037    */
1038   private AllocateResponse sendContainerAskToRM(List<ContainerRequest> containerRequests, List<ContainerRequest> removedContainerRequests, List<ContainerId> releasedContainers) throws YarnException, IOException
1039   {
1040     if (removedContainerRequests.size() > 0) {
1041       LOG.info(" Removing container request: " + removedContainerRequests);
1042       for (ContainerRequest cr : removedContainerRequests) {
1043         LOG.info("Removed container: {}", cr.toString());
1044         amRmClient.removeContainerRequest(cr);
1045       }
1046     }
1047     if (containerRequests.size() > 0) {
1048       LOG.info("Asking RM for containers: " + containerRequests);
1049       for (ContainerRequest cr : containerRequests) {
1050         LOG.info("Requested container: {}", cr.toString());
1051         amRmClient.addContainerRequest(cr);
1052       }
1053     }
1054 
1055     for (ContainerId containerId : releasedContainers) {
1056       LOG.info("Released container, id={}", containerId.getId());
1057       amRmClient.releaseAssignedContainer(containerId);
1058     }
1059 
1060     for (String containerIdStr : dnmgr.containerStopRequests.values()) {
1061       AllocatedContainer allocatedContainer = this.allocatedContainers.get(containerIdStr);
1062       if (allocatedContainer != null && !allocatedContainer.stopRequested) {
1063         nmClient.stopContainerAsync(allocatedContainer.container.getId(), allocatedContainer.container.getNodeId());
1064         LOG.info("Requested stop container {}", containerIdStr);
1065         allocatedContainer.stopRequested = true;
1066       }
1067       dnmgr.containerStopRequests.remove(containerIdStr);
1068     }
1069 
1070     return amRmClient.allocate(0);
1071   }
