44  /**
45   * The base class for node implementation<p>
46   * <br>
47   * Implements the base interface {@link com.datatorrent.stram.engine.Node}<br>
48   * <br>
49   * This is the basic functional block of the DAG. It is responsible for the following<br>
50   * It emits and consumes tuples<br>
51   * Upon window boundary it does house cleaning, state sync up etc<br>
52   * Interacts with Stram with a heartbeat protocol<br>
53   * <br>
54   *
55   * @since 0.3.2
56   */
57  public class GenericNode extends Node<Operator>
58  {
59    protected final HashMap<String, SweepableReservoir> inputs = new HashMap<String, SweepableReservoir>();
60    protected ArrayList<DeferredInputConnection> deferredInputConnections = new ArrayList<DeferredInputConnection>();
61  
62    @Override
63    @SuppressWarnings("unchecked")
64    public void addSinks(Map<String, Sink<Object>> sinks)
65    {
66      for (Entry<String, Sink<Object>> e : sinks.entrySet()) {
67        SweepableReservoir original = inputs.get(e.getKey());
68        if (original instanceof TappedReservoir) {
69          TappedReservoir tr = (TappedReservoir) original;
70          tr.add(e.getValue());
71        }
72        else if (original != null) {
73          TappedReservoir tr = new TappedReservoir(original, e.getValue());
74          inputs.put(e.getKey(), tr);
75        }
76      }
77  
78      super.addSinks(sinks);
79    }
80  
81    @Override
82    public void removeSinks(Map<String, Sink<Object>> sinks)
83    {
84      for (Entry<String, Sink<Object>> e : sinks.entrySet()) {
85        SweepableReservoir reservoir = inputs.get(e.getKey());
86        if (reservoir instanceof TappedReservoir) {
87          TappedReservoir tr = (TappedReservoir) reservoir;
88          tr.remove(e.getValue());
89          if (tr.getSinks().length == 0) {
90            tr.reservoir.setSink(tr.setSink(null));
91            inputs.put(e.getKey(), tr.reservoir);
92          }
93        }
94      }
95  
96      super.removeSinks(sinks);
97    }

104   @SuppressWarnings("unchecked")
105   public InputPort<Object> getInputPort(String port)
106   {
107     return (InputPort<Object>) descriptor.inputPorts.get(port).component;
108   }

110   @Override
111   public void connectInputPort(String port, final SweepableReservoir reservoir)
112   {
113     if (reservoir == null) {
114       throw new IllegalArgumentException("Reservoir cannot be null for port '" + port + "' on operator '" + operator + "'");
115     }
116 
117     InputPort<Object> inputPort = getInputPort(port);
118     if (inputPort == null) {
119       throw new IllegalArgumentException("Port '" + port + "' does not exist on operator '" + operator + "'");
120     }
121 
122     if (inputs.containsKey(port)) {
123       deferredInputConnections.add(new DeferredInputConnection(port, reservoir));
124     }
125     else {
126       inputPort.setConnected(true);
127       inputs.put(port, reservoir);
128       reservoir.setSink(inputPort.getSink());
129     }
130   }

132   /**
133    * @param endWindowTuple the value of endWindowTuple
134    */
135   protected void processEndWindow(Tuple endWindowTuple)
136   {
137     endWindowEmitTime = System.currentTimeMillis();
138 
139     if (++applicationWindowCount == APPLICATION_WINDOW_COUNT) {
140       insideWindow = false;
141       operator.endWindow();
142       applicationWindowCount = 0;
143     }
144 
145     if (endWindowTuple == null) {
146       emitEndWindow();
147     }
148     else {
149       for (int s = sinks.length; s-- > 0; ) {
150         sinks[s].put(endWindowTuple);
151       }
152       controlTupleCount++;
153     }
154 
155     if (++checkpointWindowCount == CHECKPOINT_WINDOW_COUNT) {
156       checkpointWindowCount = 0;
157       if (doCheckpoint) {
158         checkpoint(currentWindowId);
159         doCheckpoint = false;
160       }
161       else if (PROCESSING_MODE == ProcessingMode.EXACTLY_ONCE) {
162         checkpoint(currentWindowId);
163       }
164     }
165 
166     ContainerStats.OperatorStats stats = new ContainerStats.OperatorStats();
167     reportStats(stats, currentWindowId);
168     if (!insideWindow) {
169       stats.metrics = collectMetrics();
170     }
171     handleRequests(currentWindowId);
172   }

175   class TupleTracker
176   {
177     final Tuple tuple;
178     SweepableReservoir[] ports;
179 
180     TupleTracker(Tuple base, int count)
181     {
182       tuple = base;
183       ports = new SweepableReservoir[count];
184     }
185 
186   }

199   /**
200    * Originally this method was defined in an attempt to implement the interface Runnable.
201    *
202    * Note that activate does not return as long as there is useful workload for the node.
203    */
204   @Override
205   @SuppressWarnings({"SleepWhileInLoop", "UseSpecificCatch", "BroadCatchBlock", "TooBroadCatch"})
206   public final void run()
207   {
208     doCheckpoint = false;
209 
210     long spinMillis = context.getValue(OperatorContext.SPIN_MILLIS);
211     final boolean handleIdleTime = operator instanceof IdleTimeHandler;
212     int totalQueues = inputs.size();
213 
214     ArrayList<SweepableReservoir> activeQueues = new ArrayList<SweepableReservoir>();
215     activeQueues.addAll(inputs.values());
216 
217     int expectingBeginWindow = activeQueues.size();
218     int receivedEndWindow = 0;
219 
220     TupleTracker tracker;
221     LinkedList<TupleTracker> resetTupleTracker = new LinkedList<TupleTracker>();
222 
223     try {
224       do {
225         Iterator<SweepableReservoir> buffers = activeQueues.iterator();
226   activequeue:
227         while (buffers.hasNext()) {
228           SweepableReservoir activePort = buffers.next();
229           Tuple t = activePort.sweep();
230           if (t != null) {
231             switch (t.getType()) {
232               case BEGIN_WINDOW:
233                 if (expectingBeginWindow == totalQueues) {
234                   activePort.remove();
235                   expectingBeginWindow--;
236                   currentWindowId = t.getWindowId();
237                   for (int s = sinks.length; s-- > 0; ) {
238                     sinks[s].put(t);
239                   }
240                   controlTupleCount++;
241 
242                   if (applicationWindowCount == 0) {
243                     insideWindow = true;
244                     operator.beginWindow(currentWindowId);
245                   }
246                   receivedEndWindow = 0;
247                 }
248                 else if (t.getWindowId() == currentWindowId) {
249                   activePort.remove();
250                   expectingBeginWindow--;
251                 }
252                 else {
253                   buffers.remove();
254 
255                   /* find the name of the port which got out of sequence tuple */
256                   String port = null;
257                   for (Entry<String, SweepableReservoir> e : inputs.entrySet()) {
258                     if (e.getValue() == activePort) {
259                       port = e.getKey();
260                     }
261                   }
262 
263                   assert (port != null); /* we should always find the port */
264 
265                   if (PROCESSING_MODE == ProcessingMode.AT_MOST_ONCE) {
266                     if (t.getWindowId() < currentWindowId) {
267                       /*
268                        * we need to fast forward this stream till we find the current
269                        * window or the window which is bigger than the current window.
270                        */
271 
272                       /* lets move the current reservoir in the background */
273                       Sink<Object> sink = activePort.setSink(Sink.BLACKHOLE);
274                       deferredInputConnections.add(0, new DeferredInputConnection(port, activePort));
275 
276                       /* replace it with the reservoir which blocks the tuples in the past */
277                       WindowIdActivatedReservoir wiar = new WindowIdActivatedReservoir(port, activePort, currentWindowId);
278                       wiar.setSink(sink);
279                       inputs.put(port, wiar);
280                       activeQueues.add(wiar);
281                       break activequeue;
282                     }
283                     else {
284                       expectingBeginWindow--;
285                       if (++receivedEndWindow == totalQueues) {
286                         processEndWindow(null);
287                         activeQueues.addAll(inputs.values());
288                         expectingBeginWindow = activeQueues.size();
289                         break activequeue;
290                       }
291                     }
292                   }
293                   else {
294                     logger.error("Catastrophic Error: Out of sequence tuple {} on port {} while expecting {}", Codec.getStringWindowId(t.getWindowId()), port, Codec.getStringWindowId(currentWindowId));
295                     System.exit(2);
296                   }
297                 }
298                 break;
299 
300               case END_WINDOW:
301                 buffers.remove();
302                 if (t.getWindowId() == currentWindowId) {
303                   activePort.remove();
304                   endWindowDequeueTimes.put(activePort, System.currentTimeMillis());
305                   if (++receivedEndWindow == totalQueues) {
306                     assert (activeQueues.isEmpty());
307                     processEndWindow(t);
308                     activeQueues.addAll(inputs.values());
309                     expectingBeginWindow = activeQueues.size();
310                     break activequeue;
311                   }
312                 }
313                 break;
314 
315               case CHECKPOINT:
316                 activePort.remove();
317                 long checkpointWindow = t.getWindowId();
318                 if (lastCheckpointWindowId < checkpointWindow) {
319                   if (PROCESSING_MODE == ProcessingMode.EXACTLY_ONCE) {
320                     lastCheckpointWindowId = checkpointWindow;
321                   }
322                   else if (!doCheckpoint) {
323                     if (checkpointWindowCount == 0) {
324                       checkpoint(checkpointWindow);
325                       lastCheckpointWindowId = checkpointWindow;
326                     }
327                     else {
328                       doCheckpoint = true;
329                     }
330                   }
331 
332                   for (int s = sinks.length; s-- > 0; ) {
333                     sinks[s].put(t);
334                   }
335                   controlTupleCount++;
336                 }
337                 break;
338 
339               case RESET_WINDOW:
340                 /**
341                  * we will receive tuples which are equal to the number of input streams.
342                  */
343                 activePort.remove();
344                 buffers.remove();
345 
346                 int baseSeconds = t.getBaseSeconds();
347                 tracker = null;
348                 Iterator<TupleTracker> trackerIterator = resetTupleTracker.iterator();
349                 while (trackerIterator.hasNext()) {
350                   tracker = trackerIterator.next();
351                   if (tracker.tuple.getBaseSeconds() == baseSeconds) {
352                     break;
353                   }
354                 }
355 
356                 if (tracker == null) {
357                   tracker = new TupleTracker(t, totalQueues);
358                   resetTupleTracker.add(tracker);
359                 }
360                 int trackerIndex = 0;
361                 while (trackerIndex < tracker.ports.length) {
362                   if (tracker.ports[trackerIndex] == null) {
363                     tracker.ports[trackerIndex++] = activePort;
364                     break;
365                   }
366                   else if (tracker.ports[trackerIndex] == activePort) {
367                     break;
368                   }
369 
370                   trackerIndex++;
371                 }
372 
373                 if (trackerIndex == totalQueues) {
374                   trackerIterator = resetTupleTracker.iterator();
375                   while (trackerIterator.hasNext()) {
376                     if (trackerIterator.next().tuple.getBaseSeconds() <= baseSeconds) {
377                       trackerIterator.remove();
378                     }
379                   }
380                   for (int s = sinks.length; s-- > 0; ) {
381                     sinks[s].put(t);
382                   }
383                   controlTupleCount++;
384 
385                   assert (activeQueues.isEmpty());
386                   activeQueues.addAll(inputs.values());
387                   expectingBeginWindow = activeQueues.size();
388                   break activequeue;
389                 }
390                 break;
391 
392               case END_STREAM:
393                 activePort.remove();
394                 buffers.remove();
395                 for (Iterator<Entry<String, SweepableReservoir>> it = inputs.entrySet().iterator(); it.hasNext(); ) {
396                   Entry<String, SweepableReservoir> e = it.next();
397                   if (e.getValue() == activePort) {
398                     if (!descriptor.inputPorts.isEmpty()) {
399                       descriptor.inputPorts.get(e.getKey()).component.setConnected(false);
400                     }
401                     it.remove();
402 
403                     /* check the deferred connection list for any new port that should be connected here */
404                     Iterator<DeferredInputConnection> dici = deferredInputConnections.iterator();
405                     while (dici.hasNext()) {
406                       DeferredInputConnection dic = dici.next();
407                       if (e.getKey().equals(dic.portname)) {
408                         connectInputPort(dic.portname, dic.reservoir);
409                         dici.remove();
410                         activeQueues.add(dic.reservoir);
411                         break activequeue;
412                       }
413                     }
414 
415                     break;
416                   }
417                 }
418 
419                 /**
420                  * We are not going to receive begin window on this ever!
421                  */
422                 expectingBeginWindow--;
423 
424                 /**
425                  * Since one of the operators we care about it gone, we should relook at our ports.
426                  * We need to make sure that the END_STREAM comes outside of the window.
427                  */
428                 totalQueues--;
429 
430                 boolean break_activequeue = false;
431                 if (totalQueues == 0) {
432                   alive = false;
433                   break_activequeue = true;
434                 }
435                 else if (activeQueues.isEmpty()) {
436                   assert (!inputs.isEmpty());
437                   processEndWindow(null);
438                   activeQueues.addAll(inputs.values());
439                   expectingBeginWindow = activeQueues.size();
440                   break_activequeue = true;
441                 }
442 
443                 /**
444                  * also make sure that we update the reset tuple tracker if this stream had delivered any reset tuples.
445                  * Check all the reset buffers to see if current input port has already delivered reset tuple. If it has
446                  * then we are waiting for something else to deliver the reset tuple, so just clear current reservoir
447                  * from the list of tracked reservoirs. If the current input port has not delivered the reset tuple, and
448                  * it's the only one which has not, then we consider it delivered and release the reset tuple downstream.
449                  */
450                 Tuple tuple = null;
451                 for (trackerIterator = resetTupleTracker.iterator(); trackerIterator.hasNext(); ) {
452                   tracker = trackerIterator.next();
453 
454                   trackerIndex = 0;
455                   while (trackerIndex < tracker.ports.length) {
456                     if (tracker.ports[trackerIndex] == activePort) {
457                       SweepableReservoir[] ports = new SweepableReservoir[totalQueues];
458                       System.arraycopy(tracker.ports, 0, ports, 0, trackerIndex);
459                       if (trackerIndex < totalQueues) {
460                         System.arraycopy(tracker.ports, trackerIndex + 1, ports, trackerIndex, tracker.ports.length - trackerIndex - 1);
461                       }
462                       tracker.ports = ports;
463                       break;
464                     }
465                     else if (tracker.ports[trackerIndex] == null) {
466                       if (trackerIndex == totalQueues) { /* totalQueues is already adjusted above */
467                         if (tuple == null || tuple.getBaseSeconds() < tracker.tuple.getBaseSeconds()) {
468                           tuple = tracker.tuple;
469                         }
470 
471                         trackerIterator.remove();
472                       }
473                       break;
474                     }
475                     else {
476                       tracker.ports = Arrays.copyOf(tracker.ports, totalQueues);
477                     }
478 
479                     trackerIndex++;
480                   }
481                 }
482 
483                 /*
484                  * Since we were waiting for a reset tuple on this stream, we should not any longer.
485                  */
486                 if (tuple != null) {
487                   for (int s = sinks.length; s-- > 0; ) {
488                     sinks[s].put(tuple);
489                   }
490                   controlTupleCount++;
491                 }
492 
493                 if (break_activequeue) {
494                   break activequeue;
495                 }
496                 break;
497 
498               default:
499                 throw new UnhandledException("Unrecognized Control Tuple", new IllegalArgumentException(t.toString()));
500             }
501           }
502         }
503 
504         if (activeQueues.isEmpty() && alive) {
505           logger.error("Catastrophic Error: Invalid State - the operator blocked forever!");
506           System.exit(2);
507         }
508         else {
509           boolean need2sleep = true;
510           for (SweepableReservoir cb : activeQueues) {
511             if (cb.size() > 0) {
512               need2sleep = false;
513               break;
514             }
515           }
516 
517           if (need2sleep) {
518             if (handleIdleTime && insideWindow) {
519               ((IdleTimeHandler) operator).handleIdleTime();
520             }
521             else {
522               Thread.sleep(spinMillis);
523             }
524           }
525         }
526       }
527       while (alive);
528     }
529     catch (ShutdownException se) {
530       logger.debug("Shutdown requested by the operator when alive = {}.", alive);
531       alive = false;
532     }
533     catch (Throwable cause) {
534       synchronized (this) {
535         if (alive) {
536           DTThrowable.rethrow(cause);
537         }
538       }
539 
540       Throwable rootCause = cause;
541       while (rootCause != null) {
542         if (rootCause instanceof InterruptedException) {
543           break;
544         }
545         rootCause = rootCause.getCause();
546       }
547 
548       if (rootCause == null) {
549         DTThrowable.rethrow(cause);
550       }
551       else {
552         logger.debug("Ignoring InterruptedException after shutdown", cause);
553       }
554     }
555 
556     if (insideWindow) {
557       endWindowEmitTime = System.currentTimeMillis();
558       operator.endWindow();
559       if (++applicationWindowCount == APPLICATION_WINDOW_COUNT) {
560         applicationWindowCount = 0;
561       }
562 
563       if (++checkpointWindowCount == CHECKPOINT_WINDOW_COUNT) {
564         checkpointWindowCount = 0;
565         if (doCheckpoint || PROCESSING_MODE == ProcessingMode.EXACTLY_ONCE) {
566           checkpoint(currentWindowId);
567         }
568       }
569 
570       ContainerStats.OperatorStats stats = new ContainerStats.OperatorStats();
571       fixEndWindowDequeueTimesBeforeDeactivate();
572       reportStats(stats, currentWindowId);
573       stats.metrics = collectMetrics();
574       handleRequests(currentWindowId);
575     }
576 
577   }

579   /**
580    * End window dequeue times may not have been saved for all the input ports during deactivate,
581    * so save them for reporting. SPOI-1324.
582    */
583   private void fixEndWindowDequeueTimesBeforeDeactivate()
584   {
585     long endWindowDequeueTime = System.currentTimeMillis();
586     for (SweepableReservoir sr : inputs.values()) {
587       if (endWindowDequeueTimes.get(sr) == null) {
588         endWindowDequeueTimes.put(sr, endWindowDequeueTime);
589       }
590     }
591   }

593   @Override
594   protected void reportStats(ContainerStats.OperatorStats stats, long windowId)
595   {
596     ArrayList<ContainerStats.OperatorStats.PortStats> ipstats = new ArrayList<ContainerStats.OperatorStats.PortStats>();
597     for (Entry<String, SweepableReservoir> e : inputs.entrySet()) {
598       SweepableReservoir ar = e.getValue();
599       ContainerStats.OperatorStats.PortStats portStats = new ContainerStats.OperatorStats.PortStats(e.getKey());
600       portStats.queueSize = ar.size();
601       if(DATA_TUPLE_AWARE) {
602         if (ar instanceof CircularBuffer) {
603           Iterator iterator = ((CircularBuffer) ar).getFrozenIterator();
604           while (iterator.hasNext()) {
605             if (iterator.next() instanceof Tuple) {
606               portStats.queueSize--;
607             }
608           }
609         }
610       }
611       portStats.tupleCount = ar.getCount(true);
612       portStats.endWindowTimestamp = endWindowDequeueTimes.get(e.getValue());
613       ipstats.add(portStats);
614     }
615     stats.inputPorts = ipstats;
616     super.reportStats(stats, windowId);
617   }
