154   public void setup(StreamingContainerContext ctx)
155   {
156     containerContext = ctx;
157 
158     /* add a request factory local to this container */
159     this.requestFactory = new RequestFactory();
160     ctx.attributes.put(ContainerContext.REQUEST_FACTORY, requestFactory);
161 
162     heartbeatIntervalMillis = ctx.getValue(Context.DAGContext.HEARTBEAT_INTERVAL_MILLIS);
163     firstWindowMillis = ctx.startWindowMillis;
164     windowWidthMillis = ctx.getValue(Context.DAGContext.STREAMING_WINDOW_SIZE_MILLIS);
165     checkpointWindowCount = ctx.getValue(Context.DAGContext.CHECKPOINT_WINDOW_COUNT);
166 
167     fastPublisherSubscriber = ctx.getValue(LogicalPlan.FAST_PUBLISHER_SUBSCRIBER);
168 
169     Map<Class<?>, Class<? extends StringCodec<?>>> codecs = ctx.getValue(Context.DAGContext.STRING_CODECS);
170     StringCodecs.loadConverters(codecs);
171 
172     try {
173       if (ctx.deployBufferServer) {
174         eventloop.start();
175 
176         int bufferServerRAM = ctx.getValue(ContainerContext.BUFFER_SERVER_MB);
177         logger.debug("buffer server memory {}", bufferServerRAM);
178         int blockCount;
179         int blocksize;
180         if (bufferServerRAM < ContainerContext.BUFFER_SERVER_MB.defaultValue) {
181           blockCount = 8;
182           blocksize = bufferServerRAM / blockCount;
183           if (blocksize < 1) {
184             blocksize = 1;
185           }
186         }
187         else {
188           blocksize = 64;
189           blockCount = bufferServerRAM / blocksize;
190         }
191         // start buffer server, if it was not set externally
192         bufferServer = new Server(0, blocksize * 1024 * 1024, blockCount);
193         bufferServer.setAuthToken(ctx.getValue(StreamingContainerContext.BUFFER_SERVER_TOKEN));
194         if (ctx.getValue(Context.DAGContext.BUFFER_SPOOLING)) {
195           bufferServer.setSpoolStorage(new DiskStorage());
196         }
197         SocketAddress bindAddr = bufferServer.run(eventloop);
198         logger.debug("Buffer server started: {}", bindAddr);
199         this.bufferServerAddress = NetUtils.getConnectAddress(((InetSocketAddress) bindAddr));
200       }
201     }
202     catch (IOException ex) {
203       logger.warn("deploy request failed due to {}", ex);
204       throw new IllegalStateException("Failed to deploy buffer server", ex);
205     }
206 
207     for (Class<?> clazz : ContainerEvent.CONTAINER_EVENTS_LISTENERS) {
208       try {
209         Object newInstance = clazz.newInstance();
210         singletons.put(clazz.getName(), newInstance);
211 
212         if (newInstance instanceof Component) {
213           components.add((Component<ContainerContext>) newInstance);
214         }
215 
216         eventBus.subscribe(newInstance);
217       }
218       catch (InstantiationException ex) {
219         logger.warn("Container Event Listener Instantiation", ex);
220       }
221       catch (IllegalAccessException ex) {
222         logger.warn("Container Event Listener Instantiation", ex);
223       }
224     }
225 
226     operateListeners(ctx, true);
227   }

245   /**
246    * Initialize container. Establishes heartbeat connection to the master
247    * distribute through the callback address provided on the command line. Deploys
248    * initial modules, then enters the heartbeat loop, which will only terminate
249    * once container receives shutdown request from the master. On shutdown,
250    * after exiting heartbeat loop, shutdown all modules and terminate
251    * processing threads.
252    *
253    * @param args
254    * @throws Throwable
255    */
256   public static void main(String[] args) throws Throwable
257   {
258     StdOutErrLog.tieSystemOutAndErrToLog();
259     logger.debug("PID: " + System.getenv().get("JVM_PID"));
260     logger.info("Child starting with classpath: {}", System.getProperty("java.class.path"));
261 
262     String appPath = System.getProperty(PROP_APP_PATH);
263     if (appPath == null) {
264       logger.error("{} not set in container environment.", PROP_APP_PATH);
265       System.exit(1);
266     }
267 
268     int exitStatus = 1; // interpreted as unrecoverable container failure
269 
270     RecoverableRpcProxy rpcProxy = new RecoverableRpcProxy(appPath, new Configuration());
271     final StreamingContainerUmbilicalProtocol umbilical = rpcProxy.getProxy();
272     final String childId = System.getProperty(StreamingApplication.DT_PREFIX + "cid");
273     try {
274       StreamingContainerContext ctx = umbilical.getInitContext(childId);
275       StreamingContainer stramChild = new StreamingContainer(childId, umbilical);
276       logger.debug("Container Context = {}", ctx);
277       stramChild.setup(ctx);
278       try {
279         /* main thread enters heartbeat loop */
280         stramChild.heartbeatLoop();
281         exitStatus = 0;
282       }
283       finally {
284         stramChild.teardown();
285       }
286     }
287     catch (Error error) {
288       logger.error("Fatal error in container!", error);
289       /* Report back any failures, for diagnostic purposes */
290       String msg = ExceptionUtils.getStackTrace(error);
291       umbilical.reportError(childId, null, "FATAL: " + msg);
292     }
293     catch (Exception exception) {
294       logger.error("Fatal exception in container!", exception);
295       /* Report back any failures, for diagnostic purposes */
296       String msg = ExceptionUtils.getStackTrace(exception);
297       umbilical.reportError(childId, null, msg);
298     }
299     finally {
300       rpcProxy.close();
301       DefaultMetricsSystem.shutdown();
302       logger.info("Exit status for container: {}", exitStatus);
303       LogManager.shutdown();
304     }
305 
306     if (exitStatus != 0) {
307       System.exit(exitStatus);
308     }
309   }

311   public synchronized void deactivate()
312   {
313     ArrayList<Thread> activeThreads = new ArrayList<Thread>();
314     ArrayList<Integer> activeOperators = new ArrayList<Integer>();
315 
316     for (Map.Entry<Integer, Node<?>> e : nodes.entrySet()) {
317       Thread t = e.getValue().context.getThread();
318       if (t == null || !t.isAlive()) {
319         disconnectNode(e.getKey());
320       }
321       else {
322         activeThreads.add(t);
323         activeOperators.add(e.getKey());
324         e.getValue().shutdown();
325       }
326     }
327 
328     try {
329       Iterator<Integer> iterator = activeOperators.iterator();
330       for (Thread t : activeThreads) {
331         t.join(1000);
332         if (!t.getState().equals(State.TERMINATED)) {
333           t.interrupt();
334         }
335         disconnectNode(iterator.next());
336       }
337     }
338     catch (InterruptedException ex) {
339       logger.warn("Aborting wait for operators to get deactivated!", ex);
340     }
341 
342     for (WindowGenerator wg : activeGenerators.keySet()) {
343       wg.deactivate();
344     }
345     activeGenerators.clear();
346 
347     for (Stream stream : activeStreams.keySet()) {
348       stream.deactivate();
349     }
350     activeStreams.clear();
351   }

353   private void disconnectNode(int nodeid)
354   {
355     Node<?> node = nodes.get(nodeid);
356     disconnectWindowGenerator(nodeid, node);
357 
358     PortMappingDescriptor portMappingDescriptor = node.getPortMappingDescriptor();
359     Iterator<String> outputPorts = portMappingDescriptor.outputPorts.keySet().iterator();
360     while (outputPorts.hasNext()) {
361       String sourceIdentifier = String.valueOf(nodeid).concat(Component.CONCAT_SEPARATOR).concat(outputPorts.next());
362       ComponentContextPair<Stream, StreamContext> pair = streams.remove(sourceIdentifier);
363       if (pair != null) {
364         if (activeStreams.remove(pair.component) != null) {
365           pair.component.deactivate();
366           eventBus.publish(new StreamDeactivationEvent(pair));
367         }
368 
369         if (pair.component instanceof Stream.MultiSinkCapableStream) {
370           String sinks = pair.context.getSinkId();
371           if (sinks == null) {
372             logger.error("mux sinks found connected at {} with sink id null", sourceIdentifier);
373           }
374           else {
375             String[] split = sinks.split(MuxStream.MULTI_SINK_ID_CONCAT_SEPARATOR);
376             for (int i = split.length; i-- > 0; ) {
377               ComponentContextPair<Stream, StreamContext> spair = streams.remove(split[i]);
378               if (spair == null) {
379                 logger.error("mux is missing the stream for sink {}", split[i]);
380               }
381               else {
382                 if (activeStreams.remove(spair.component) != null) {
383                   spair.component.deactivate();
384                   eventBus.publish(new StreamDeactivationEvent(spair));
385                 }
386 
387                 spair.component.teardown();
388               }
389             }
390           }
391         }
392         else {
393           // it's either inline stream or it's bufferserver publisher.
394         }
395 
396         pair.component.teardown();
397       }
398     }
399 
400     Iterator<String> inputPorts = portMappingDescriptor.inputPorts.keySet().iterator();
401     while (inputPorts.hasNext()) {
402       String sinkIdentifier = String.valueOf(nodeid).concat(Component.CONCAT_SEPARATOR).concat(inputPorts.next());
403       ComponentContextPair<Stream, StreamContext> pair = streams.remove(sinkIdentifier);
404       if (pair != null) {
405         if (activeStreams.remove(pair.component) != null) {
406           pair.component.deactivate();
407           eventBus.publish(new StreamDeactivationEvent(pair));
408         }
409 
410         pair.component.teardown();
411         /**
412          * we should also make sure that if this stream is connected to mux stream,
413          * we deregister it from the mux stream to avoid clogged sink problem.
414          */
415         ComponentContextPair<Stream, StreamContext> sourcePair = streams.get(pair.context.getSourceId());
416         if (sourcePair != null) {
417           if (sourcePair == pair) {
418             /* for some reason we had the stream stored against both source and sink identifiers */
419             streams.remove(pair.context.getSourceId());
420           }
421           else {
422             /* the stream was one of the many streams sourced by a muxstream */
423             unregisterSinkFromMux(sourcePair, sinkIdentifier);
424           }
425         }
426       }
427     }
428   }

430   private boolean unregisterSinkFromMux(ComponentContextPair<Stream, StreamContext> muxpair, String sinkIdentifier)
431   {
432     String[] sinks = muxpair.context.getSinkId().split(MuxStream.MULTI_SINK_ID_CONCAT_SEPARATOR);
433     boolean found = false;
434     for (int i = sinks.length; i-- > 0; ) {
435       if (sinks[i].equals(sinkIdentifier)) {
436         sinks[i] = null;
437         found = true;
438         break;
439       }
440     }
441 
442     if (found) {
443       ((Stream.MultiSinkCapableStream) muxpair.component).setSink(sinkIdentifier, null);
444 
445       if (sinks.length == 1) {
446         muxpair.context.setSinkId(null);
447         streams.remove(muxpair.context.getSourceId());
448         if (activeStreams.remove(muxpair.component) != null) {
449           muxpair.component.deactivate();
450           eventBus.publish(new StreamDeactivationEvent(muxpair));
451         }
452         muxpair.component.teardown();
453       }
454       else {
455         StringBuilder builder = new StringBuilder(muxpair.context.getSinkId().length() - MuxStream.MULTI_SINK_ID_CONCAT_SEPARATOR.length() - sinkIdentifier.length());
456 
457         found = false;
458         for (int i = sinks.length; i-- > 0; ) {
459           if (sinks[i] != null) {
460             if (found) {
461               builder.append(MuxStream.MULTI_SINK_ID_CONCAT_SEPARATOR).append(sinks[i]);
462             }
463             else {
464               builder.append(sinks[i]);
465               found = true;
466             }
467           }
468         }
469 
470         muxpair.context.setSinkId(builder.toString());
471       }
472     }
473     else {
474       logger.error("{} was not connected to stream connected to {}", sinkIdentifier, muxpair.context.getSourceId());
475     }
476 
477     return found;
478   }

480   private void disconnectWindowGenerator(int nodeid, Node<?> node)
481   {
482     WindowGenerator chosen1 = generators.remove(nodeid);
483     if (chosen1 != null) {
484       chosen1.releaseReservoir(Integer.toString(nodeid).concat(Component.CONCAT_SEPARATOR).concat(Node.INPUT));
485       // should we send the closure of the port to the node?
486 
487       int count = 0;
488       for (WindowGenerator wg : generators.values()) {
489         if (chosen1 == wg) {
490           count++;
491         }
492       }
493 
494       if (count == 0) {
495         activeGenerators.remove(chosen1);
496         chosen1.deactivate();
497         chosen1.teardown();
498       }
499     }

502   private synchronized void undeploy(List<Integer> nodeList)
503   {
504     /**
505      * make sure that all the operators which we are asked to undeploy are in this container.
506      */
507     HashMap<Integer, Node<?>> toUndeploy = new HashMap<Integer, Node<?>>();
508     for (Integer operatorId : nodeList) {
509       Node<?> node = nodes.get(operatorId);
510       if (node == null) {
511         throw new IllegalArgumentException("Node " + operatorId + " is not hosted in this container!");
512       }
513       else if (toUndeploy.containsKey(operatorId)) {
514         throw new IllegalArgumentException("Node " + operatorId + " is requested to be undeployed more than once");
515       }
516       else {
517         toUndeploy.put(operatorId, node);
518       }
519     }
520 
521     ArrayList<Thread> joinList = new ArrayList<Thread>();
522     ArrayList<Integer> discoList = new ArrayList<Integer>();
523     for (Integer operatorId : nodeList) {
524       Thread t = nodes.get(operatorId).context.getThread();
525       if (t == null || !t.isAlive()) {
526         disconnectNode(operatorId);
527       }
528       else {
529         joinList.add(t);
530         discoList.add(operatorId);
531         nodes.get(operatorId).shutdown();
532       }
533     }
534 
535     try {
536       Iterator<Integer> iterator = discoList.iterator();
537       for (Thread t : joinList) {
538         t.join(1000);
539         if (!t.getState().equals(State.TERMINATED)) {
540           t.interrupt();
541         }
542         disconnectNode(iterator.next());
543       }
544       logger.info("Undeploy complete.");
545     }
546     catch (InterruptedException ex) {
547       logger.warn("Aborting wait for operators to get deactivated!", ex);
548     }
549 
550     for (Integer operatorId : nodeList) {
551       nodes.remove(operatorId);
552     }
553   }

589   public void heartbeatLoop() throws Exception
590   {
591     umbilical.log(containerId, "[" + containerId + "] Entering heartbeat loop..");
592     logger.debug("Entering heartbeat loop (interval is {} ms)", this.heartbeatIntervalMillis);
593     final YarnConfiguration conf = new YarnConfiguration();
594     long tokenLifeTime = (long)(containerContext.getValue(LogicalPlan.TOKEN_REFRESH_ANTICIPATORY_FACTOR) * containerContext.getValue(LogicalPlan.HDFS_TOKEN_LIFE_TIME));
595     long expiryTime = System.currentTimeMillis();
596     final Credentials credentials = UserGroupInformation.getCurrentUser().getCredentials();
597     Iterator<Token<?>> iter = credentials.getAllTokens().iterator();
598     while (iter.hasNext()) {
599       Token<?> token = iter.next();
600       logger.debug("token: {}", token);
601     }
602     String hdfsKeyTabFile = containerContext.getValue(LogicalPlan.KEY_TAB_FILE);
603     while (!exitHeartbeatLoop) {
604 
605       if (UserGroupInformation.isSecurityEnabled() && System.currentTimeMillis() >= expiryTime && hdfsKeyTabFile != null) {
606         expiryTime = StramUserLogin.refreshTokens(tokenLifeTime, FileUtils.getTempDirectoryPath(), containerId, conf, hdfsKeyTabFile, credentials, null, false);
607       }
608       synchronized (this.heartbeatTrigger) {
609         try {
610           this.heartbeatTrigger.wait(heartbeatIntervalMillis);
611         }
612         catch (InterruptedException e1) {
613           logger.warn("Interrupted in heartbeat loop, exiting..");
614           break;
615         }
616       }
617 
618       long currentTime = System.currentTimeMillis();
619       ContainerHeartbeat msg = new ContainerHeartbeat();
620       msg.jvmName = jvmName;
621       if (this.bufferServerAddress != null) {
622         msg.bufferServerHost = this.bufferServerAddress.getHostName();
623         msg.bufferServerPort = this.bufferServerAddress.getPort();
624         if (bufferServer != null && !eventloop.isActive()) {
625           logger.warn("Requesting restart due to terminated event loop");
626           msg.restartRequested = true;
627         }
628       }
629       msg.memoryMBFree = ((int) (Runtime.getRuntime().freeMemory() / (1024 * 1024)));
630       garbageCollectorMXBeans = ManagementFactory.getGarbageCollectorMXBeans();
631       for (GarbageCollectorMXBean bean : garbageCollectorMXBeans) {
632         msg.gcCollectionTime += bean.getCollectionTime();
633         msg.gcCollectionCount += bean.getCollectionCount();
634       }
635 
636       ContainerHeartbeatResponse rsp;
637       do {
638         ContainerStats stats = new ContainerStats(containerId);
639         // gather heartbeat info for all operators
640         for (Map.Entry<Integer, Node<?>> e : nodes.entrySet()) {
641           OperatorHeartbeat hb = new OperatorHeartbeat();
642           hb.setNodeId(e.getKey());
643           hb.setGeneratedTms(currentTime);
644           hb.setIntervalMs(heartbeatIntervalMillis);
645           if (e.getValue().commandResponse.size() > 0) {
646             BlockingQueue<StatsListener.OperatorResponse> commandResponse = e.getValue().commandResponse;
647             ArrayList<StatsListener.OperatorResponse> response = new ArrayList<StatsListener.OperatorResponse>();
648             for (int i = 0; i < commandResponse.size(); i++) {
649               response.add(commandResponse.poll());
650             }
651             hb.requestResponse = response;
652           }
653           OperatorContext context = e.getValue().context;
654           context.drainStats(hb.getOperatorStatsContainer());
655 
656           if (context.getThread() == null || context.getThread().getState() != Thread.State.TERMINATED) {
657             hb.setState(DeployState.ACTIVE);
658           }
659           else if (failedNodes.contains(hb.nodeId)) {
660             hb.setState(DeployState.FAILED);
661           }
662           else {
663             logger.debug("Reporting SHUTDOWN state because thread is {} and failedNodes is {}", context.getThread(), failedNodes);
664             hb.setState(DeployState.SHUTDOWN);
665           }
666 
667           stats.addNodeStats(hb);
668         }
669 
670         /**
671          * Container stats published for whoever is interested in listening.
672          * Currently interested candidates are TupleRecorderCollection and BufferServerStatsSubscriber
673          */
674         eventBus.publish(new ContainerStatsEvent(stats));
675 
676         msg.setContainerStats(stats);
677 
678         // heartbeat call and follow-up processing
679         //logger.debug("Sending heartbeat for {} operators.", msg.getContainerStats().size());
680         msg.sentTms = System.currentTimeMillis();
681         rsp = umbilical.processHeartbeat(msg);
682         processHeartbeatResponse(rsp);
683         if (rsp.hasPendingRequests) {
684           logger.info("Waiting for pending request.");
685           synchronized (this.heartbeatTrigger) {
686             try {
687               this.heartbeatTrigger.wait(500);
688             }
689             catch (InterruptedException ie) {
690               logger.warn("Interrupted in heartbeat loop", ie);
691               break;
692             }
693           }
694         }
695       }
696       while (rsp.hasPendingRequests);
697 
698     }
699     logger.debug("Exiting hearbeat loop");
700     umbilical.log(containerId, "[" + containerId + "] Exiting heartbeat loop..");
701   }

705   private void processNodeRequests(boolean flagInvalid)
706   {
707     for (StramToNodeRequest req : nodeRequests) {
708       if (req.isDeleted()) {
709         continue;
710       }
711       if (req instanceof StramToNodeChangeLoggersRequest) {
712         handleChangeLoggersRequest((StramToNodeChangeLoggersRequest) req);
713         continue;
714       }
715 
716       Node<?> node = nodes.get(req.getOperatorId());
717       if (node == null) {
718         logger.warn("Node for operator {} is not found, probably not deployed yet", req.getOperatorId());
719         continue;
720       }
721 
722       Thread thread = node.context.getThread();
723       if (thread == null || !thread.isAlive()) {
724         if (flagInvalid) {
725           logger.warn("Received request with invalid operator id {} ({})", req.getOperatorId(), req);
726           req.setDeleted(true);
727         }
728       }
729       else {
730         logger.debug("request received: {}", req);
731         OperatorRequest requestExecutor = requestFactory.getRequestExecutor(nodes.get(req.operatorId), req);
732         if (requestExecutor != null) {
733           node.context.request(requestExecutor);
734         }
735         else {
736           logger.warn("No executor identified for the request {}", req);
737         }
738         req.setDeleted(true);
739       }
740     }

743   public void processHeartbeatResponse(ContainerHeartbeatResponse rsp)
744   {
745     if (rsp.nodeRequests != null) {
746       nodeRequests = rsp.nodeRequests;
747     }
748 
749     if (rsp.committedWindowId != lastCommittedWindowId) {
750       lastCommittedWindowId = rsp.committedWindowId;
751       OperatorRequest nr = null;
752       for (Entry<Integer, Node<?>> e : nodes.entrySet()) {
753         final Thread thread = e.getValue().context.getThread();
754         if (thread == null || !thread.isAlive()) {
755           continue;
756         }
757 
758         if (e.getValue().getOperator() instanceof Operator.CheckpointListener) {
759           if (nr == null) {
760             nr = new OperatorRequest()
761             {
762               @Override
763               public StatsListener.OperatorResponse execute(Operator operator, int operatorId, long windowId) throws IOException
764               {
765                 ((Operator.CheckpointListener) operator).committed(lastCommittedWindowId);
766                 return null;
767               }
768 
769             };
770           }
771           e.getValue().context.request(nr);
772         }
773       }
774     }
775 
776     if (rsp.undeployRequest != null) {
777       logger.info("Undeploy request: {}", rsp.undeployRequest);
778       processNodeRequests(false);
779       undeploy(rsp.undeployRequest);
780     }
781 
782     if (rsp.shutdown) {
783       logger.info("Received shutdown request");
784       processNodeRequests(false);
785       this.exitHeartbeatLoop = true;
786       return;
787     }
788 
789     if (rsp.deployRequest != null) {
790       logger.info("Deploy request: {}", rsp.deployRequest);
791       try {
792         deploy(rsp.deployRequest);
793       }
794       catch (Exception e) {
795         logger.error("deploy request failed", e);
796         try {
797           umbilical.log(this.containerId, "deploy request failed: " + rsp.deployRequest + " " + ExceptionUtils.getStackTrace(e));
798         }
799         catch (IOException ioe) {
800           // ignore
801         }
802         this.exitHeartbeatLoop = true;
803         throw new IllegalStateException("Deploy request failed: " + rsp.deployRequest, e);
804       }
805     }
806 
807     processNodeRequests(true);
808   }

825   private synchronized void deploy(List<OperatorDeployInfo> nodeList) throws Exception
826   {
827     /*
828      * A little bit of up front sanity check would reduce the percentage of deploy failures later.
829      */
830     for (OperatorDeployInfo ndi : nodeList) {
831       if (nodes.containsKey(ndi.id)) {
832         throw new IllegalStateException("Node with id: " + ndi.id + " already present in container " + containerId + "!");
833       }
834     }
835 
836     deployNodes(nodeList);
837 
838     HashMap<String, ArrayList<String>> groupedInputStreams = new HashMap<String, ArrayList<String>>();

840       groupInputStreams(groupedInputStreams, ndi);
841     }
842

844     deployInputStreams(nodeList, newStreams);
845     for (ComponentContextPair<Stream, StreamContext> pair : newStreams.values()) {

847     }
848     streams.putAll(newStreams);
849 
850     HashMap<Integer, OperatorDeployInfo> operatorMap = new HashMap<Integer, OperatorDeployInfo>(nodeList.size());
851     for (OperatorDeployInfo o : nodeList) {
852       operatorMap.put(o.id, o);
853     }
854     activate(operatorMap, newStreams);
855   }

922   private HashMap<String, ComponentContextPair<Stream, StreamContext>> deployOutputStreams(
923     List<OperatorDeployInfo> nodeList, HashMap<String, ArrayList<String>> groupedInputStreams)
924     throws Exception
925   {
926     HashMap<String, ComponentContextPair<Stream, StreamContext>> newStreams = new HashMap<String, ComponentContextPair<Stream, StreamContext>>();
927     /*
928      * We proceed to deploy all the output streams. At the end of this block, our streams collection
929      * will contain all the streams which originate at the output port of the operators. The streams
930      * are generally mapped against the "nodename.portname" string. But the BufferServerPublishers which
931      * share the output port with other inline streams are mapped against the Buffer Server port to
932      * avoid collision and at the same time keep track of these buffer streams.
933      */
934     for (OperatorDeployInfo ndi : nodeList) {
935       Node<?> node = nodes.get(ndi.id);
936       long checkpointWindowId = ndi.checkpoint.windowId;
937 
938       for (OperatorDeployInfo.OutputDeployInfo nodi : ndi.outputs) {
939         String sourceIdentifier = Integer.toString(ndi.id).concat(Component.CONCAT_SEPARATOR).concat(nodi.portName);
940         int queueCapacity = getValue(PortContext.QUEUE_CAPACITY, nodi, ndi);
941         logger.debug("for stream {} the queue capacity is {}", sourceIdentifier, queueCapacity);
942 
943         ArrayList<String> collection = groupedInputStreams.get(sourceIdentifier);
944         Map<Integer, StreamCodec<?>> streamCodecs = nodi.streamCodecs;
945         if ((collection == null) && (streamCodecs.size() == 1)) {
946           assert (nodi.bufferServerHost != null) : "resulting stream cannot be inline: " + nodi;
947           /*
948            * Let's create a stream to carry the data to the Buffer Server.
949            * Nobody in this container is interested in the output placed on this stream, but
950            * this stream exists. That means someone outside of this container must be interested.
951            */
952           Map.Entry<Integer, StreamCodec<?>> entry = streamCodecs.entrySet().iterator().next();
953           StreamCodec<?> streamCodec = entry.getValue();
954           Integer streamCodecIdentifier = entry.getKey();
955           String connIdentifier = sourceIdentifier + Component.CONCAT_SEPARATOR + streamCodecIdentifier;
956 
957           SimpleEntry<String, ComponentContextPair<Stream, StreamContext>> deployBufferServerPublisher =
958             deployBufferServerPublisher(connIdentifier, streamCodec, checkpointWindowId, queueCapacity, nodi);
959           newStreams.put(sourceIdentifier, deployBufferServerPublisher.getValue());
960           node.connectOutputPort(nodi.portName, deployBufferServerPublisher.getValue().component);
961         }
962         else {
963           /*
964            * In this case we have 2 possibilities, either we have 1 inline or multiple streams.
965            * Since we cannot tell at this point, we assume that we will have multiple streams and
966            * plan accordingly. we possibly will come to this code block multiple times. We create
967            * the MuxStream only the first time and use it for subsequent calls of this block.
968            *
969            * There is also the possibility that we have a stream with multiple sinks having distinct codecs
970            */
971           ComponentContextPair<Stream, StreamContext> pair = newStreams.get(sourceIdentifier);
972           if (pair == null) {
973             /**
974              * Let's multiplex the output placed on this stream.
975              * This container itself contains more than one parties interested.
976              */
977             StreamContext context = new StreamContext(nodi.declaredStreamId);
978             context.setSourceId(sourceIdentifier);
979             context.setFinishedWindowId(checkpointWindowId);
980             Stream stream = new MuxStream();
981 
982             newStreams.put(sourceIdentifier, pair = new ComponentContextPair<Stream, StreamContext>(stream, context));
983             node.connectOutputPort(nodi.portName, stream);
984           }
985 
986           if (nodi.bufferServerHost != null) {
987             /*
988              * Although there is a node in this container interested in output placed on this stream, there
989              * seems to at least one more party interested but placed in a container other than this one.
990              */
991             for (Map.Entry<Integer, StreamCodec<?>> entry : streamCodecs.entrySet()) {
992               Integer streamCodecIdentifier = entry.getKey();
993               StreamCodec<?> streamCodec = entry.getValue();
994 
995               String connIdentifier = sourceIdentifier + Component.CONCAT_SEPARATOR + streamCodecIdentifier;
996 
997               SimpleEntry<String, ComponentContextPair<Stream, StreamContext>> deployBufferServerPublisher =
998                 deployBufferServerPublisher(connIdentifier, streamCodec, checkpointWindowId, queueCapacity, nodi);
999               newStreams.put(deployBufferServerPublisher.getKey(), deployBufferServerPublisher.getValue());
1000 
1001               String sinkIdentifier = pair.context.getSinkId();
1002               if (sinkIdentifier == null) {
1003                 pair.context.setSinkId(deployBufferServerPublisher.getKey());
1004               }
1005               else {
1006                 pair.context.setSinkId(sinkIdentifier.concat(", ").concat(deployBufferServerPublisher.getKey()));
1007               }
1008 
1009               ((Stream.MultiSinkCapableStream) pair.component).setSink(deployBufferServerPublisher.getKey(), deployBufferServerPublisher.getValue().component);
1010             }
1011           }
1012         }
1013       }
1014     }
1015 
1016     return newStreams;
1017   }

1037   @SuppressWarnings("unchecked")
1038   private void deployInputStreams(List<OperatorDeployInfo> operatorList, HashMap<String, ComponentContextPair<Stream, StreamContext>> newStreams) throws UnknownHostException
1039   {
1040     /*
1041      * collect any input operators along with their smallest window id,
1042      * those are subsequently used to setup the window generator
1043      */
1044     ArrayList<OperatorDeployInfo> inputNodes = new ArrayList<OperatorDeployInfo>();
1045     long smallestCheckpointedWindowId = Long.MAX_VALUE;
1046     //a simple map which maps the oio node to it's the node which owns the thread.
1047     Map<Integer, Integer> oioNodes = new ConcurrentHashMap<Integer, Integer>();
1048 
1049     /*
1050      * Hook up all the downstream ports. There are 2 places where we deal with more than 1
1051      * downstream ports. The first one follows immediately for WindowGenerator. The second
1052      * case is when source for the input port of some node in this container is in another
1053      * container. So we need to create the stream. We need to track this stream along with
1054      * other streams,and many such streams may exist, we hash them against buffer server
1055      * info as we did for outputs but throw in the sinkid in the mix as well.
1056      */
1057     for (OperatorDeployInfo ndi : operatorList) {
1058       if (ndi.inputs == null || ndi.inputs.isEmpty()) {
1059         /*
1060          * This has to be InputNode, so let's hook the WindowGenerator to it.
1061          * A node which does not take any input cannot exist in the DAG since it would be completely
1062          * unaware of the windows. So for that reason, AbstractInputNode allows Component.INPUT port.
1063          */
1064         inputNodes.add(ndi);
1065         /*
1066          * When we activate the window Generator, we plan to activate it only from required windowId.
1067          */
1068         ndi.checkpoint = getFinishedCheckpoint(ndi);
1069         if (ndi.checkpoint.windowId < smallestCheckpointedWindowId) {
1070           smallestCheckpointedWindowId = ndi.checkpoint.windowId;
1071         }
1072       }
1073       else {
1074         Node<?> node = nodes.get(ndi.id);
1075 
1076         for (OperatorDeployInfo.InputDeployInfo nidi : ndi.inputs) {
1077           if (nidi.streamCodecs.size() != 1) {
1078             throw new IllegalStateException("Only one input codec configuration should be present");
1079           }
1080           Map.Entry<Integer, StreamCodec<?>> entry = nidi.streamCodecs.entrySet().iterator().next();
1081           Integer streamCodecIdentifier = entry.getKey();
1082           StreamCodec<?> streamCodec = entry.getValue();
1083           String sourceIdentifier = Integer.toString(nidi.sourceNodeId).concat(Component.CONCAT_SEPARATOR).concat(nidi.sourcePortName);
1084           String sinkIdentifier = Integer.toString(ndi.id).concat(Component.CONCAT_SEPARATOR).concat(nidi.portName);
1085 
1086           int queueCapacity = getValue(PortContext.QUEUE_CAPACITY, nidi, ndi);
1087 
1088           Checkpoint checkpoint = getFinishedCheckpoint(ndi);
1089           ComponentContextPair<Stream, StreamContext> pair = streams.get(sourceIdentifier);
1090           if (pair == null) {
1091             pair = newStreams.get(sourceIdentifier);
1092           }
1093 
1094           if (pair == null) {
1095             /*
1096              * We connect to the buffer server for the input on this port.
1097              * We have already placed all the output streams for all the operators in this container.
1098              * Yet, there is no stream which can source this port so it has to come from the buffer
1099              * server, so let's make a connection to it.
1100              */
1101             assert (nidi.locality != Locality.CONTAINER_LOCAL && nidi.locality != Locality.THREAD_LOCAL);
1102 
1103             StreamContext context = new StreamContext(nidi.declaredStreamId);
1104             context.setBufferServerAddress(InetSocketAddress.createUnresolved(nidi.bufferServerHost, nidi.bufferServerPort));
1105             InetAddress inetAddress = context.getBufferServerAddress().getAddress();
1106             if (inetAddress != null && NetUtils.isLocalAddress(inetAddress)) {
1107               context.setBufferServerAddress(new InetSocketAddress(InetAddress.getByName(null), nidi.bufferServerPort));
1108             }
1109             context.put(StreamContext.BUFFER_SERVER_TOKEN, nidi.bufferServerToken);
1110             String connIdentifier = sourceIdentifier + Component.CONCAT_SEPARATOR + streamCodecIdentifier;
1111             context.setPortId(nidi.portName);
1112             context.put(StreamContext.CODEC, streamCodec);
1113             context.put(StreamContext.EVENT_LOOP, eventloop);
1114             context.setPartitions(nidi.partitionMask, nidi.partitionKeys);
1115             //context.setSourceId(sourceIdentifier);
1116             context.setSourceId(connIdentifier);
1117             context.setSinkId(sinkIdentifier);
1118             context.setFinishedWindowId(checkpoint.windowId);
1119 
1120             BufferServerSubscriber subscriber = fastPublisherSubscriber
1121               ? new FastSubscriber("tcp://".concat(nidi.bufferServerHost).concat(":").concat(String.valueOf(nidi.bufferServerPort)).concat("/").concat(connIdentifier), queueCapacity)
1122               : new BufferServerSubscriber("tcp://".concat(nidi.bufferServerHost).concat(":").concat(String.valueOf(nidi.bufferServerPort)).concat("/").concat(connIdentifier), queueCapacity);
1123             if(streamCodec instanceof StreamCodecWrapperForPersistance) {
1124               subscriber.acquireReservoirForPersistStream(sinkIdentifier, queueCapacity, streamCodec);
1125             }
1126             SweepableReservoir reservoir = subscriber.acquireReservoir(sinkIdentifier, queueCapacity);
1127             if (checkpoint.windowId >= 0) {
1128               node.connectInputPort(nidi.portName, new WindowIdActivatedReservoir(sinkIdentifier, reservoir, checkpoint.windowId));
1129             }
1130             node.connectInputPort(nidi.portName, reservoir);
1131 
1132             newStreams.put(sinkIdentifier, new ComponentContextPair<Stream, StreamContext>(subscriber, context));
1133             logger.debug("put input stream {} against key {}", subscriber, sinkIdentifier);
1134           }
1135           else {
1136             assert (nidi.locality == Locality.CONTAINER_LOCAL || nidi.locality == Locality.THREAD_LOCAL);
1137             /* we are still dealing with the MuxStream originating at the output of the source port */
1138             StreamContext inlineContext = new StreamContext(nidi.declaredStreamId);
1139             inlineContext.setSourceId(sourceIdentifier);
1140             inlineContext.setSinkId(sinkIdentifier);
1141 
1142             Stream stream;
1143             switch (nidi.locality) {
1144               case CONTAINER_LOCAL:
1145                 int outputQueueCapacity = getOutputQueueCapacity(operatorList, nidi.sourceNodeId, nidi.sourcePortName);
1146                 if (outputQueueCapacity > queueCapacity) {
1147                   queueCapacity = outputQueueCapacity;
1148                 }
1149 
1150                 stream = new InlineStream(queueCapacity);
1151                 if (checkpoint.windowId >= 0) {
1152                   node.connectInputPort(nidi.portName, new WindowIdActivatedReservoir(sinkIdentifier, (SweepableReservoir) stream, checkpoint.windowId));
1153                 }
1154                 break;
1155 
1156               case THREAD_LOCAL:
1157                 stream = new OiOStream();
1158                 oioNodes.put(ndi.id, nidi.sourceNodeId);
1159                 break;
1160 
1161               default:
1162                 throw new IllegalStateException("Locality can be either ContainerLocal or ThreadLocal");
1163             }
1164 
1165             node.connectInputPort(nidi.portName, (SweepableReservoir) stream);
1166             newStreams.put(sinkIdentifier, new ComponentContextPair<Stream, StreamContext>(stream, inlineContext));
1167 
1168             if (!(pair.component instanceof Stream.MultiSinkCapableStream)) {
1169               String originalSinkId = pair.context.getSinkId();
1170 
1171               /* we come here only if we are trying to augment the dag */
1172               StreamContext muxContext = new StreamContext(nidi.declaredStreamId);
1173               muxContext.setSourceId(sourceIdentifier);
1174               muxContext.setFinishedWindowId(checkpoint.windowId);
1175               muxContext.setSinkId(originalSinkId);
1176 
1177               MuxStream muxStream = new MuxStream();
1178               muxStream.setSink(originalSinkId, pair.component);
1179               streams.put(originalSinkId, pair);
1180 
1181               Node<?> sourceNode = nodes.get(nidi.sourceNodeId);
1182               sourceNode.connectOutputPort(nidi.sourcePortName, muxStream);
1183               newStreams.put(sourceIdentifier, pair = new ComponentContextPair<Stream, StreamContext>(muxStream, muxContext));
1184             }
1185 
1186             /* here everything should be multisink capable */
1187             if (streamCodec instanceof StreamCodecWrapperForPersistance) {
1188               PartitionAwareSinkForPersistence pas;
1189               if (nidi.partitionKeys == null) {
1190                 pas = new PartitionAwareSinkForPersistence((StreamCodecWrapperForPersistance<Object>) streamCodec, nidi.partitionMask, stream);
1191               } else {
1192                 pas = new PartitionAwareSinkForPersistence((StreamCodecWrapperForPersistance<Object>) streamCodec, nidi.partitionKeys, nidi.partitionMask, stream);
1193               }
1194               ((Stream.MultiSinkCapableStream) pair.component).setSink(sinkIdentifier, pas);
1195             } else if (nidi.partitionKeys == null || nidi.partitionKeys.isEmpty()) {
1196               ((Stream.MultiSinkCapableStream) pair.component).setSink(sinkIdentifier, stream);
1197             } else {
1198               /*
1199                * generally speaking we do not have partitions on the inline streams so the control should not
1200                * come here but if it comes, then we are ready to handle it using the partition aware streams.
1201                */
1202               PartitionAwareSink<Object> pas = new PartitionAwareSink<Object>(streamCodec == null ? nonSerializingStreamCodec : (StreamCodec<Object>) streamCodec, nidi.partitionKeys, nidi.partitionMask, stream);
1203               ((Stream.MultiSinkCapableStream) pair.component).setSink(sinkIdentifier, pas);
1204             }
1205 
1206             String streamSinkId = pair.context.getSinkId();
1207             if (streamSinkId == null) {
1208               pair.context.setSinkId(sinkIdentifier);
1209             }
1210             else {
1211               pair.context.setSinkId(streamSinkId.concat(", ").concat(sinkIdentifier));
1212             }
1213           }
1214         }
1215       }
1216     }
1217 
1218     setupOiOGroups(oioNodes);
1219 
1220     if (!inputNodes.isEmpty()) {
1221       WindowGenerator windowGenerator = setupWindowGenerator(smallestCheckpointedWindowId);
1222       for (OperatorDeployInfo ndi : inputNodes) {
1223         generators.put(ndi.id, windowGenerator);
1224 
1225         Node<?> node = nodes.get(ndi.id);
1226         SweepableReservoir reservoir = windowGenerator.acquireReservoir(String.valueOf(ndi.id), 1024);
1227         if (ndi.checkpoint.windowId >= 0) {
1228           node.connectInputPort(Node.INPUT, new WindowIdActivatedReservoir(Integer.toString(ndi.id), reservoir, ndi.checkpoint.windowId));
1229         }
1230         node.connectInputPort(Node.INPUT, reservoir);
1231       }
1232     }
1233 
1234   }

1283   {

1291     LinkedHashMap<String, PortContextPair<InputPort<?>>> newInputPorts = new LinkedHashMap<String, PortContextPair<InputPort<?>>>(inputPorts.size());
1292     for (OperatorDeployInfo.InputDeployInfo idi : ndi.inputs) {
1293       InputPort<?> port = inputPorts.get(idi.portName).component;
1294       PortContext context = new PortContext(idi.contextAttributes, node.context);
1295       newInputPorts.put(idi.portName, new PortContextPair<InputPort<?>>(port, context));
1296       port.setup(context);
1297     }
1298     inputPorts.putAll(newInputPorts);

1300     /* setup context for all the output ports */
1301     LinkedHashMap<String, PortContextPair<OutputPort<?>>> outputPorts = node.getPortMappingDescriptor().outputPorts;
1302     LinkedHashMap<String, PortContextPair<OutputPort<?>>> newOutputPorts = new LinkedHashMap<String, PortContextPair<OutputPort<?>>>(outputPorts.size());
1303     for (OperatorDeployInfo.OutputDeployInfo odi : ndi.outputs) {
1304       OutputPort<?> port = outputPorts.get(odi.portName).component;

1306       newOutputPorts.put(odi.portName, new PortContextPair<OutputPort<?>>(port, context));
1307       port.setup(context);

1309     outputPorts.putAll(newOutputPorts);

1311     logger.debug("activating {} in container {}", node, containerId);
1312     /* This introduces need for synchronization on processNodeRequest which was solved by adding deleted field in StramToNodeRequest  */
1313     processNodeRequests(false);
1314     node.activate();

1332   public synchronized void activate(final Map<Integer, OperatorDeployInfo> nodeMap, Map<String, ComponentContextPair<Stream, StreamContext>> newStreams)
1333   {
1334     for (ComponentContextPair<Stream, StreamContext> pair : newStreams.values()) {
1335       if (!(pair.component instanceof BufferServerSubscriber)) {
1336         activeStreams.put(pair.component, pair.context);
1337         pair.component.activate(pair.context);
1338         eventBus.publish(new StreamActivationEvent(pair));
1339       }
1340     }
1341 
1342     final CountDownLatch signal = new CountDownLatch(nodeMap.size());
1343     for (final OperatorDeployInfo ndi : nodeMap.values()) {
1344       /*
1345        * OiO nodes get activated with their primary nodes.
1346        */
1347       if (ndi.type == OperatorType.OIO) {
1348         continue;
1349       }
1350 
1351       final Node<?> node = nodes.get(ndi.id);
1352       final String name = new StringBuilder(Integer.toString(ndi.id))
1353           .append('/')
1354           .append(ndi.name)
1355           .append(':')
1356           .append(node.getOperator().getClass().getSimpleName())
1357           .toString();
1358       final Thread thread = new Thread(name)
1359       {
1360         @Override
1361         public void run()
1362         {
1363           HashSet<OperatorDeployInfo> setOperators = new HashSet<OperatorDeployInfo>();
1364           OperatorDeployInfo currentdi = ndi;
1365           try {
1366             /* primary operator initialization */
1367             setupNode(currentdi);
1368             setOperators.add(currentdi);
1369 
1370             /* lets go for OiO operator initialization */
1371             List<Integer> oioNodeIdList = oioGroups.get(ndi.id);
1372             if (oioNodeIdList != null) {
1373               for (Integer oioNodeId : oioNodeIdList) {
1374                 currentdi = nodeMap.get(oioNodeId);
1375                 setupNode(currentdi);
1376                 setOperators.add(currentdi);
1377               }
1378             }
1379 
1380             currentdi = null;
1381 
1382             for (int i = setOperators.size(); i-- > 0; ) {
1383               signal.countDown();
1384             }
1385 
1386             node.run(); /* this is a blocking call */
1387           }
1388           catch (Error error) {
1389             int[] operators;
1390             if (currentdi == null) {
1391               logger.error("Voluntary container termination due to an error in operator set {}.", setOperators, error);
1392               operators = new int[setOperators.size()];
1393               int i = 0;
1394               for (Iterator<OperatorDeployInfo> it = setOperators.iterator(); it.hasNext(); i++) {
1395                 operators[i] = it.next().id;
1396               }
1397             }
1398             else {
1399               logger.error("Voluntary container termination due to an error in operator {}.", currentdi, error);
1400               operators = new int[]{currentdi.id};
1401             }
1402             umbilical.reportError(containerId, operators, "Voluntary container termination due to an error. " + ExceptionUtils.getStackTrace(error));
1403             System.exit(1);
1404           }
1405           catch (Exception ex) {
1406             if (currentdi == null) {
1407               failedNodes.add(ndi.id);
1408               logger.error("Operator set {} stopped running due to an exception.", setOperators, ex);
1409               int[] operators = new int[]{ndi.id};
1410               umbilical.reportError(containerId, operators, "Stopped running due to an exception. " + ExceptionUtils.getStackTrace(ex));
1411             }
1412             else {
1413               failedNodes.add(currentdi.id);
1414               logger.error("Abandoning deployment of operator {} due to setup failure.", currentdi, ex);
1415               int[] operators = new int[]{currentdi.id};
1416               umbilical.reportError(containerId, operators, "Abandoning deployment due to setup failure. " + ExceptionUtils.getStackTrace(ex));
1417             }
1418           }
1419           finally {
1420             if (setOperators.contains(ndi)) {
1421               try {
1422                 teardownNode(ndi);
1423               }
1424               catch (Exception ex) {
1425                 failedNodes.add(ndi.id);
1426                 logger.error("Shutdown of operator {} failed due to an exception.", ndi, ex);
1427               }
1428             }
1429             else {
1430               signal.countDown();
1431             }
1432 
1433             List<Integer> oioNodeIdList = oioGroups.get(ndi.id);
1434             if (oioNodeIdList != null) {
1435               for (Integer oioNodeId : oioNodeIdList) {
1436                 OperatorDeployInfo oiodi = nodeMap.get(oioNodeId);
1437                 if (setOperators.contains(oiodi)) {
1438                   try {
1439                     teardownNode(oiodi);
1440                   }
1441                   catch (Exception ex) {
1442                     failedNodes.add(oiodi.id);
1443                     logger.error("Shutdown of operator {} failed due to an exception.", oiodi, ex);
1444                   }
1445                 }
1446                 else {
1447                   signal.countDown();
1448                 }
1449               }
1450             }
1451           }
1452         }
1453       };
1454       node.context.setThread(thread);
1455       thread.start();
1456     }
1457 
1458     /**
1459      * we need to make sure that before any of the operators gets the first message, it's activated.
1460      */
1461     try {
1462       signal.await();
1463     }
1464     catch (InterruptedException ex) {
1465       logger.debug("Activation of operators interruped.", ex);
1466     }
1467 
1468     for (ComponentContextPair<Stream, StreamContext> pair : newStreams.values()) {
1469       if (pair.component instanceof BufferServerSubscriber) {
1470         activeStreams.put(pair.component, pair.context);
1471         pair.component.activate(pair.context);
1472         eventBus.publish(new StreamActivationEvent(pair));
1473       }
1474     }
1475 
1476     for (WindowGenerator wg : generators.values()) {
1477       if (!activeGenerators.containsKey(wg)) {
1478         activeGenerators.put(wg, generators);
1479         wg.activate(null);
1480       }
1481     }
1482   }

1484   private void groupInputStreams(HashMap<String, ArrayList<String>> groupedInputStreams, OperatorDeployInfo ndi)
1485   {
1486     for (OperatorDeployInfo.InputDeployInfo nidi : ndi.inputs) {
1487       String source = Integer.toString(nidi.sourceNodeId).concat(Component.CONCAT_SEPARATOR).concat(nidi.sourcePortName);
1488 
1489       /*
1490        * if we do not want to combine multiple streams with different partitions from the
1491        * same upstream node, we could also use the partition to group the streams together.
1492        * This logic comes with the danger that the performance of the group which shares
1493        * the same stream is bounded on the higher side by the performance of the lowest
1494        * performer upstream port. May be combining the streams is not such a good thing
1495        * but let's see if we allow this as an option to the user, what they end up choosing
1496        * the most.
1497        */
1498       ArrayList<String> collection = groupedInputStreams.get(source);
1499       if (collection == null) {
1500         collection = new ArrayList<String>();
1501         groupedInputStreams.put(source, collection);
1502       }
1503       collection.add(Integer.toString(ndi.id).concat(Component.CONCAT_SEPARATOR).concat(nidi.portName));
1504     }

1511       && ndi.contextAttributes.get(OperatorContext.PROCESSING_MODE) == ProcessingMode.AT_MOST_ONCE) {

1514

1516       if (temp == null) {

1520

1522       if (temp == null) {
1523         temp = containerContext.getValue(OperatorContext.CHECKPOINT_WINDOW_COUNT);

1525       int lCheckpointWindowCount = (int) (windowCount % temp);
1526       checkpoint = new Checkpoint(WindowGenerator.getWindowId(now, firstWindowMillis, windowWidthMillis), appWindowCount, lCheckpointWindowCount);
1527       logger.debug("using {} on {} at {}", ProcessingMode.AT_MOST_ONCE, ndi.name, checkpoint);

1530       checkpoint = ndi.checkpoint;
1531       logger.debug("using {} on {} at {}", ndi.contextAttributes == null ? ProcessingMode.AT_LEAST_ONCE :

1533           ndi.contextAttributes.get(OperatorContext.PROCESSING_MODE)), ndi.name, checkpoint);
